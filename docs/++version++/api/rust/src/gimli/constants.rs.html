30// The `dw!` macro turns this:
31//
32//     dw!(DwFoo(u32) {
33//         DW_FOO_bar = 0,
34//         DW_FOO_baz = 1,
35//         DW_FOO_bang = 2,
36//     });
37//
38// into this:
39//
40//     #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
41//     pub struct DwFoo(pub u32);
42//
43//     pub const DW_FOO_bar: DwFoo = DwFoo(0);
44//     pub const DW_FOO_baz: DwFoo = DwFoo(1);
45//     pub const DW_FOO_bang: DwFoo = DwFoo(2);
46//
47//     impl DwFoo {
48//         pub fn static_string(&self) -> Option<&'static str> {
49//             ...
50//         }
51//     }
52//
53//     impl fmt::Display for DwFoo {
54//         fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
55//             ...
56//         }
57//     }
58macro_rules! dw {
59    ($(#[$meta:meta])* $struct_name:ident($struct_type:ty)
60        { $($name:ident = $val:expr),+ $(,)? }
61        $(, aliases { $($alias_name:ident = $alias_val:expr),+ $(,)? })?
62    ) => {
63        $(#[$meta])*
64        #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
65        pub struct $struct_name(pub $struct_type);
66
67        $(
68            pub const $name: $struct_name = $struct_name($val);
69        )+
70        $($(
71            pub const $alias_name: $struct_name = $struct_name($alias_val);
72        )+)*
73
74        impl $struct_name {
75            pub fn static_string(&self) -> Option<&'static str> {
76                Some(match *self {
77                    $(
78                        $name => stringify!($name),
79                    )+
80                    _ => return None,
81                })
82            }
83        }
84
85        impl fmt::Display for $struct_name {
86            fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
87                if let Some(s) = self.static_string() {
88                    f.pad(s)
89                } else {
90                    #[cfg(feature = "read")]
91                    {
92                        f.pad(&format!("Unknown {}: {}", stringify!($struct_name), self.0))
93                    }
94                    #[cfg(not(feature = "read"))]
95                    {
96                        write!(f, "Unknown {}: {}", stringify!($struct_name), self.0)
97                    }
98                }
99            }
100        }
101    };
102}

1285dw!(
1286/// Pointer encoding used by `.eh_frame`.
1287///
1288/// The four lower bits describe the
1289/// format of the pointer, the upper four bits describe how the encoding should
1290/// be applied.
1291///
1292/// Defined in `<https://refspecs.linuxfoundation.org/LSB_4.0.0/LSB-Core-generic/LSB-Core-generic/dwarfext.html>`
1293DwEhPe(u8) {
1294// Format of pointer encoding.
1295
1296// "Unsigned value is encoded using the Little Endian Base 128"
1297    DW_EH_PE_uleb128 = 0x1,
1298// "A 2 bytes unsigned value."
1299    DW_EH_PE_udata2 = 0x2,
1300// "A 4 bytes unsigned value."
1301    DW_EH_PE_udata4 = 0x3,
1302// "An 8 bytes unsigned value."
1303    DW_EH_PE_udata8 = 0x4,
1304// "Signed value is encoded using the Little Endian Base 128"
1305    DW_EH_PE_sleb128 = 0x9,
1306// "A 2 bytes signed value."
1307    DW_EH_PE_sdata2 = 0x0a,
1308// "A 4 bytes signed value."
1309    DW_EH_PE_sdata4 = 0x0b,
1310// "An 8 bytes signed value."
1311    DW_EH_PE_sdata8 = 0x0c,
1312
1313// How the pointer encoding should be applied.
1314
1315// `DW_EH_PE_pcrel` pointers are relative to their own location.
1316    DW_EH_PE_pcrel = 0x10,
1317// "Value is relative to the beginning of the .text section."
1318    DW_EH_PE_textrel = 0x20,
1319// "Value is relative to the beginning of the .got or .eh_frame_hdr section."
1320    DW_EH_PE_datarel = 0x30,
1321// "Value is relative to the beginning of the function."
1322    DW_EH_PE_funcrel = 0x40,
1323// "Value is aligned to an address unit sized boundary."
1324    DW_EH_PE_aligned = 0x50,
1325
1326// This bit can be set for any of the above encoding applications. When set,
1327// the encoded value is the address of the real pointer result, not the
1328// pointer result itself.
1329//
1330// This isn't defined in the DWARF or the `.eh_frame` standards, but is
1331// generated by both GNU/Linux and macOS tooling.
1332    DW_EH_PE_indirect = 0x80,
1333
1334// These constants apply to both the lower and upper bits.
1335
1336// "The Value is a literal pointer whose size is determined by the
1337// architecture."
1338    DW_EH_PE_absptr = 0x0,
1339// The absence of a pointer and encoding.
1340    DW_EH_PE_omit = 0xff,
1341});

1343const DW_EH_PE_FORMAT_MASK: u8 = 0b0000_1111;
1344
1345// Ignores indirection bit.
1346const DW_EH_PE_APPLICATION_MASK: u8 = 0b0111_0000;
1347
1348impl ops::BitOr for DwEhPe {
1349    type Output = DwEhPe;
1350
1351    fn bitor(self, rhs: DwEhPe) -> DwEhPe {
1352        DwEhPe(self.0 | rhs.0)
1353    }
1354}
1355
1356impl DwEhPe {
1357    /// Get the pointer encoding's format.
1358    #[inline]
1359    pub fn format(self) -> DwEhPe {
1360        DwEhPe(self.0 & DW_EH_PE_FORMAT_MASK)
1361    }
1362
1363    /// Get the pointer encoding's application.
1364    #[inline]
1365    pub fn application(self) -> DwEhPe {
1366        DwEhPe(self.0 & DW_EH_PE_APPLICATION_MASK)
1367    }
1368
1369    /// Is this encoding the absent pointer encoding?
1370    #[inline]
1371    pub fn is_absent(self) -> bool {
1372        self == DW_EH_PE_omit
1373    }
1374
1375    /// Is this coding indirect? If so, its encoded value is the address of the
1376    /// real pointer result, not the pointer result itself.
1377    #[inline]
1378    pub fn is_indirect(self) -> bool {
1379        self.0 & DW_EH_PE_indirect.0 != 0
1380    }
1381
1382    /// Is this a known, valid pointer encoding?
1383    pub fn is_valid_encoding(self) -> bool {
1384        if self.is_absent() {
1385            return true;
1386        }
1387
1388        match self.format() {
1389            DW_EH_PE_absptr | DW_EH_PE_uleb128 | DW_EH_PE_udata2 | DW_EH_PE_udata4
1390            | DW_EH_PE_udata8 | DW_EH_PE_sleb128 | DW_EH_PE_sdata2 | DW_EH_PE_sdata4
1391            | DW_EH_PE_sdata8 => {}
1392            _ => return false,
1393        }
1394
1395        match self.application() {
1396            DW_EH_PE_absptr | DW_EH_PE_pcrel | DW_EH_PE_textrel | DW_EH_PE_datarel
1397            | DW_EH_PE_funcrel | DW_EH_PE_aligned => {}
1398            _ => return false,
1399        }
1400
1401        true
1402    }
1403}
