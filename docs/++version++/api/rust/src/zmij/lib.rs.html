7//! <br>
8//!
9//! A double-to-string conversion algorithm based on [Schubfach] and [yy].
10//!
11//! This Rust implementation is a line-by-line port of Victor Zverovich's
12//! implementation in C++, <https://github.com/vitaut/zmij>.
13//!
14//! [Schubfach]: https://fmt.dev/papers/Schubfach4.pdf
15//! [yy]: https://github.com/ibireme/c_numconv_benchmark/blob/master/vendor/yy_double/yy_double.c
16//!
17//! <br>
18//!
19//! # Example
20//!
21//! ```
22//! fn main() {
23//!     let mut buffer = zmij::Buffer::new();
24//!     let printed = buffer.format(1.234);
25//!     assert_eq!(printed, "1.234");
26//! }
27//! ```
28//!
29//! <br>
30//!
31//! ## Performance
32//!
33//! The [dtoa-benchmark] compares this library and other Rust floating point
34//! formatting implementations across a range of precisions. The vertical axis
35//! in this chart shows nanoseconds taken by a single execution of
36//! `zmij::Buffer::new().format_finite(value)` so a lower result indicates a
37//! faster library.
38//!
39//! [dtoa-benchmark]: https://github.com/dtolnay/dtoa-benchmark
40//!

711// Computes 128-bit result of multiplication of two 64-bit unsigned integers.
712#[cfg_attr(feature = "no-panic", no_panic)]
713fn umul128(x: u64, y: u64) -> u128 {
714    u128::from(x) * u128::from(y)
715}
716
717#[cfg_attr(feature = "no-panic", no_panic)]
718fn umul192_upper128(x_hi: u64, x_lo: u64, y: u64) -> uint128 {
719    let p = umul128(x_hi, y);
720    let lo = (p as u64).wrapping_add((umul128(x_lo, y) >> 64) as u64);
721    uint128 {
722        hi: (p >> 64) as u64 + u64::from(lo < p as u64),
723        lo,
724    }
725}

727// Computes upper 64 bits of multiplication of x and y, discards the least
728// significant bit and rounds to odd, where x = uint128_t(x_hi << 64) | x_lo.
729#[cfg_attr(feature = "no-panic", no_panic)]
730fn umul_upper_inexact_to_odd<UInt>(x_hi: u64, x_lo: u64, y: UInt) -> UInt
731where
732    UInt: traits::UInt,
733{
734    let num_bits = mem::size_of::<UInt>() * 8;
735    if num_bits == 64 {
736        let uint128 { hi, lo } = umul192_upper128(x_hi, x_lo, y.into());
737        UInt::truncate(hi | u64::from((lo >> 1) != 0))
738    } else {
739        let result = (umul128(x_hi, y.into()) >> 32) as u64;
740        UInt::enlarge((result >> 32) as u32 | u32::from((result as u32 >> 1) != 0))
741    }
742}

744// Returns {value / 100, value % 100} correct for values of up to 4 digits.
745fn divmod100(value: u32) -> (u32, u32) {
746    debug_assert!(value < 10_000);
747    const EXP: u32 = 19; // 19 is faster or equal to 12 even for 3 digits.
748    const SIG: u32 = (1 << EXP) / 100 + 1;
749    let div = (value * SIG) >> EXP; // value / 100
750    (div, value - div * 100)
751}

753#[cfg_attr(feature = "no-panic", no_panic)]
754fn count_trailing_nonzeros(x: u64) -> usize {
755    // We count the number of bytes until there are only zeros left.
756    // The code is equivalent to
757    //    8 - x.leading_zeros() / 8
758    // but if the BSR instruction is emitted (as gcc on x64 does with default
759    // settings), subtracting the constant before dividing allows the compiler
760    // to combine it with the subtraction which it inserts due to BSR counting
761    // in the opposite direction.
762    //
763    // Additionally, the BSR instruction requires a zero check. Since the high
764    // bit is unused we can avoid the zero check by shifting the datum left by
765    // one and inserting a sentinel bit at the end. This can be faster than the
766    // automatically inserted range check.
767    (70 - ((x.to_le() << 1) | 1).leading_zeros()) as usize / 8
768}

795#[cfg_attr(feature = "no-panic", no_panic)]
796fn to_bcd8(abcdefgh: u64) -> u64 {
797    // An optimization from Xiang JunBo.
798    // Three steps BCD. Base 10000 -> base 100 -> base 10.
799    // div and mod are evaluated simultaneously as, e.g.
800    //   (abcdefgh / 10000) << 32 + (abcdefgh % 10000)
801    //      == abcdefgh + (2^32 - 10000) * (abcdefgh / 10000)))
802    // where the division on the RHS is implemented by the usual multiply + shift
803    // trick and the fractional bits are masked away.
804    let abcd_efgh = abcdefgh + (0x100000000 - 10000) * ((abcdefgh * 0x68db8bb) >> 40);
805    let ab_cd_ef_gh = abcd_efgh + (0x10000 - 100) * (((abcd_efgh * 0x147b) >> 19) & 0x7f0000007f);
806    let a_b_c_d_e_f_g_h =
807        ab_cd_ef_gh + (0x100 - 10) * (((ab_cd_ef_gh * 0x67) >> 10) & 0xf000f000f000f);
808    a_b_c_d_e_f_g_h.to_be()
809}

913            let high_10000: int32x2_t = mem::transmute::<uint32x2_t, int32x2_t>(vshr_n_u32(
914                mem::transmute::<int32x2_t, uint32x2_t>(vqdmulh_n_s32(
915                    mem::transmute::<uint64x1_t, int32x2_t>(hundredmillions),
916                    mem::transmute::<int32x4_t, [i32; 4]>(c.multipliers32)[0],
917                )),
918                9,
919            ));
920            let tenthousands: int32x2_t = vmla_n_s32(
921                mem::transmute::<uint64x1_t, int32x2_t>(hundredmillions),
922                high_10000,
923                mem::transmute::<int32x4_t, [i32; 4]>(c.multipliers32)[1],
924            );
925
926            let mut extended: int32x4_t = mem::transmute::<uint32x4_t, int32x4_t>(vshll_n_u16(
927                mem::transmute::<int32x2_t, uint16x4_t>(tenthousands),
928                0,
929            ));
930
931            // Compiler barrier, or clang breaks the subsequent MLA into UADDW +
932            // MUL.
933            asm!("/*{:v}*/", inout(vreg) extended);
934
935            let high_100: int32x4_t = vqdmulhq_n_s32(
936                extended,
937                mem::transmute::<int32x4_t, [i32; 4]>(c.multipliers32)[2],
938            );
939            let hundreds: int32x4_t = vmlaq_n_s32(
940                extended,
941                high_100,
942                mem::transmute::<int32x4_t, [i32; 4]>(c.multipliers32)[3],
943            );
944            let high_10: int16x8_t = vqdmulhq_n_s16(
945                mem::transmute::<int32x4_t, int16x8_t>(hundreds),
946                mem::transmute::<int16x8_t, [i16; 8]>(c.multipliers16)[0],
947            );
948            let digits: int16x8_t =
949                mem::transmute::<uint8x16_t, int16x8_t>(vrev64q_u8(mem::transmute::<
950                    int16x8_t,
951                    uint8x16_t,
952                >(vmlaq_n_s16(
953                    mem::transmute::<int32x4_t, int16x8_t>(hundreds),
954                    high_10,
955                    mem::transmute::<int16x8_t, [i16; 8]>(c.multipliers16)[1],
956                ))));
957            let ascii: int16x8_t = mem::transmute::<uint16x8_t, int16x8_t>(vaddq_u16(
958                mem::transmute::<int16x8_t, uint16x8_t>(digits),
959                mem::transmute::<int8x16_t, uint16x8_t>(vdupq_n_s8(b'0' as i8)),
960            ));
961
962            buffer.cast::<int16x8_t>().write_unaligned(ascii);
963
964            let is_zero: uint16x8_t =
965                mem::transmute::<uint8x16_t, uint16x8_t>(vceqzq_u8(mem::transmute::<
966                    int16x8_t,
967                    uint8x16_t,
968                >(digits)));
969            let zeros: u64 = vget_lane_u64(vreinterpret_u64_u8(vshrn_n_u16(is_zero, 4)), 0);
970
971            buffer.add(16 - ((!zeros).leading_zeros() as usize >> 2))
972        }
973    }
974}

996fn normalize<UInt>(mut dec: fp, subnormal: bool) -> fp
997where
998    UInt: traits::UInt,
999{
1000    if !subnormal {
1001        return dec;
1002    }
1003    let num_bits = mem::size_of::<UInt>() * 8;
1004    while dec.sig
1005        < if num_bits == 64 {
1006            10_000_000_000_000_000
1007        } else {
1008            100_000_000
1009        }
1010    {
1011        dec.sig *= 10;
1012        dec.exp -= 1;
1013    }
1014    dec
1015}

1017// Computes the decimal exponent as floor(log10(2**bin_exp)) if regular or
1018// floor(log10(3/4 * 2**bin_exp)) otherwise, without branching.
1019const fn compute_dec_exp(bin_exp: i32, regular: bool) -> i32 {
1020    debug_assert!(bin_exp >= -1334 && bin_exp <= 2620);
1021    // log10_3_over_4_sig = round(log10(3/4) * 2**log10_2_exp)
1022    const LOG10_3_OVER_4_SIG: i32 = -131_008;
1023    // log10_2_sig = round(log10(2) * 2**log10_2_exp)
1024    const LOG10_2_SIG: i32 = 315_653;
1025    const LOG10_2_EXP: i32 = 20;
1026    (bin_exp * LOG10_2_SIG + !regular as i32 * LOG10_3_OVER_4_SIG) >> LOG10_2_EXP
1027}

1029// Computes a shift so that, after scaling by a power of 10, the intermediate
1030// result always has a fixed 128-bit fractional part (for double).
1031//
1032// Different binary exponents can map to the same decimal exponent, but place
1033// the decimal point at different bit positions. The shift compensates for this.
1034//
1035// For example, both 3 * 2**59 and 3 * 2**60 have dec_exp = 2, but dividing by
1036// 10^dec_exp puts the decimal point in different bit positions:
1037//   3 * 2**59 / 100 = 1.72...e+16  (needs shift = 1 + 1)
1038//   3 * 2**60 / 100 = 3.45...e+16  (needs shift = 2 + 1)
1039const fn compute_exp_shift(bin_exp: i32, dec_exp: i32) -> i32 {
1040    // log2_pow10_sig = round(log2(10) * 2**log2_pow10_exp) + 1
1041    const LOG2_POW10_SIG: i32 = 217_707;
1042    const LOG2_POW10_EXP: i32 = 16;
1043    debug_assert!(dec_exp >= -350 && dec_exp <= 350);
1044    // pow10_bin_exp = floor(log2(10**-dec_exp))
1045    let pow10_bin_exp = (-dec_exp * LOG2_POW10_SIG) >> LOG2_POW10_EXP;
1046    // pow10 = ((pow10_hi << 64) | pow10_lo) * 2**(pow10_bin_exp - 127)
1047
1048    // Shift to ensure the intermediate result of multiplying by a power of 10
1049    // has a fixed 128-bit fractional part. For example, 3 * 2**59 and 3 * 2**60
1050    // both have dec_exp = 2 and dividing them by 10**dec_exp would have the
1051    // decimal point in different (bit) positions without the shift:
1052    //   3 * 2**59 / 100 = 1.72...e+16 (exp_shift = 1 + 1)
1053    //   3 * 2**60 / 100 = 3.45...e+16 (exp_shift = 2 + 1)
1054    bin_exp + pow10_bin_exp + 1
1055}

1057// Converts a binary FP number bin_sig * 2**bin_exp to the shortest decimal
1058// representation.
1059#[cfg_attr(feature = "no-panic", no_panic)]
1060fn to_decimal<UInt>(bin_sig: UInt, bin_exp: i32, regular: bool, subnormal: bool) -> fp
1061where
1062    UInt: traits::UInt,
1063{
1064    let dec_exp = compute_dec_exp(bin_exp, regular);
1065    let exp_shift = compute_exp_shift(bin_exp, dec_exp);
1066    let (mut pow10_hi, mut pow10_lo) =
1067        *unsafe { POW10_SIGNIFICANDS.get_unchecked((-dec_exp - DEC_EXP_MIN) as usize) };
1068
1069    let num_bits = mem::size_of::<UInt>() as i32 * 8;
1070    if regular && !subnormal {
1071        let integral; // integral part of bin_sig * pow10
1072        let fractional; // fractional part of bin_sig * pow10
1073        if num_bits == 64 {
1074            let result = umul192_upper128(pow10_hi, pow10_lo, (bin_sig << exp_shift).into());
1075            integral = UInt::truncate(result.hi);
1076            fractional = result.lo;
1077        } else {
1078            let result = umul128(pow10_hi, (bin_sig << exp_shift).into());
1079            integral = UInt::truncate((result >> 64) as u64);
1080            fractional = result as u64;
1081        }
1082        let digit = integral.into() % 10;
1083
1084        // Switch to a fixed-point representation with the least significant
1085        // integral digit in the upper bits and fractional digits in the lower
1086        // bits.
1087        let num_integral_bits = if num_bits == 64 { 4 } else { 32 };
1088        let num_fractional_bits = 64 - num_integral_bits;
1089        let ten = 10u64 << num_fractional_bits;
1090        // Fixed-point remainder of the scaled significand modulo 10.
1091        let scaled_sig_mod10 = (digit << num_fractional_bits) | (fractional >> num_integral_bits);
1092
1093        // scaled_half_ulp = 0.5 * pow10 in the fixed-point format.
1094        // dec_exp is chosen so that 10**dec_exp <= 2**bin_exp < 10**(dec_exp + 1).
1095        // Since 1ulp == 2**bin_exp it will be in the range [1, 10) after scaling
1096        // by 10**dec_exp. Add 1 to combine the shift with division by two.
1097        let scaled_half_ulp = pow10_hi >> (num_integral_bits - exp_shift + 1);
1098        let upper = scaled_sig_mod10 + scaled_half_ulp;
1099        const HALF_ULP: u64 = 1 << 63;
1100
1101        // An optimization from yy by Yaoyuan Guo:
1102        if {
1103            // Exact half-ulp tie when rounding to nearest integer.
1104            fractional != HALF_ULP &&
1105            // Exact half-ulp tie when rounding to nearest 10.
1106            scaled_sig_mod10 != scaled_half_ulp &&
1107            // Near-boundary case for rounding to nearest 10.
1108            ten.wrapping_sub(upper) > 1
1109        } {
1110            let round_up = upper >= ten;
1111            let shorter = integral.into() - digit + u64::from(round_up) * 10;
1112            let longer = integral.into() + u64::from(fractional >= HALF_ULP);
1113            let use_shorter = scaled_sig_mod10 <= scaled_half_ulp || round_up;
1114            return fp {
1115                #[cfg(zmij_no_select_unpredictable)]
1116                sig: if use_shorter { shorter } else { longer },
1117                #[cfg(not(zmij_no_select_unpredictable))]
1118                sig: hint::select_unpredictable(use_shorter, shorter, longer),
1119                exp: dec_exp,
1120            };
1121        }
1122    }
1123
1124    // Fallback to Schubfach to guarantee correctness in boundary cases and
1125    // switch to strict overestimates of powers of 10.
1126    if num_bits == 64 {
1127        pow10_lo += 1;
1128    } else {
1129        pow10_hi += 1;
1130    }
1131
1132    // Shift the significand so that boundaries are integer.
1133    const BOUND_SHIFT: u32 = 2;
1134    let bin_sig_shifted = bin_sig << BOUND_SHIFT;
1135
1136    // Compute the estimates of lower and upper bounds of the rounding interval
1137    // by multiplying them by the power of 10 and applying modified rounding.
1138    let lsb = bin_sig & UInt::from(1);
1139    let lower = (bin_sig_shifted - (UInt::from(regular) + UInt::from(1))) << exp_shift;
1140    let lower = umul_upper_inexact_to_odd(pow10_hi, pow10_lo, lower) + lsb;
1141    let upper = (bin_sig_shifted + UInt::from(2)) << exp_shift;
1142    let upper = umul_upper_inexact_to_odd(pow10_hi, pow10_lo, upper) - lsb;
1143
1144    // The idea of using a single shorter candidate is by Cassio Neri.
1145    // It is less or equal to the upper bound by construction.
1146    let shorter = UInt::from(10) * ((upper >> BOUND_SHIFT) / UInt::from(10));
1147    if (shorter << BOUND_SHIFT) >= lower {
1148        return normalize::<UInt>(
1149            fp {
1150                sig: shorter.into(),
1151                exp: dec_exp,
1152            },
1153            subnormal,
1154        );
1155    }
1156
1157    let scaled_sig = umul_upper_inexact_to_odd(pow10_hi, pow10_lo, bin_sig_shifted << exp_shift);
1158    let dec_sig_below = scaled_sig >> BOUND_SHIFT;
1159    let dec_sig_above = dec_sig_below + UInt::from(1);
1160
1161    // Pick the closest of dec_sig_below and dec_sig_above and check if it's in
1162    // the rounding interval.
1163    let cmp = scaled_sig
1164        .wrapping_sub((dec_sig_below + dec_sig_above) << 1)
1165        .to_signed();
1166    let below_closer = cmp < UInt::from(0).to_signed()
1167        || (cmp == UInt::from(0).to_signed() && (dec_sig_below & UInt::from(1)) == UInt::from(0));
1168    let below_in = (dec_sig_below << BOUND_SHIFT) >= lower;
1169    let dec_sig = if below_closer & below_in {
1170        dec_sig_below
1171    } else {
1172        dec_sig_above
1173    };
1174    normalize::<UInt>(
1175        fp {
1176            sig: dec_sig.into(),
1177            exp: dec_exp,
1178        },
1179        subnormal,
1180    )
1181}

1183/// Writes the shortest correctly rounded decimal representation of `value` to
1184/// `buffer`. `buffer` should point to a buffer of size `buffer_size` or larger.
1185#[cfg_attr(feature = "no-panic", no_panic)]
1186unsafe fn write<Float>(value: Float, mut buffer: *mut u8) -> *mut u8
1187where
1188    Float: traits::Float,
1189{
1190    let num_bits = mem::size_of::<Float>() as i32 * 8;
1191    let bits = value.to_bits();
1192
1193    unsafe {
1194        *buffer = b'-';
1195        buffer = buffer.add((bits >> (num_bits - 1)).into() as usize);
1196    }
1197
1198    let num_sig_bits = Float::MANTISSA_DIGITS as i32 - 1;
1199    let implicit_bit = Float::UInt::from(1) << num_sig_bits;
1200    let mut bin_sig = bits & (implicit_bit - Float::UInt::from(1)); // binary significand
1201    let mut regular = bin_sig != Float::UInt::from(0);
1202
1203    let num_exp_bits = num_bits - num_sig_bits - 1;
1204    let exp_mask = (1 << num_exp_bits) - 1;
1205    let exp_bias = (1 << (num_exp_bits - 1)) - 1;
1206    let mut bin_exp = (bits >> num_sig_bits).into() as i32 & exp_mask; // binary exponent
1207
1208    let mut subnormal = false;
1209    if bin_exp == 0 {
1210        if bin_sig == Float::UInt::from(0) {
1211            return unsafe {
1212                *buffer = b'0';
1213                *buffer.add(1) = b'.';
1214                *buffer.add(2) = b'0';
1215                buffer.add(3)
1216            };
1217        }
1218        // Handle subnormals.
1219        // Setting regular is not redundant: it avoids extra data dependencies
1220        // and register pressure on the hot path (measurable perf impact).
1221        regular = true;
1222        bin_sig |= implicit_bit;
1223        bin_exp = 1;
1224        subnormal = true;
1225    }
1226    bin_sig ^= implicit_bit;
1227    bin_exp -= num_sig_bits + exp_bias;
1228
1229    let fp {
1230        sig: mut dec_sig,
1231        exp: mut dec_exp,
1232    } = to_decimal(bin_sig, bin_exp, regular, subnormal);
1233    let num_digits = Float::MAX_DIGITS10 as i32 - 2;
1234    let end = if num_bits == 64 {
1235        dec_exp += num_digits + i32::from(dec_sig >= 10_000_000_000_000_000);
1236        unsafe { write_significand17(buffer.add(1), dec_sig) }
1237    } else {
1238        if dec_sig < 10_000_000 {
1239            dec_sig *= 10;
1240            dec_exp -= 1;
1241        }
1242        dec_exp += num_digits + i32::from(dec_sig >= 100_000_000);
1243        unsafe { write_significand9(buffer.add(1), dec_sig as u32) }
1244    };
1245
1246    let length = unsafe { end.offset_from(buffer.add(1)) } as usize;
1247
1248    if num_bits == 32 && (-6..=12).contains(&dec_exp)
1249        || num_bits == 64 && (-5..=15).contains(&dec_exp)
1250    {
1251        if length as i32 - 1 <= dec_exp {
1252            // 1234e7 -> 12340000000.0
1253            return unsafe {
1254                ptr::copy(buffer.add(1), buffer, length);
1255                ptr::write_bytes(buffer.add(length), b'0', dec_exp as usize + 3 - length);
1256                *buffer.add(dec_exp as usize + 1) = b'.';
1257                buffer.add(dec_exp as usize + 3)
1258            };
1259        } else if 0 <= dec_exp {
1260            // 1234e-2 -> 12.34
1261            return unsafe {
1262                ptr::copy(buffer.add(1), buffer, dec_exp as usize + 1);
1263                *buffer.add(dec_exp as usize + 1) = b'.';
1264                buffer.add(length + 1)
1265            };
1266        } else {
1267            // 1234e-6 -> 0.001234
1268            return unsafe {
1269                ptr::copy(buffer.add(1), buffer.add((1 - dec_exp) as usize), length);
1270                ptr::write_bytes(buffer, b'0', (1 - dec_exp) as usize);
1271                *buffer.add(1) = b'.';
1272                buffer.add((1 - dec_exp) as usize + length)
1273            };
1274        }
1275    }
1276
1277    unsafe {
1278        // 1234e30 -> 1.234e33
1279        *buffer = *buffer.add(1);
1280        *buffer.add(1) = b'.';
1281        buffer = buffer.add(length + usize::from(length > 1));
1282        *buffer = b'e';
1283        buffer = buffer.add(1);
1284    }
1285    let sign_ptr = buffer;
1286    let sign = b'-'.wrapping_add(u8::from(dec_exp >= 0) * b'+'.wrapping_sub(b'-'));
1287    let mask = i32::from(dec_exp >= 0) - 1;
1288    dec_exp = (dec_exp + mask) ^ mask; // absolute value
1289    unsafe {
1290        buffer = buffer.add(usize::from(dec_exp >= 10));
1291    }
1292    let (a, bb) = divmod100(dec_exp as u32);
1293    unsafe {
1294        *buffer = b'0' + a as u8;
1295        buffer = buffer.add(usize::from(dec_exp >= 100));
1296        buffer.cast::<u16>().write_unaligned(*digits2(bb as usize));
1297        *sign_ptr = sign;
1298        buffer.add(2)
1299    }
1300}

1302/// Safe API for formatting floating point numbers to text.
1303///
1304/// ## Example
1305///
1306/// ```
1307/// let mut buffer = zmij::Buffer::new();
1308/// let printed = buffer.format_finite(1.234);
1309/// assert_eq!(printed, "1.234");
1310/// ```
1311pub struct Buffer {
1312    bytes: [MaybeUninit<u8>; BUFFER_SIZE],
1313}
1314
1315impl Buffer {
1316    /// This is a cheap operation; you don't need to worry about reusing buffers
1317    /// for efficiency.
1318    #[inline]
1319    #[cfg_attr(feature = "no-panic", no_panic)]
1320    pub fn new() -> Self {
1321        let bytes = [MaybeUninit::<u8>::uninit(); BUFFER_SIZE];
1322        Buffer { bytes }
1323    }
1324
1325    /// Print a floating point number into this buffer and return a reference to
1326    /// its string representation within the buffer.
1327    ///
1328    /// # Special cases
1329    ///
1330    /// This function formats NaN as the string "NaN", positive infinity as
1331    /// "inf", and negative infinity as "-inf" to match std::fmt.
1332    ///
1333    /// If your input is known to be finite, you may get better performance by
1334    /// calling the `format_finite` method instead of `format` to avoid the
1335    /// checks for special cases.
1336    #[cfg_attr(feature = "no-panic", no_panic)]
1337    pub fn format<F: Float>(&mut self, f: F) -> &str {
1338        if f.is_nonfinite() {
1339            f.format_nonfinite()
1340        } else {
1341            self.format_finite(f)
1342        }
1343    }
1344
1345    /// Print a floating point number into this buffer and return a reference to
1346    /// its string representation within the buffer.
1347    ///
1348    /// # Special cases
1349    ///
1350    /// This function **does not** check for NaN or infinity. If the input
1351    /// number is not a finite float, the printed representation will be some
1352    /// correctly formatted but unspecified numerical value.
1353    ///
1354    /// Please check [`is_finite`] yourself before calling this function, or
1355    /// check [`is_nan`] and [`is_infinite`] and handle those cases yourself.
1356    ///
1357    /// [`is_finite`]: f64::is_finite
1358    /// [`is_nan`]: f64::is_nan
1359    /// [`is_infinite`]: f64::is_infinite
1360    #[cfg_attr(feature = "no-panic", no_panic)]
1361    pub fn format_finite<F: Float>(&mut self, f: F) -> &str {
1362        unsafe {
1363            let end = f.write_to_zmij_buffer(self.bytes.as_mut_ptr().cast::<u8>());
1364            let len = end.offset_from(self.bytes.as_ptr().cast::<u8>()) as usize;
1365            let slice = slice::from_raw_parts(self.bytes.as_ptr().cast::<u8>(), len);
1366            str::from_utf8_unchecked(slice)
1367        }
1368    }
1369}
