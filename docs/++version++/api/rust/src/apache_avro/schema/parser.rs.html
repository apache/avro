34pub(crate) struct Parser {

36    /// Used to resolve cyclic references, i.e. when a
37    /// field's type is a reference to its record's type

40    /// Used to avoid parsing the same schema twice

62    /// Create a `Schema` from a string representing a JSON Avro schema.
63    pub(super) fn parse_str(&mut self, input: &str) -> AvroResult<Schema> {
64        let value = serde_json::from_str(input).map_err(Details::ParseSchemaJson)?;
65        self.parse(&value, &None)
66    }
67
68    /// Create an array of `Schema`s from an iterator of JSON Avro schemas.
69    ///
70    /// It is allowed that the schemas have cross-dependencies; these will be resolved during parsing.
71    pub(super) fn parse_list(&mut self) -> AvroResult<Vec<Schema>> {
72        self.parse_input_schemas()?;
73
74        let mut parsed_schemas = Vec::with_capacity(self.parsed_schemas.len());
75        for name in self.input_order.drain(0..) {
76            let parsed = self
77                .parsed_schemas
78                .remove(&name)
79                .expect("One of the input schemas was unexpectedly not parsed");
80            parsed_schemas.push(parsed);
81        }
82        Ok(parsed_schemas)
83    }
84
85    /// Convert the input schemas to `parsed_schemas`.
86    pub(super) fn parse_input_schemas(&mut self) -> Result<(), Error> {
87        while !self.input_schemas.is_empty() {
88            let next_name = self
89                .input_schemas
90                .keys()
91                .next()
92                .expect("Input schemas unexpectedly empty")
93                .to_owned();
94            let (name, value) = self
95                .input_schemas
96                .remove_entry(&next_name)
97                .expect("Key unexpectedly missing");
98            let parsed = self.parse(&value, &None)?;
99            self.parsed_schemas
100                .insert(self.get_schema_type_name(name, value), parsed);
101        }
102        Ok(())
103    }

105    /// Create a `Schema` from a `serde_json::Value` representing a JSON Avro schema.
106    pub(super) fn parse(
107        &mut self,
108        value: &Value,
109        enclosing_namespace: &Namespace,
110    ) -> AvroResult<Schema> {
111        match *value {
112            Value::String(ref t) => self.parse_known_schema(t.as_str(), enclosing_namespace),
113            Value::Object(ref data) => {
114                self.parse_complex(data, enclosing_namespace, RecordSchemaParseLocation::Root)
115            }
116            Value::Array(ref data) => self.parse_union(data, enclosing_namespace),
117            _ => Err(Details::ParseSchemaFromValidJson.into()),
118        }
119    }

121    /// Parse a string as a primitive type or reference to `parsed_schemas`.
122    fn parse_known_schema(
123        &mut self,
124        name: &str,
125        enclosing_namespace: &Namespace,
126    ) -> AvroResult<Schema> {
127        match name {
128            "null" => Ok(Schema::Null),
129            "boolean" => Ok(Schema::Boolean),
130            "int" => Ok(Schema::Int),
131            "long" => Ok(Schema::Long),
132            "double" => Ok(Schema::Double),
133            "float" => Ok(Schema::Float),
134            "bytes" => Ok(Schema::Bytes),
135            "string" => Ok(Schema::String),
136            _ => self.fetch_schema_ref(name, enclosing_namespace),
137        }
138    }

140    /// Given a name, tries to retrieve the parsed schema from `parsed_schemas`.
141    ///
142    /// If a parsed schema is not found, it checks if a currently resolving
143    /// schema with that name exists.
144    /// If a resolving schema is not found, it checks if a JSON with that name exists
145    /// in `input_schemas` and then parses it (removing it from `input_schemas`)
146    /// and adds the parsed schema to `parsed_schemas`.
147    ///
148    /// This method allows schemas definitions that depend on other types to
149    /// parse their dependencies (or look them up if already parsed).
150    pub(super) fn fetch_schema_ref(
151        &mut self,
152        name: &str,
153        enclosing_namespace: &Namespace,
154    ) -> AvroResult<Schema> {
155        fn get_schema_ref(parsed: &Schema) -> Schema {
156            match parsed {
157                &Schema::Record(RecordSchema { ref name, .. })
158                | &Schema::Enum(EnumSchema { ref name, .. })
159                | &Schema::Fixed(FixedSchema { ref name, .. }) => {
160                    Schema::Ref { name: name.clone() }
161                }
162                _ => parsed.clone(),
163            }
164        }
165
166        let name = Name::new(name)?;
167        let fully_qualified_name = name.fully_qualified_name(enclosing_namespace);
168
169        if self.parsed_schemas.contains_key(&fully_qualified_name) {
170            return Ok(Schema::Ref {
171                name: fully_qualified_name,
172            });
173        }
174        if let Some(resolving_schema) = self.resolving_schemas.get(&fully_qualified_name) {
175            return Ok(resolving_schema.clone());
176        }
177
178        // For good error reporting we add this check
179        match name.name.as_str() {
180            "record" | "enum" | "fixed" => {
181                return Err(Details::InvalidSchemaRecord(name.to_string()).into());
182            }
183            _ => (),
184        }
185
186        let value = self
187            .input_schemas
188            .remove(&fully_qualified_name)
189            // TODO make a better descriptive error message here that conveys that a named schema cannot be found
190            .ok_or_else(|| {
191                let full_name = fully_qualified_name.fullname(None);
192                if full_name == "bool" {
193                    Details::ParsePrimitiveSimilar(full_name, "boolean")
194                } else {
195                    Details::ParsePrimitive(full_name)
196                }
197            })?;
198
199        // parsing a full schema from inside another schema. Other full schema will not inherit namespace
200        let parsed = self.parse(&value, &None)?;
201        self.parsed_schemas
202            .insert(self.get_schema_type_name(name, value), parsed.clone());
203
204        Ok(get_schema_ref(&parsed))
205    }

207    fn get_decimal_integer(
208        &self,
209        complex: &Map<String, Value>,
210        key: &'static str,
211    ) -> AvroResult<DecimalMetadata> {
212        match complex.get(key) {
213            Some(Value::Number(value)) => self.parse_json_integer_for_decimal(value),
214            None => {
215                if key == "scale" {
216                    Ok(0)
217                } else {
218                    Err(Details::GetDecimalMetadataFromJson(key).into())
219                }
220            }
221            Some(value) => Err(Details::GetDecimalMetadataValueFromJson {
222                key: key.into(),
223                value: value.clone(),
224            }
225            .into()),
226        }
227    }
228
229    fn parse_precision_and_scale(
230        &self,
231        complex: &Map<String, Value>,
232    ) -> AvroResult<(Precision, Scale)> {
233        let precision = self.get_decimal_integer(complex, "precision")?;
234        let scale = self.get_decimal_integer(complex, "scale")?;
235
236        if precision < 1 {
237            return Err(Details::DecimalPrecisionMuBePositive { precision }.into());
238        }
239
240        if precision < scale {
241            Err(Details::DecimalPrecisionLessThanScale { precision, scale }.into())
242        } else {
243            Ok((precision, scale))
244        }
245    }

247    /// Parse a `serde_json::Value` representing a complex Avro type into a `Schema`.
248    ///
249    /// Avro supports "recursive" definition of types.
250    /// e.g: `{"type": {"type": "string"}}`
251    pub(super) fn parse_complex(
252        &mut self,
253        complex: &Map<String, Value>,
254        enclosing_namespace: &Namespace,
255        parse_location: RecordSchemaParseLocation,
256    ) -> AvroResult<Schema> {
257        // Try to parse this as a native complex type.
258        fn parse_as_native_complex(
259            complex: &Map<String, Value>,
260            parser: &mut Parser,
261            enclosing_namespace: &Namespace,
262        ) -> AvroResult<Schema> {
263            match complex.get("type") {
264                Some(value) => match value {
265                    Value::String(s) if s == "fixed" => {
266                        parser.parse_fixed(complex, enclosing_namespace)
267                    }
268                    _ => parser.parse(value, enclosing_namespace),
269                },
270                None => Err(Details::GetLogicalTypeField.into()),
271            }
272        }
273
274        // This crate support some logical types natively, and this function tries to convert
275        // a native complex type with a logical type attribute to these logical types.
276        // This function:
277        // 1. Checks whether the native complex type is in the supported kinds.
278        // 2. If it is, using the convert function to convert the native complex type to
279        // a logical type.
280        fn try_convert_to_logical_type<F>(
281            logical_type: &str,
282            schema: Schema,
283            supported_schema_kinds: &[SchemaKind],
284            convert: F,
285        ) -> AvroResult<Schema>
286        where
287            F: Fn(Schema) -> AvroResult<Schema>,
288        {
289            let kind = SchemaKind::from(schema.clone());
290            if supported_schema_kinds.contains(&kind) {
291                convert(schema)
292            } else {
293                warn!(
294                    "Ignoring unknown logical type '{logical_type}' for schema of type: {schema:?}!"
295                );
296                Ok(schema)
297            }
298        }
299
300        match complex.get("logicalType") {
301            Some(Value::String(t)) => match t.as_str() {
302                "decimal" => {
303                    return try_convert_to_logical_type(
304                        "decimal",
305                        parse_as_native_complex(complex, self, enclosing_namespace)?,
306                        &[SchemaKind::Fixed, SchemaKind::Bytes],
307                        |inner| -> AvroResult<Schema> {
308                            match self.parse_precision_and_scale(complex) {
309                                Ok((precision, scale)) => Ok(Schema::Decimal(DecimalSchema {
310                                    precision,
311                                    scale,
312                                    inner: inner.try_into()?,
313                                })),
314                                Err(err) => {
315                                    warn!("Ignoring invalid decimal logical type: {err}");
316                                    Ok(inner)
317                                }
318                            }
319                        },
320                    );
321                }
322                "big-decimal" => {
323                    return try_convert_to_logical_type(
324                        "big-decimal",
325                        parse_as_native_complex(complex, self, enclosing_namespace)?,
326                        &[SchemaKind::Bytes],
327                        |_| -> AvroResult<Schema> { Ok(Schema::BigDecimal) },
328                    );
329                }
330                "uuid" => {
331                    return try_convert_to_logical_type(
332                        "uuid",
333                        parse_as_native_complex(complex, self, enclosing_namespace)?,
334                        &[SchemaKind::String, SchemaKind::Fixed, SchemaKind::Bytes],
335                        |schema| match schema {
336                            Schema::String => Ok(Schema::Uuid(UuidSchema::String)),
337                            Schema::Fixed(fixed @ FixedSchema { size: 16, .. }) => {
338                                Ok(Schema::Uuid(UuidSchema::Fixed(fixed)))
339                            }
340                            Schema::Fixed(FixedSchema { size, .. }) => {
341                                warn!(
342                                    "Ignoring uuid logical type for a Fixed schema because its size ({size:?}) is not 16! Schema: {schema:?}"
343                                );
344                                Ok(schema)
345                            }
346                            Schema::Bytes => Ok(Schema::Uuid(UuidSchema::Bytes)),
347                            _ => {
348                                warn!("Ignoring invalid uuid logical type for schema: {schema:?}");
349                                Ok(schema)
350                            }
351                        },
352                    );
353                }
354                "date" => {
355                    return try_convert_to_logical_type(
356                        "date",
357                        parse_as_native_complex(complex, self, enclosing_namespace)?,
358                        &[SchemaKind::Int],
359                        |_| -> AvroResult<Schema> { Ok(Schema::Date) },
360                    );
361                }
362                "time-millis" => {
363                    return try_convert_to_logical_type(
364                        "date",
365                        parse_as_native_complex(complex, self, enclosing_namespace)?,
366                        &[SchemaKind::Int],
367                        |_| -> AvroResult<Schema> { Ok(Schema::TimeMillis) },
368                    );
369                }
370                "time-micros" => {
371                    return try_convert_to_logical_type(
372                        "time-micros",
373                        parse_as_native_complex(complex, self, enclosing_namespace)?,
374                        &[SchemaKind::Long],
375                        |_| -> AvroResult<Schema> { Ok(Schema::TimeMicros) },
376                    );
377                }
378                "timestamp-millis" => {
379                    return try_convert_to_logical_type(
380                        "timestamp-millis",
381                        parse_as_native_complex(complex, self, enclosing_namespace)?,
382                        &[SchemaKind::Long],
383                        |_| -> AvroResult<Schema> { Ok(Schema::TimestampMillis) },
384                    );
385                }
386                "timestamp-micros" => {
387                    return try_convert_to_logical_type(
388                        "timestamp-micros",
389                        parse_as_native_complex(complex, self, enclosing_namespace)?,
390                        &[SchemaKind::Long],
391                        |_| -> AvroResult<Schema> { Ok(Schema::TimestampMicros) },
392                    );
393                }
394                "timestamp-nanos" => {
395                    return try_convert_to_logical_type(
396                        "timestamp-nanos",
397                        parse_as_native_complex(complex, self, enclosing_namespace)?,
398                        &[SchemaKind::Long],
399                        |_| -> AvroResult<Schema> { Ok(Schema::TimestampNanos) },
400                    );
401                }
402                "local-timestamp-millis" => {
403                    return try_convert_to_logical_type(
404                        "local-timestamp-millis",
405                        parse_as_native_complex(complex, self, enclosing_namespace)?,
406                        &[SchemaKind::Long],
407                        |_| -> AvroResult<Schema> { Ok(Schema::LocalTimestampMillis) },
408                    );
409                }
410                "local-timestamp-micros" => {
411                    return try_convert_to_logical_type(
412                        "local-timestamp-micros",
413                        parse_as_native_complex(complex, self, enclosing_namespace)?,
414                        &[SchemaKind::Long],
415                        |_| -> AvroResult<Schema> { Ok(Schema::LocalTimestampMicros) },
416                    );
417                }
418                "local-timestamp-nanos" => {
419                    return try_convert_to_logical_type(
420                        "local-timestamp-nanos",
421                        parse_as_native_complex(complex, self, enclosing_namespace)?,
422                        &[SchemaKind::Long],
423                        |_| -> AvroResult<Schema> { Ok(Schema::LocalTimestampNanos) },
424                    );
425                }
426                "duration" => {
427                    return try_convert_to_logical_type(
428                        "duration",
429                        parse_as_native_complex(complex, self, enclosing_namespace)?,
430                        &[SchemaKind::Fixed],
431                        |schema| -> AvroResult<Schema> {
432                            match schema {
433                                Schema::Fixed(fixed @ FixedSchema { size: 12, .. }) => {
434                                    Ok(Schema::Duration(fixed))
435                                }
436                                Schema::Fixed(FixedSchema { size, .. }) => {
437                                    warn!(
438                                        "Ignoring duration logical type on fixed type because size ({size}) is not 12! Schema: {schema:?}"
439                                    );
440                                    Ok(schema)
441                                }
442                                _ => {
443                                    warn!(
444                                        "Ignoring invalid duration logical type for schema: {schema:?}"
445                                    );
446                                    Ok(schema)
447                                }
448                            }
449                        },
450                    );
451                }
452                // In this case, of an unknown logical type, we just pass through the underlying
453                // type.
454                _ => {}
455            },
456            // The spec says to ignore invalid logical types and just pass through the
457            // underlying type. It is unclear whether that applies to this case or not, where the
458            // `logicalType` is not a string.
459            Some(value) => return Err(Details::GetLogicalTypeFieldType(value.clone()).into()),
460            _ => {}
461        }
462        match complex.get("type") {
463            Some(Value::String(t)) => match t.as_str() {
464                "record" => match parse_location {
465                    RecordSchemaParseLocation::Root => {
466                        self.parse_record(complex, enclosing_namespace)
467                    }
468                    RecordSchemaParseLocation::FromField => {
469                        self.fetch_schema_ref(t, enclosing_namespace)
470                    }
471                },
472                "enum" => self.parse_enum(complex, enclosing_namespace),
473                "array" => self.parse_array(complex, enclosing_namespace),
474                "map" => self.parse_map(complex, enclosing_namespace),
475                "fixed" => self.parse_fixed(complex, enclosing_namespace),
476                other => self.parse_known_schema(other, enclosing_namespace),
477            },
478            Some(Value::Object(data)) => {
479                self.parse_complex(data, enclosing_namespace, RecordSchemaParseLocation::Root)
480            }
481            Some(Value::Array(variants)) => self.parse_union(variants, enclosing_namespace),
482            Some(unknown) => Err(Details::GetComplexType(unknown.clone()).into()),
483            None => Err(Details::GetComplexTypeField.into()),
484        }
485    }

487    fn register_resolving_schema(&mut self, name: &Name, aliases: &Aliases) {
488        let resolving_schema = Schema::Ref { name: name.clone() };
489        self.resolving_schemas
490            .insert(name.clone(), resolving_schema.clone());
491
492        let namespace = &name.namespace;
493
494        if let Some(aliases) = aliases {
495            aliases.iter().for_each(|alias| {
496                let alias_fullname = alias.fully_qualified_name(namespace);
497                self.resolving_schemas
498                    .insert(alias_fullname, resolving_schema.clone());
499            });
500        }
501    }
502
503    fn register_parsed_schema(
504        &mut self,
505        fully_qualified_name: &Name,
506        schema: &Schema,
507        aliases: &Aliases,
508    ) {
509        // FIXME, this should be globally aware, so if there is something overwriting something
510        // else then there is an ambiguous schema definition. An appropriate error should be thrown
511        self.parsed_schemas
512            .insert(fully_qualified_name.clone(), schema.clone());
513        self.resolving_schemas.remove(fully_qualified_name);
514
515        let namespace = &fully_qualified_name.namespace;
516
517        if let Some(aliases) = aliases {
518            aliases.iter().for_each(|alias| {
519                let alias_fullname = alias.fully_qualified_name(namespace);
520                self.resolving_schemas.remove(&alias_fullname);
521                self.parsed_schemas.insert(alias_fullname, schema.clone());
522            });
523        }
524    }

526    /// Returns already parsed schema or a schema that is currently being resolved.
527    fn get_already_seen_schema(
528        &self,
529        complex: &Map<String, Value>,
530        enclosing_namespace: &Namespace,
531    ) -> Option<&Schema> {
532        match complex.get("type") {
533            Some(Value::String(typ)) => {
534                let name = Name::new(typ.as_str())
535                    .unwrap()
536                    .fully_qualified_name(enclosing_namespace);
537                self.resolving_schemas
538                    .get(&name)
539                    .or_else(|| self.parsed_schemas.get(&name))
540            }
541            _ => None,
542        }
543    }

545    /// Parse a `serde_json::Value` representing an Avro record type into a `Schema`.
546    fn parse_record(
547        &mut self,
548        complex: &Map<String, Value>,
549        enclosing_namespace: &Namespace,
550    ) -> AvroResult<Schema> {
551        let fields_opt = complex.get("fields");
552
553        if fields_opt.is_none()
554            && let Some(seen) = self.get_already_seen_schema(complex, enclosing_namespace)
555        {
556            return Ok(seen.clone());
557        }
558
559        let fully_qualified_name = Name::parse(complex, enclosing_namespace)?;
560        let aliases =
561            self.fix_aliases_namespace(complex.aliases(), &fully_qualified_name.namespace);
562
563        let mut lookup = BTreeMap::new();
564
565        self.register_resolving_schema(&fully_qualified_name, &aliases);
566
567        debug!("Going to parse record schema: {:?}", &fully_qualified_name);
568
569        let fields: Vec<RecordField> = fields_opt
570            .and_then(|fields| fields.as_array())
571            .ok_or_else(|| Error::new(Details::GetRecordFieldsJson))
572            .and_then(|fields| {
573                fields
574                    .iter()
575                    .filter_map(|field| field.as_object())
576                    .enumerate()
577                    .map(|(position, field)| {
578                        RecordField::parse(field, position, self, &fully_qualified_name)
579                    })
580                    .collect::<Result<_, _>>()
581            })?;
582
583        for field in &fields {
584            if let Some(_old) = lookup.insert(field.name.clone(), field.position) {
585                return Err(Details::FieldNameDuplicate(field.name.clone()).into());
586            }
587
588            if let Some(ref field_aliases) = field.aliases {
589                for alias in field_aliases {
590                    lookup.insert(alias.clone(), field.position);
591                }
592            }
593        }
594
595        let schema = Schema::Record(RecordSchema {
596            name: fully_qualified_name.clone(),
597            aliases: aliases.clone(),
598            doc: complex.doc(),
599            fields,
600            lookup,
601            attributes: self.get_custom_attributes(complex, vec!["fields"]),
602        });
603
604        self.register_parsed_schema(&fully_qualified_name, &schema, &aliases);
605        Ok(schema)
606    }

608    fn get_custom_attributes(
609        &self,
610        complex: &Map<String, Value>,
611        excluded: Vec<&'static str>,
612    ) -> BTreeMap<String, Value> {
613        let mut custom_attributes: BTreeMap<String, Value> = BTreeMap::new();
614        for (key, value) in complex {
615            match key.as_str() {
616                "type" | "name" | "namespace" | "doc" | "aliases" | "logicalType" => continue,
617                candidate if excluded.contains(&candidate) => continue,
618                _ => custom_attributes.insert(key.clone(), value.clone()),
619            };
620        }
621        custom_attributes
622    }

624    /// Parse a `serde_json::Value` representing a Avro enum type into a `Schema`.
625    fn parse_enum(
626        &mut self,
627        complex: &Map<String, Value>,
628        enclosing_namespace: &Namespace,
629    ) -> AvroResult<Schema> {
630        let symbols_opt = complex.get("symbols");
631
632        if symbols_opt.is_none()
633            && let Some(seen) = self.get_already_seen_schema(complex, enclosing_namespace)
634        {
635            return Ok(seen.clone());
636        }
637
638        let fully_qualified_name = Name::parse(complex, enclosing_namespace)?;
639        let aliases =
640            self.fix_aliases_namespace(complex.aliases(), &fully_qualified_name.namespace);
641
642        let symbols: Vec<String> = symbols_opt
643            .and_then(|v| v.as_array())
644            .ok_or_else(|| Error::from(Details::GetEnumSymbolsField))
645            .and_then(|symbols| {
646                symbols
647                    .iter()
648                    .map(|symbol| symbol.as_str().map(|s| s.to_string()))
649                    .collect::<Option<_>>()
650                    .ok_or_else(|| Error::from(Details::GetEnumSymbols))
651            })?;
652
653        let mut existing_symbols: HashSet<&String> = HashSet::with_capacity(symbols.len());
654        for symbol in symbols.iter() {
655            validate_enum_symbol_name(symbol)?;
656
657            // Ensure there are no duplicate symbols
658            if existing_symbols.contains(&symbol) {
659                return Err(Details::EnumSymbolDuplicate(symbol.to_string()).into());
660            }
661
662            existing_symbols.insert(symbol);
663        }
664
665        let mut default: Option<String> = None;
666        if let Some(value) = complex.get("default") {
667            if let Value::String(ref s) = *value {
668                default = Some(s.clone());
669            } else {
670                return Err(Details::EnumDefaultWrongType(value.clone()).into());
671            }
672        }
673
674        if let Some(ref value) = default {
675            let resolved = types::Value::from(value.clone())
676                .resolve_enum(&symbols, &Some(value.to_string()), &None)
677                .is_ok();
678            if !resolved {
679                return Err(Details::GetEnumDefault {
680                    symbol: value.to_string(),
681                    symbols,
682                }
683                .into());
684            }
685        }
686
687        let schema = Schema::Enum(EnumSchema {
688            name: fully_qualified_name.clone(),
689            aliases: aliases.clone(),
690            doc: complex.doc(),
691            symbols,
692            default,
693            attributes: self.get_custom_attributes(complex, vec!["symbols"]),
694        });
695
696        self.register_parsed_schema(&fully_qualified_name, &schema, &aliases);
697
698        Ok(schema)
699    }

737    /// Parse a `serde_json::Value` representing a Avro union type into a `Schema`.
738    fn parse_union(
739        &mut self,
740        items: &[Value],
741        enclosing_namespace: &Namespace,
742    ) -> AvroResult<Schema> {
743        items
744            .iter()
745            .map(|v| self.parse(v, enclosing_namespace))
746            .collect::<Result<Vec<_>, _>>()
747            .and_then(|schemas| {
748                if schemas.is_empty() {
749                    error!(
750                        "Union schemas should have at least two members! \
751                    Please enable debug logging to find out which Record schema \
752                    declares the union with 'RUST_LOG=apache_avro::schema=debug'."
753                    );
754                } else if schemas.len() == 1 {
755                    warn!(
756                        "Union schema with just one member! Consider dropping the union! \
757                    Please enable debug logging to find out which Record schema \
758                    declares the union with 'RUST_LOG=apache_avro::schema=debug'."
759                    );
760                }
761                Ok(Schema::Union(UnionSchema::new(schemas)?))
762            })
763    }

765    /// Parse a `serde_json::Value` representing a Avro fixed type into a `Schema`.
766    fn parse_fixed(
767        &mut self,
768        complex: &Map<String, Value>,
769        enclosing_namespace: &Namespace,
770    ) -> AvroResult<Schema> {
771        let size_opt = complex.get("size");
772        if size_opt.is_none()
773            && let Some(seen) = self.get_already_seen_schema(complex, enclosing_namespace)
774        {
775            return Ok(seen.clone());
776        }
777
778        let doc = complex.get("doc").and_then(|v| match &v {
779            &Value::String(docstr) => Some(docstr.clone()),
780            _ => None,
781        });
782
783        let size = match size_opt {
784            Some(size) => size
785                .as_u64()
786                .ok_or_else(|| Details::GetFixedSizeFieldPositive(size.clone())),
787            None => Err(Details::GetFixedSizeField),
788        }?;
789
790        let default = complex.get("default").and_then(|v| match &v {
791            &Value::String(default) => Some(default.clone()),
792            _ => None,
793        });
794
795        if default.is_some() {
796            let len = default.clone().unwrap().len();
797            if len != size as usize {
798                return Err(Details::FixedDefaultLenSizeMismatch(len, size).into());
799            }
800        }
801
802        let fully_qualified_name = Name::parse(complex, enclosing_namespace)?;
803        let aliases =
804            self.fix_aliases_namespace(complex.aliases(), &fully_qualified_name.namespace);
805
806        let schema = Schema::Fixed(FixedSchema {
807            name: fully_qualified_name.clone(),
808            aliases: aliases.clone(),
809            doc,
810            size: size as usize,
811            default,
812            attributes: self.get_custom_attributes(complex, vec!["size"]),
813        });
814
815        self.register_parsed_schema(&fully_qualified_name, &schema, &aliases);
816
817        Ok(schema)
818    }

820    // A type alias may be specified either as a fully namespace-qualified, or relative
821    // to the namespace of the name it is an alias for. For example, if a type named "a.b"
822    // has aliases of "c" and "x.y", then the fully qualified names of its aliases are "a.c"
823    // and "x.y".
824    // https://avro.apache.org/docs/++version++/specification/#aliases
825    fn fix_aliases_namespace(
826        &self,
827        aliases: Option<Vec<String>>,
828        namespace: &Namespace,
829    ) -> Aliases {
830        aliases.map(|aliases| {
831            aliases
832                .iter()
833                .map(|alias| {
834                    if alias.find('.').is_none() {
835                        match namespace {
836                            Some(ns) => format!("{ns}.{alias}"),
837                            None => alias.clone(),
838                        }
839                    } else {
840                        alias.clone()
841                    }
842                })
843                .map(|alias| Alias::new(alias.as_str()).unwrap())
844                .collect()
845        })
846    }

848    fn get_schema_type_name(&self, name: Name, value: Value) -> Name {
849        match value.get("type") {
850            Some(Value::Object(complex_type)) => match complex_type.name() {
851                Some(name) => Name::new(name.as_str()).unwrap(),
852                _ => name,
853            },
854            _ => name,
855        }
856    }

858    fn parse_json_integer_for_decimal(
859        &self,
860        value: &serde_json::Number,
861    ) -> AvroResult<DecimalMetadata> {
862        Ok(if value.is_u64() {
863            let num = value
864                .as_u64()
865                .ok_or_else(|| Details::GetU64FromJson(value.clone()))?;
866            num.try_into()
867                .map_err(|e| Details::ConvertU64ToUsize(e, num))?
868        } else if value.is_i64() {
869            let num = value
870                .as_i64()
871                .ok_or_else(|| Details::GetI64FromJson(value.clone()))?;
872            num.try_into()
873                .map_err(|e| Details::ConvertI64ToUsize(e, num))?
874        } else {
875            return Err(Details::GetPrecisionOrScaleFromJson(value.clone()).into());
876        })
877    }
