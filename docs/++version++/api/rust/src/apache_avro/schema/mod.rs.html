84/// Represents any valid Avro schema
85/// More information about Avro schemas can be found in the
86/// [Avro Specification](https://avro.apache.org/docs/++version++/specification/#schema-declaration)
87#[derive(Clone, Debug, EnumDiscriminants, Display)]
88#[strum_discriminants(name(SchemaKind), derive(Hash, Ord, PartialOrd))]
89pub enum Schema {
90    /// A `null` Avro schema.
91    Null,
92    /// A `boolean` Avro schema.
93    Boolean,
94    /// An `int` Avro schema.
95    Int,
96    /// A `long` Avro schema.
97    Long,
98    /// A `float` Avro schema.
99    Float,
100    /// A `double` Avro schema.
101    Double,
102    /// A `bytes` Avro schema.
103    ///
104    /// `Bytes` represents a sequence of 8-bit unsigned bytes.
105    Bytes,
106    /// A `string` Avro schema.
107    ///
108    /// `String` represents a unicode character sequence.
109    String,
110    /// An `array` Avro schema.
111    ///
112    /// All items will have the same schema.
113    Array(ArraySchema),
114    /// A `map` Avro schema.
115    ///
116    /// Keys are always a `Schema::String` and all values will have the same schema.
117    Map(MapSchema),
118    /// A `union` Avro schema.
119    Union(UnionSchema),
120    /// A `record` Avro schema.
121    Record(RecordSchema),
122    /// An `enum` Avro schema.
123    Enum(EnumSchema),
124    /// A `fixed` Avro schema.
125    Fixed(FixedSchema),
126    /// Logical type which represents `Decimal` values.
127    ///
128    /// The underlying type is serialized and deserialized as `Schema::Bytes` or `Schema::Fixed`.
129    Decimal(DecimalSchema),
130    /// Logical type which represents `Decimal` values without predefined scale.
131    ///
132    /// The underlying type is serialized and deserialized as `Schema::Bytes`
133    BigDecimal,
134    /// A universally unique identifier, annotating a string, bytes or fixed.
135    Uuid(UuidSchema),
136    /// Logical type which represents the number of days since the unix epoch.
137    ///
138    /// Serialization format is `Schema::Int`.
139    Date,
140    /// The time of day in number of milliseconds after midnight.
141    ///
142    /// This type has no reference to any calendar, time zone or date in particular.
143    TimeMillis,
144    /// The time of day in number of microseconds after midnight.
145    ///
146    /// This type has no reference to any calendar, time zone or date in particular.
147    TimeMicros,
148    /// An instant in time represented as the number of milliseconds after the UNIX epoch.
149    TimestampMillis,
150    /// An instant in time represented as the number of microseconds after the UNIX epoch.
151    TimestampMicros,
152    /// An instant in time represented as the number of nanoseconds after the UNIX epoch.
153    TimestampNanos,
154    /// An instant in localtime represented as the number of milliseconds after the UNIX epoch.
155    LocalTimestampMillis,
156    /// An instant in local time represented as the number of microseconds after the UNIX epoch.
157    LocalTimestampMicros,
158    /// An instant in local time represented as the number of nanoseconds after the UNIX epoch.
159    LocalTimestampNanos,
160    /// An amount of time defined by a number of months, days and milliseconds.
161    Duration(FixedSchema),
162    /// A reference to another schema.
163    Ref { name: Name },
164}

291impl FixedSchema {
292    fn serialize_to_map<S>(&self, mut map: S::SerializeMap) -> Result<S::SerializeMap, S::Error>
293    where
294        S: Serializer,
295    {
296        map.serialize_entry("type", "fixed")?;
297        if let Some(ref n) = self.name.namespace {
298            map.serialize_entry("namespace", n)?;
299        }
300        map.serialize_entry("name", &self.name.name)?;
301        if let Some(ref docstr) = self.doc {
302            map.serialize_entry("doc", docstr)?;
303        }
304        map.serialize_entry("size", &self.size)?;
305
306        if let Some(ref aliases) = self.aliases {
307            map.serialize_entry("aliases", aliases)?;
308        }
309
310        for attr in &self.attributes {
311            map.serialize_entry(attr.0, attr.1)?;
312        }
313
314        Ok(map)
315    }

387    /// Converts `self` into its [Parsing Canonical Form].
388    ///
389    /// [Parsing Canonical Form]:
390    /// https://avro.apache.org/docs/++version++/specification/#parsing-canonical-form-for-schemas
391    pub fn canonical_form(&self) -> String {
392        let json = serde_json::to_value(self)
393            .unwrap_or_else(|e| panic!("Cannot parse Schema from JSON: {e}"));
394        let mut defined_names = HashSet::new();
395        parsing_canonical_form(&json, &mut defined_names)
396    }
397
398    /// Returns the [Parsing Canonical Form] of `self` that is self contained (not dependent on
399    /// any definitions in `schemata`)
400    ///
401    /// [Parsing Canonical Form]:
402    /// https://avro.apache.org/docs/++version++/specification/#parsing-canonical-form-for-schemas
403    pub fn independent_canonical_form(&self, schemata: &[Schema]) -> Result<String, Error> {
404        let mut this = self.clone();
405        this.denormalize(schemata, &mut HashSet::with_capacity(schemata.len()))?;
406        Ok(this.canonical_form())
407    }

409    /// Generate the [fingerprint] of the schema's [Parsing Canonical Form].
410    ///
411    /// # Example
412    /// ```
413    /// use apache_avro::rabin::Rabin;
414    /// use apache_avro::{Schema, Error};
415    /// use md5::Md5;
416    /// use sha2::Sha256;
417    ///
418    /// fn main() -> Result<(), Error> {
419    ///     let raw_schema = r#"
420    ///         {
421    ///             "type": "record",
422    ///             "name": "test",
423    ///             "fields": [
424    ///                 {"name": "a", "type": "long", "default": 42},
425    ///                 {"name": "b", "type": "string"}
426    ///             ]
427    ///         }
428    ///     "#;
429    ///     let schema = Schema::parse_str(raw_schema)?;
430    ///     println!("{}", schema.fingerprint::<Sha256>());
431    ///     println!("{}", schema.fingerprint::<Md5>());
432    ///     println!("{}", schema.fingerprint::<Rabin>());
433    ///     Ok(())
434    /// }
435    /// ```
436    ///
437    /// [Parsing Canonical Form]:
438    /// https://avro.apache.org/docs/++version++/specification/#parsing-canonical-form-for-schemas
439    /// [fingerprint]:
440    /// https://avro.apache.org/docs/++version++/specification/#schema-fingerprints
441    pub fn fingerprint<D: Digest>(&self) -> SchemaFingerprint {
442        let mut d = D::new();
443        d.update(self.canonical_form());
444        SchemaFingerprint {
445            bytes: d.finalize().to_vec(),
446        }
447    }

455    /// Create an array of `Schema`'s from a list of named JSON Avro schemas (Record, Enum, and
456    /// Fixed).
457    ///
458    /// It is allowed that the schemas have cross-dependencies; these will be resolved
459    /// during parsing.
460    ///
461    /// If two of the input schemas have the same fullname, an Error will be returned.
462    pub fn parse_list(input: impl IntoIterator<Item = impl AsRef<str>>) -> AvroResult<Vec<Schema>> {
463        let input = input.into_iter();
464        let input_len = input.size_hint().0;
465        let mut input_schemas: HashMap<Name, Value> = HashMap::with_capacity(input_len);
466        let mut input_order: Vec<Name> = Vec::with_capacity(input_len);
467        for json in input {
468            let json = json.as_ref();
469            let schema: Value = serde_json::from_str(json).map_err(Details::ParseSchemaJson)?;
470            if let Value::Object(inner) = &schema {
471                let name = Name::parse(inner, &None)?;
472                let previous_value = input_schemas.insert(name.clone(), schema);
473                if previous_value.is_some() {
474                    return Err(Details::NameCollision(name.fullname(None)).into());
475                }
476                input_order.push(name);
477            } else {
478                return Err(Details::GetNameField.into());
479            }
480        }
481        let mut parser = Parser::new(
482            input_schemas,
483            input_order,
484            HashMap::with_capacity(input_len),
485        );
486        parser.parse_list()
487    }

489    /// Create a `Schema` from a string representing a JSON Avro schema,
490    /// along with an array of `Schema`'s from a list of named JSON Avro schemas (Record, Enum, and
491    /// Fixed).
492    ///
493    /// It is allowed that the schemas have cross-dependencies; these will be resolved
494    /// during parsing.
495    ///
496    /// If two of the named input schemas have the same fullname, an Error will be returned.
497    ///
498    /// # Arguments
499    /// * `schema` - the JSON string of the schema to parse
500    /// * `schemata` - a slice of additional schemas that is used to resolve cross-references
501    pub fn parse_str_with_list(
502        schema: &str,
503        schemata: impl IntoIterator<Item = impl AsRef<str>>,
504    ) -> AvroResult<(Schema, Vec<Schema>)> {
505        let schemata = schemata.into_iter();
506        let schemata_len = schemata.size_hint().0;
507        let mut input_schemas: HashMap<Name, Value> = HashMap::with_capacity(schemata_len);
508        let mut input_order: Vec<Name> = Vec::with_capacity(schemata_len);
509        for json in schemata {
510            let json = json.as_ref();
511            let schema: Value = serde_json::from_str(json).map_err(Details::ParseSchemaJson)?;
512            if let Value::Object(inner) = &schema {
513                let name = Name::parse(inner, &None)?;
514                if let Some(_previous) = input_schemas.insert(name.clone(), schema) {
515                    return Err(Details::NameCollision(name.fullname(None)).into());
516                }
517                input_order.push(name);
518            } else {
519                return Err(Details::GetNameField.into());
520            }
521        }
522        let mut parser = Parser::new(
523            input_schemas,
524            input_order,
525            HashMap::with_capacity(schemata_len),
526        );
527        parser.parse_input_schemas()?;
528
529        let value = serde_json::from_str(schema).map_err(Details::ParseSchemaJson)?;
530        let schema = parser.parse(&value, &None)?;
531        let schemata = parser.parse_list()?;
532        Ok((schema, schemata))
533    }

680    /// # Errors
681    /// Will return an error if `schemas` has duplicate unnamed schemas or if `schemas`
682    /// contains a union.
683    pub fn union(schemas: Vec<Schema>) -> AvroResult<Schema> {
684        UnionSchema::new(schemas).map(Schema::Union)
685    }
686
687    fn denormalize(
688        &mut self,
689        schemata: &[Schema],
690        defined_names: &mut HashSet<Name>,
691    ) -> AvroResult<()> {
692        // If this name already exists in this schema we can reference it.
693        // This makes the denormalized form as small as possible and prevent infinite loops for recursive types.
694        if let Some(name) = self.name()
695            && defined_names.contains(name)
696        {
697            *self = Schema::Ref { name: name.clone() };
698            return Ok(());
699        }
700        match self {
701            Schema::Ref { name } => {
702                let replacement_schema = schemata
703                    .iter()
704                    .find(|s| s.name().map(|n| *n == *name).unwrap_or(false));
705                if let Some(schema) = replacement_schema {
706                    let mut denorm = schema.clone();
707                    denorm.denormalize(schemata, defined_names)?;
708                    *self = denorm;
709                } else {
710                    return Err(Details::SchemaResolutionError(name.clone()).into());
711                }
712            }
713            Schema::Record(record_schema) => {
714                defined_names.insert(record_schema.name.clone());
715                for field in &mut record_schema.fields {
716                    field.schema.denormalize(schemata, defined_names)?;
717                }
718            }
719            Schema::Array(array_schema) => {
720                array_schema.items.denormalize(schemata, defined_names)?;
721            }
722            Schema::Map(map_schema) => {
723                map_schema.types.denormalize(schemata, defined_names)?;
724            }
725            Schema::Union(union_schema) => {
726                for schema in &mut union_schema.schemas {
727                    schema.denormalize(schemata, defined_names)?;
728                }
729            }
730            schema if schema.is_named() => {
731                defined_names.insert(schema.name().expect("Schema is named").clone());
732            }
733            _ => (),
734        }
735        Ok(())
736    }
737}

739impl Serialize for Schema {
740    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
741    where
742        S: Serializer,
743    {
744        match &self {
745            Schema::Ref { name } => serializer.serialize_str(&name.fullname(None)),
746            Schema::Null => serializer.serialize_str("null"),
747            Schema::Boolean => serializer.serialize_str("boolean"),
748            Schema::Int => serializer.serialize_str("int"),
749            Schema::Long => serializer.serialize_str("long"),
750            Schema::Float => serializer.serialize_str("float"),
751            Schema::Double => serializer.serialize_str("double"),
752            Schema::Bytes => serializer.serialize_str("bytes"),
753            Schema::String => serializer.serialize_str("string"),
754            Schema::Array(inner) => {
755                let mut map = serializer.serialize_map(Some(2 + inner.attributes.len()))?;
756                map.serialize_entry("type", "array")?;
757                map.serialize_entry("items", &*inner.items.clone())?;
758                for attr in &inner.attributes {
759                    map.serialize_entry(attr.0, attr.1)?;
760                }
761                map.end()
762            }
763            Schema::Map(inner) => {
764                let mut map = serializer.serialize_map(Some(2 + inner.attributes.len()))?;
765                map.serialize_entry("type", "map")?;
766                map.serialize_entry("values", &*inner.types.clone())?;
767                for attr in &inner.attributes {
768                    map.serialize_entry(attr.0, attr.1)?;
769                }
770                map.end()
771            }
772            Schema::Union(inner) => {
773                let variants = inner.variants();
774                let mut seq = serializer.serialize_seq(Some(variants.len()))?;
775                for v in variants {

777                }
778                seq.end()
779            }
780            Schema::Record(RecordSchema {
781                name,
782                aliases,

784                fields,

786                ..

941/// Parses a valid Avro schema into [the Parsing Canonical Form].
942///
943/// [the Parsing Canonical Form](https://avro.apache.org/docs/++version++/specification/#parsing-canonical-form-for-schemas)
944fn parsing_canonical_form(schema: &Value, defined_names: &mut HashSet<String>) -> String {
945    match schema {
946        Value::Object(map) => pcf_map(map, defined_names),
947        Value::String(s) => pcf_string(s),
948        Value::Array(v) => pcf_array(v, defined_names),
949        json => panic!("got invalid JSON value for canonical form of schema: {json}"),
950    }
951}
952
953fn pcf_map(schema: &Map<String, Value>, defined_names: &mut HashSet<String>) -> String {
954    let typ = schema.get("type").and_then(|v| v.as_str());
955    let name = if is_named_type(typ) {
956        let ns = schema.get("namespace").and_then(|v| v.as_str());
957        let raw_name = schema.get("name").and_then(|v| v.as_str());
958        Some(format!(
959            "{}{}",
960            ns.map_or("".to_string(), |n| { format!("{n}.") }),
961            raw_name.unwrap_or_default()
962        ))
963    } else {
964        None
965    };
966
967    //if this is already a defined type, early return
968    if let Some(ref n) = name {
969        if defined_names.contains(n) {
970            return pcf_string(n);
971        } else {
972            defined_names.insert(n.clone());
973        }
974    }
975
976    let mut fields = Vec::new();
977    for (k, v) in schema {
978        // Reduce primitive types to their simple form. ([PRIMITIVE] rule)
979        if schema.len() == 1 && k == "type" {
980            // Invariant: function is only callable from a valid schema, so this is acceptable.
981            if let Value::String(s) = v {
982                return pcf_string(s);
983            }
984        }
985
986        // Strip out unused fields ([STRIP] rule)
987        if field_ordering_position(k).is_none()
988            || k == "default"
989            || k == "doc"
990            || k == "aliases"
991            || k == "logicalType"
992        {
993            continue;
994        }
995
996        // Fully qualify the name, if it isn't already ([FULLNAMES] rule).
997        if k == "name"
998            && let Some(ref n) = name
999        {
1000            fields.push(("name", format!("{}:{}", pcf_string(k), pcf_string(n))));
1001            continue;
1002        }
1003
1004        // Strip off quotes surrounding "size" type, if they exist ([INTEGERS] rule).
1005        if k == "size" || k == "precision" || k == "scale" {
1006            let i = match v.as_str() {
1007                Some(s) => s.parse::<i64>().expect("Only valid schemas are accepted!"),
1008                None => v.as_i64().unwrap(),
1009            };
1010            fields.push((k, format!("{}:{}", pcf_string(k), i)));
1011            continue;
1012        }
1013
1014        // For anything else, recursively process the result.
1015        fields.push((
1016            k,
1017            format!(
1018                "{}:{}",
1019                pcf_string(k),
1020                parsing_canonical_form(v, defined_names)
1021            ),
1022        ));
1023    }
1024
1025    // Sort the fields by their canonical ordering ([ORDER] rule).
1026    fields.sort_unstable_by_key(|(k, _)| field_ordering_position(k).unwrap());
1027    let inter = fields
1028        .into_iter()
1029        .map(|(_, v)| v)
1030        .collect::<Vec<_>>()
1031        .join(",");
1032    format!("{{{inter}}}")
1033}
1034
1035fn is_named_type(typ: Option<&str>) -> bool {
1036    matches!(
1037        typ,
1038        Some("record") | Some("enum") | Some("fixed") | Some("ref")
1039    )
1040}
1041
1042fn pcf_array(arr: &[Value], defined_names: &mut HashSet<String>) -> String {
1043    let inter = arr
1044        .iter()
1045        .map(|a| parsing_canonical_form(a, defined_names))
1046        .collect::<Vec<String>>()
1047        .join(",");
1048    format!("[{inter}]")
1049}

1051fn pcf_string(s: &str) -> String {
1052    format!("\"{s}\"")
1053}

1056    "name",
1057    "type",
1058    "fields",
1059    "symbols",

1073fn field_ordering_position(field: &str) -> Option<usize> {

1076        .position(|&f| f == field)
1077        .map(|pos| pos + 1)
