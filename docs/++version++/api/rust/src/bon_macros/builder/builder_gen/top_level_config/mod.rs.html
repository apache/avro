75impl TopLevelConfig {
76    pub(crate) fn parse_for_fn(fn_item: &ItemFn, config: Option<TokenStream>) -> Result<Self> {
77        let other_configs = fn_item
78            .attrs
79            .iter()
80            .filter(|attr| attr.path().is_ident("builder"))
81            .map(|attr| {
82                if let syn::Meta::List(_) = attr.meta {
83                    crate::parsing::require_non_empty_paren_meta_list_or_name_value(&attr.meta)?;
84                }
85                let meta_list = darling::util::parse_attribute_to_meta_list(attr)?;
86                Ok(meta_list.tokens)
87            });
88
89        let configs = config
90            .map(Ok)
91            .into_iter()
92            .chain(other_configs)
93            .collect::<Result<Vec<_>>>()?;
94
95        let me = Self::parse_for_any(configs)?;
96
97        if me.start_fn.name.is_none() {
98            let ItemSigConfig { name: _, vis, docs } = &me.start_fn;
99
100            let unexpected_param = None
101                .or_else(|| vis.as_ref().map(SpannedKey::key))
102                .or_else(|| docs.as_ref().map(SpannedKey::key));
103
104            if let Some(unexpected_param) = unexpected_param {
105                bail!(
106                    unexpected_param,
107                    "#[builder(start_fn({unexpected_param}))] requires that you \
108                    also specify #[builder(start_fn(name))] which makes the starting \
109                    function not to replace the positional function under the #[builder] \
110                    attribute; by default (without the explicit #[builder(start_fn(name))]) \
111                    the name, visibility and documentation of the positional \
112                    function are all copied to the starting function, and the positional \
113                    function under the #[builder] attribute becomes private with \
114                    #[doc(hidden)] and it's renamed (the name is not guaranteed \
115                    to be stable) to make it inaccessible even within the current module",
116                );
117            }
118        }
119
120        Ok(me)
121    }
122
123    pub(crate) fn parse_for_struct(configs: Vec<TokenStream>) -> Result<Self> {
124        Self::parse_for_any(configs)
125    }
126
127    fn parse_for_any(configs: Vec<TokenStream>) -> Result<Self> {
128        let configs = configs
129            .into_iter()
130            .map(NestedMeta::parse_meta_list)
131            .collect::<Result<Vec<_>, _>>()?
132            .into_iter()
133            .flatten()
134            .collect::<Vec<_>>();
135
136        // This is a temporary hack. We only allow `on(_, required)` as the
137        // first `on(...)` clause. Instead we should implement an extended design:
138        // https://github.com/elastio/bon/issues/152
139        let mut on_configs = configs
140            .iter()
141            .enumerate()
142            .filter_map(|(i, meta)| match meta {
143                NestedMeta::Meta(syn::Meta::List(meta)) if meta.path.is_ident("on") => {
144                    Some((i, meta))
145                }
146                _ => None,
147            })
148            .peekable();
149
150        while let Some((i, _)) = on_configs.next() {
151            if let Some((j, next_on)) = on_configs.peek() {
152                if *j != i + 1 {
153                    bail!(
154                        next_on,
155                        "this `on(...)` clause is out of order; all `on(...)` \
156                        clauses must be consecutive; there shouldn't be any \
157                        other parameters between them",
158                    )
159                }
160            }
161        }
162
163        let me = Self::from_list(&configs)?;
164
165        if let Some(on) = me.on.iter().skip(1).find(|on| on.required.is_present()) {
166            bail!(
167                &on.required.span(),
168                "`required` can only be specified in the first `on(...)` clause; \
169                this restriction may be lifted in the future",
170            );
171        }
172
173        if let Some(first_on) = me.on.first().filter(|on| on.required.is_present()) {
174            if !matches!(first_on.type_pattern, syn::Type::Infer(_)) {
175                bail!(
176                    &first_on.type_pattern,
177                    "`required` can only be used with the wildcard type pattern \
178                    i.e. `on(_, required)`; this restriction may be lifted in the future",
179                );
180            }
181        }
182
183        Ok(me)
184    }
185}

204impl FromMeta for DeriveConfig {
205    fn from_meta(meta: &syn::Meta) -> Result<Self> {
206        if let syn::Meta::Path(_) = meta {
207            return Ok(Self { bounds: None });
208        }
209
210        meta.require_list()?.require_parens_delim()?;
211
212        #[derive(FromMeta)]
213        struct Parsed {
214            #[darling(with = crate::parsing::parse_paren_meta_list_with_terminated)]
215            bounds: Punctuated<syn::WherePredicate, syn::Token![,]>,
216        }
217
218        let Parsed { bounds } = Parsed::from_meta(meta)?;
219
220        Ok(Self {
221            bounds: Some(bounds),
222        })
223    }
