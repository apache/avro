232/// A list of deflate flush types.
233#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
234#[non_exhaustive]
235pub enum TDEFLFlush {
236    /// Normal operation.
237    ///
238    /// Compress as much as there is space for, and then return waiting for more input.
239    None = 0,
240
241    /// Try to flush all the current data and output an empty fixed
242    /// block (10 bits) to synchonize the stream.
243    Partial = 1,
244
245    /// Try to flush all the current data and output an empty raw
246    /// block (3-10 bits + 32 bits) to synchonize the stream.
247    Sync = 2,
248
249    /// Same as [`Sync`][Self::Sync], but reset the dictionary so that the following data does not
250    /// depend on previous data.
251    Full = 3,
252
253    /// Try to flush everything and end the deflate stream.
254    ///
255    /// On success this will yield a [`TDEFLStatus::Done`] return status.
256    Finish = 4,
257
258    /// Try to flush all the current data and, if data is unaligned,
259    /// output an empty fixed block (10 bits) to synchonize the
260    /// stream.
261    PartialOpt = 5,
262
263    /// Try to flush all the current data and, if data is unaligned,
264    /// output an empty raw block (3-10 bits + 32 bits) to synchonize
265    /// the stream.
266    SyncOpt = 6,
267
268    /// Try to flush the current data but without any sync, which may
269    /// leave up to 7 bits of data not output.  You can use
270    /// `TDEFLFlush::PartialOpt` or `TDEFLFlush::SyncOpt` to add a
271    /// sync sequence on a future call if you later decide that you
272    /// have space downstream to forward that final byte.
273    NoSync = 7,
274}

354fn write_u16_le(val: u16, slice: &mut [u8], pos: usize) {

359// Read the two bytes starting at pos and interpret them as an u16.

362    // The compiler is smart enough to optimize this into an unaligned load.

391        // If less than 12, use RLE (i.e window size of 1) unless huffman only or stored blocks are requested.
392        if current_strategy != CompressionStrategy::HuffmanOnly && current_level != 0 {
393            (1, CompressionStrategy::RLE)
394        } else {
395            (current_level, current_strategy)
396        }
397    // If less than 15 but 12 or more, use fast mode that uses a 4k window size.
398    } else if window_bits < MZ_DEFAULT_WINDOW_BITS as u8 {
399        (cmp::min(current_level, 1), current_strategy)
400    // If window bits is the full 15 just use the requested settings.
401    } else {
402        (current_level, current_strategy)
403    }
404}

611struct CallbackBuf<'a> {

617        &mut self,
618        saved_output: SavedOutputBufferOxide,
619        params: &mut ParamsOxide,
620    ) -> i32 {
621        if saved_output.local {
622            let n = cmp::min(saved_output.pos, self.out_buf.len() - params.out_buf_ofs);

629                params.flush_remaining = (saved_output.pos - n) as u32;
630            }
631        } else {
632            params.out_buf_ofs += saved_output.pos;
633        }
634
635        params.flush_remaining as i32
636    }

730struct OutputBufferOxide<'a> {
731    pub inner: &'a mut [u8],
732    pub inner_pos: usize,
733    pub local: bool,
734
735    pub bit_buffer: u32,
736    pub bits_in: u32,
737}
738
739impl OutputBufferOxide<'_> {
740    /// Write bits to the bit buffer and flushes
741    /// the bit buffer so any whole bytes are output
742    /// to the underlying buffer.
743    fn put_bits(&mut self, bits: u32, len: u32) {
744        // TODO: Removing this assertion worsens performance
745        // Need to figure out why
746        assert!(bits <= ((1u32 << len) - 1u32));
747        self.bit_buffer |= bits << self.bits_in;
748        self.bits_in += len;
749
750        while self.bits_in >= 8 {
751            self.inner[self.inner_pos] = self.bit_buffer as u8;
752            self.inner_pos += 1;
753            self.bit_buffer >>= 8;
754            self.bits_in -= 8;
755        }
756    }
757
758    #[inline]
759    /// Write the provided bits to the bit buffer without flushing
760    /// anything. Does not check if there is actually space for it.
761    fn put_bits_no_flush(&mut self, bits: u32, len: u32) {
762        self.bit_buffer |= bits << self.bits_in;
763        self.bits_in += len;
764    }
765
766    const fn save(&self) -> SavedOutputBufferOxide {
767        SavedOutputBufferOxide {
768            pos: self.inner_pos,
769            bit_buffer: self.bit_buffer,
770            bits_in: self.bits_in,
771            local: self.local,
772        }
773    }
774
775    fn load(&mut self, saved: SavedOutputBufferOxide) {
776        self.inner_pos = saved.pos;
777        self.bit_buffer = saved.bit_buffer;
778        self.bits_in = saved.bits_in;
779        self.local = saved.local;
780    }
781
782    #[inline]
783    /// Pad the bit buffer to a whole byte with
784    /// zeroes and write that byte to the output buffer.
785    fn pad_to_bytes(&mut self) {
786        if self.bits_in != 0 {
787            let len = 8 - self.bits_in;
788            self.put_bits(0, len);
789        }
790    }
791
792    /// Test whether the output is currently on a byte boundary,
793    /// i.e. all current data has been output
794    #[inline]
795    const fn is_byte_aligned(&self) -> bool {
796        self.bits_in == 0
797    }
798
799    #[inline]
800    fn write_bytes(&mut self, bytes: &[u8]) {
801        debug_assert_eq!(self.bits_in, 0);
802        self.inner[self.inner_pos..self.inner_pos + bytes.len()].copy_from_slice(bytes);
803        self.inner_pos += bytes.len();
804    }
805}

814struct BitBuffer {

816    pub bits_in: u32,

819impl BitBuffer {
820    fn put_fast(&mut self, bits: u64, len: u32) {
821        self.bit_buffer |= bits << self.bits_in;
822        self.bits_in += len;
823    }
824
825    fn flush(&mut self, output: &mut OutputBufferOxide) -> Result<()> {
826        let pos = output.inner_pos;
827        {
828            // isolation to please borrow checker
829            let inner = &mut output.inner[pos..pos + 8];
830            let bytes = u64::to_le_bytes(self.bit_buffer);
831            inner.copy_from_slice(&bytes);
832        }
833        match output.inner_pos.checked_add((self.bits_in >> 3) as usize) {
834            Some(n) if n <= output.inner.len() => output.inner_pos = n,
835            _ => return Err(Error {}),
836        }
837        self.bit_buffer >>= self.bits_in & !7;
838        self.bits_in &= 7;
839        Ok(())
840    }

843/// A struct containing data about huffman codes and symbol frequencies.
844///
845/// NOTE: Only the literal/lengths have enough symbols to actually use
846/// the full array. It's unclear why it's defined like this in miniz,
847/// it could be for cache/alignment reasons.
848#[derive(Clone)]
849pub(crate) struct HuffmanOxide {
850    /// Number of occurrences of each symbol.
851    pub count: [[u16; MAX_HUFF_SYMBOLS]; MAX_HUFF_TABLES],
852    /// The bits of the huffman code assigned to the symbol
853    pub codes: [[u16; MAX_HUFF_SYMBOLS]; MAX_HUFF_TABLES],
854    /// The length of the huffman code assigned to the symbol.
855    pub code_sizes: [[u8; MAX_HUFF_SYMBOLS]; MAX_HUFF_TABLES],
856}
857
858/// Tables used for literal/lengths in `HuffmanOxide`.
859const LITLEN_TABLE: usize = 0;
860/// Tables for distances.
861const DIST_TABLE: usize = 1;
862/// Tables for the run-length encoded huffman lengths for literals/lengths/distances.
863const HUFF_CODES_TABLE: usize = 2;
864
865/// Status of RLE encoding of huffman code lengths.
866struct Rle {
867    pub z_count: u32,
868    pub repeat_count: u16,
869    pub prev_code_size: u8,
870}
871
872impl Rle {
873    fn prev_code_size(
874        &mut self,
875        packed_code_sizes: &mut [u8],
876        packed_pos: &mut usize,
877        h: &mut HuffmanOxide,
878    ) -> Result<()> {
879        let mut write = |buf| write(buf, packed_code_sizes, packed_pos);
880        let counts = &mut h.count[HUFF_CODES_TABLE];
881        if self.repeat_count != 0 {
882            if self.repeat_count < 3 {
883                counts[self.prev_code_size as usize] =
884                    counts[self.prev_code_size as usize].wrapping_add(self.repeat_count);
885                let code = self.prev_code_size;
886                write(&[code, code, code][..self.repeat_count as usize])?;
887            } else {
888                counts[16] = counts[16].wrapping_add(1);
889                write(&[16, (self.repeat_count - 3) as u8][..])?;
890            }
891            self.repeat_count = 0;
892        }
893
894        Ok(())
895    }
896
897    fn zero_code_size(
898        &mut self,
899        packed_code_sizes: &mut [u8],
900        packed_pos: &mut usize,
901        h: &mut HuffmanOxide,
902    ) -> Result<()> {
903        let mut write = |buf| write(buf, packed_code_sizes, packed_pos);
904        let counts = &mut h.count[HUFF_CODES_TABLE];
905        if self.z_count != 0 {
906            if self.z_count < 3 {
907                counts[0] = counts[0].wrapping_add(self.z_count as u16);
908                write(&[0, 0, 0][..self.z_count as usize])?;
909            } else if self.z_count <= 10 {
910                counts[17] = counts[17].wrapping_add(1);
911                write(&[17, (self.z_count - 3) as u8][..])?;
912            } else {
913                counts[18] = counts[18].wrapping_add(1);
914                write(&[18, (self.z_count - 11) as u8][..])?;
915            }
916            self.z_count = 0;
917        }
918
919        Ok(())
920    }
921}

942impl HuffmanOxide {
943    fn radix_sort_symbols<'a>(
944        symbols0: &'a mut [SymFreq],
945        symbols1: &'a mut [SymFreq],
946    ) -> &'a mut [SymFreq] {
947        let mut hist = [[0; 256]; 2];
948
949        for freq in symbols0.iter() {
950            hist[0][(freq.key & 0xFF) as usize] += 1;
951            hist[1][((freq.key >> 8) & 0xFF) as usize] += 1;
952        }
953
954        let mut n_passes = 2;
955        if symbols0.len() == hist[1][0] {
956            n_passes -= 1;
957        }
958
959        let mut current_symbols = symbols0;
960        let mut new_symbols = symbols1;
961
962        for (pass, hist_item) in hist.iter().enumerate().take(n_passes) {
963            let mut offsets = [0; 256];
964            let mut offset = 0;
965            for i in 0..256 {
966                offsets[i] = offset;
967                offset += hist_item[i];
968            }
969
970            for sym in current_symbols.iter() {
971                let j = ((sym.key >> (pass * 8)) & 0xFF) as usize;
972                new_symbols[offsets[j]] = *sym;
973                offsets[j] += 1;
974            }
975
976            mem::swap(&mut current_symbols, &mut new_symbols);
977        }
978
979        current_symbols
980    }
981
982    fn calculate_minimum_redundancy(symbols: &mut [SymFreq]) {
983        match symbols.len() {
984            0 => (),
985            1 => symbols[0].key = 1,
986            n => {
987                symbols[0].key += symbols[1].key;
988                let mut root = 0;
989                let mut leaf = 2;
990                for next in 1..n - 1 {
991                    if (leaf >= n) || (symbols[root].key < symbols[leaf].key) {
992                        symbols[next].key = symbols[root].key;
993                        symbols[root].key = next as u16;
994                        root += 1;
995                    } else {
996                        symbols[next].key = symbols[leaf].key;
997                        leaf += 1;
998                    }
999
1000                    if (leaf >= n) || (root < next && symbols[root].key < symbols[leaf].key) {
1001                        symbols[next].key = symbols[next].key.wrapping_add(symbols[root].key);
1002                        symbols[root].key = next as u16;
1003                        root += 1;
1004                    } else {
1005                        symbols[next].key = symbols[next].key.wrapping_add(symbols[leaf].key);
1006                        leaf += 1;
1007                    }
1008                }
1009
1010                symbols[n - 2].key = 0;
1011                for next in (0..n - 2).rev() {
1012                    symbols[next].key = symbols[symbols[next].key as usize].key + 1;
1013                }
1014
1015                let mut avbl = 1;
1016                let mut used = 0;
1017                let mut dpth = 0;
1018                let mut root = (n - 2) as i32;
1019                let mut next = (n - 1) as i32;
1020                while avbl > 0 {
1021                    while (root >= 0) && (symbols[root as usize].key == dpth) {
1022                        used += 1;
1023                        root -= 1;
1024                    }
1025                    while avbl > used {
1026                        symbols[next as usize].key = dpth;
1027                        next -= 1;
1028                        avbl -= 1;
1029                    }
1030                    avbl = 2 * used;
1031                    dpth += 1;
1032                    used = 0;
1033                }
1034            }
1035        }
1036    }

1038    fn enforce_max_code_size(num_codes: &mut [i32], code_list_len: usize, max_code_size: usize) {
1039        if code_list_len <= 1 {
1040            return;
1041        }
1042
1043        num_codes[max_code_size] += num_codes[max_code_size + 1..].iter().sum::<i32>();
1044        let total = num_codes[1..=max_code_size]
1045            .iter()
1046            .rev()
1047            .enumerate()
1048            .fold(0u32, |total, (i, &x)| total + ((x as u32) << i));
1049
1050        for _ in (1 << max_code_size)..total {
1051            num_codes[max_code_size] -= 1;
1052            for i in (1..max_code_size).rev() {
1053                if num_codes[i] != 0 {
1054                    num_codes[i] -= 1;
1055                    num_codes[i + 1] += 2;
1056                    break;
1057                }
1058            }
1059        }
1060    }
1061
1062    fn optimize_table(
1063        &mut self,
1064        table_num: usize,
1065        table_len: usize,
1066        code_size_limit: usize,
1067        static_table: bool,
1068    ) {
1069        let mut num_codes = [0i32; 32 + 1];
1070        let mut next_code = [0u32; MAX_SUPPORTED_HUFF_CODESIZE + 1];
1071
1072        if static_table {
1073            for &code_size in &self.code_sizes[table_num][..table_len] {
1074                num_codes[code_size as usize] += 1;
1075            }
1076        } else {
1077            let mut symbols0 = [SymFreq {
1078                key: 0,
1079                sym_index: 0,
1080            }; MAX_HUFF_SYMBOLS];
1081            let mut symbols1 = [SymFreq {
1082                key: 0,
1083                sym_index: 0,
1084            }; MAX_HUFF_SYMBOLS];
1085
1086            let mut num_used_symbols = 0;
1087            for i in 0..table_len {
1088                if self.count[table_num][i] != 0 {
1089                    symbols0[num_used_symbols] = SymFreq {
1090                        key: self.count[table_num][i],
1091                        sym_index: i as u16,
1092                    };
1093                    num_used_symbols += 1;
1094                }
1095            }
1096
1097            let symbols = Self::radix_sort_symbols(
1098                &mut symbols0[..num_used_symbols],
1099                &mut symbols1[..num_used_symbols],
1100            );
1101            Self::calculate_minimum_redundancy(symbols);
1102
1103            for symbol in symbols.iter() {
1104                num_codes[symbol.key as usize] += 1;
1105            }
1106
1107            Self::enforce_max_code_size(&mut num_codes, num_used_symbols, code_size_limit);
1108
1109            self.code_sizes[table_num].fill(0);
1110            self.codes[table_num].fill(0);
1111
1112            let mut last = num_used_symbols;
1113            for (i, &num_item) in num_codes
1114                .iter()
1115                .enumerate()
1116                .take(code_size_limit + 1)
1117                .skip(1)
1118            {
1119                let first = last - num_item as usize;
1120                for symbol in &symbols[first..last] {
1121                    self.code_sizes[table_num][symbol.sym_index as usize] = i as u8;
1122                }
1123                last = first;
1124            }
1125        }
1126
1127        let mut j = 0;
1128        next_code[1] = 0;
1129        for i in 2..=code_size_limit {
1130            j = (j + num_codes[i - 1]) << 1;
1131            next_code[i] = j as u32;
1132        }
1133
1134        for (&code_size, huff_code) in self.code_sizes[table_num]
1135            .iter()
1136            .take(table_len)
1137            .zip(self.codes[table_num].iter_mut().take(table_len))
1138        {
1139            if code_size == 0 {
1140                continue;
1141            }
1142
1143            let code = next_code[code_size as usize];
1144
1145            next_code[code_size as usize] += 1;
1146
1147            let rev_code = (code as u16).reverse_bits() >> (16 - code_size);
1148
1149            *huff_code = rev_code;
1150        }
1151    }
1152
1153    fn start_static_block(&mut self, output: &mut OutputBufferOxide) {
1154        self.code_sizes[LITLEN_TABLE][0..144].fill(8);
1155        self.code_sizes[LITLEN_TABLE][144..256].fill(9);
1156        self.code_sizes[LITLEN_TABLE][256..280].fill(7);
1157        self.code_sizes[LITLEN_TABLE][280..288].fill(8);
1158
1159        self.code_sizes[DIST_TABLE][..32].fill(5);
1160
1161        self.optimize_table(LITLEN_TABLE, 288, 15, true);
1162        self.optimize_table(DIST_TABLE, 32, 15, true);
1163
1164        output.put_bits(0b01, 2)
1165    }
1166
1167    fn start_dynamic_block(&mut self, output: &mut OutputBufferOxide) -> Result<()> {
1168        // There will always be one, and only one end of block code.
1169        self.count[0][256] = 1;
1170
1171        self.optimize_table(0, MAX_HUFF_SYMBOLS_0, 15, false);
1172        self.optimize_table(1, MAX_HUFF_SYMBOLS_1, 15, false);
1173
1174        let num_lit_codes = 286
1175            - &self.code_sizes[0][257..286]
1176                .iter()
1177                .rev()
1178                .take_while(|&x| *x == 0)
1179                .count();
1180
1181        let num_dist_codes = 30
1182            - &self.code_sizes[1][1..30]
1183                .iter()
1184                .rev()
1185                .take_while(|&x| *x == 0)
1186                .count();
1187
1188        let mut code_sizes_to_pack = [0u8; MAX_HUFF_SYMBOLS_0 + MAX_HUFF_SYMBOLS_1];
1189        let mut packed_code_sizes = [0u8; MAX_HUFF_SYMBOLS_0 + MAX_HUFF_SYMBOLS_1];
1190
1191        let total_code_sizes_to_pack = num_lit_codes + num_dist_codes;
1192
1193        code_sizes_to_pack[..num_lit_codes].copy_from_slice(&self.code_sizes[0][..num_lit_codes]);
1194
1195        code_sizes_to_pack[num_lit_codes..total_code_sizes_to_pack]
1196            .copy_from_slice(&self.code_sizes[1][..num_dist_codes]);
1197
1198        let mut rle = Rle {
1199            z_count: 0,
1200            repeat_count: 0,
1201            prev_code_size: 0xFF,
1202        };
1203
1204        self.count[HUFF_CODES_TABLE][..MAX_HUFF_SYMBOLS_2].fill(0);
1205
1206        let mut packed_pos = 0;
1207        for &code_size in &code_sizes_to_pack[..total_code_sizes_to_pack] {
1208            if code_size == 0 {
1209                rle.prev_code_size(&mut packed_code_sizes, &mut packed_pos, self)?;
1210                rle.z_count += 1;
1211                if rle.z_count == 138 {
1212                    rle.zero_code_size(&mut packed_code_sizes, &mut packed_pos, self)?;
1213                }
1214            } else {
1215                rle.zero_code_size(&mut packed_code_sizes, &mut packed_pos, self)?;
1216                if code_size != rle.prev_code_size {
1217                    rle.prev_code_size(&mut packed_code_sizes, &mut packed_pos, self)?;
1218                    self.count[HUFF_CODES_TABLE][code_size as usize] =
1219                        self.count[HUFF_CODES_TABLE][code_size as usize].wrapping_add(1);
1220                    write(&[code_size], &mut packed_code_sizes, &mut packed_pos)?;
1221                } else {
1222                    rle.repeat_count += 1;
1223                    if rle.repeat_count == 6 {
1224                        rle.prev_code_size(&mut packed_code_sizes, &mut packed_pos, self)?;
1225                    }
1226                }
1227            }
1228            rle.prev_code_size = code_size;
1229        }
1230
1231        if rle.repeat_count != 0 {
1232            rle.prev_code_size(&mut packed_code_sizes, &mut packed_pos, self)?;
1233        } else {
1234            rle.zero_code_size(&mut packed_code_sizes, &mut packed_pos, self)?;
1235        }
1236
1237        self.optimize_table(2, MAX_HUFF_SYMBOLS_2, 7, false);
1238
1239        output.put_bits_no_flush(2, 2);
1240
1241        output.put_bits_no_flush((num_lit_codes - 257) as u32, 5);
1242        output.put_bits_no_flush((num_dist_codes - 1) as u32, 5);
1243
1244        let mut num_bit_lengths = 18
1245            - HUFFMAN_LENGTH_ORDER
1246                .iter()
1247                .rev()
1248                .take_while(|&swizzle| self.code_sizes[HUFF_CODES_TABLE][*swizzle as usize] == 0)
1249                .count();
1250
1251        num_bit_lengths = cmp::max(4, num_bit_lengths + 1);
1252        output.put_bits(num_bit_lengths as u32 - 4, 4);
1253        for &swizzle in &HUFFMAN_LENGTH_ORDER[..num_bit_lengths] {
1254            output.put_bits(
1255                u32::from(self.code_sizes[HUFF_CODES_TABLE][swizzle as usize]),
1256                3,
1257            );
1258        }
1259
1260        let mut packed_code_size_index = 0;
1261        while packed_code_size_index < packed_pos {
1262            let code = packed_code_sizes[packed_code_size_index] as usize;
1263            packed_code_size_index += 1;
1264            assert!(code < MAX_HUFF_SYMBOLS_2);
1265            output.put_bits(
1266                u32::from(self.codes[HUFF_CODES_TABLE][code]),
1267                u32::from(self.code_sizes[HUFF_CODES_TABLE][code]),
1268            );
1269            if code >= 16 {
1270                output.put_bits(
1271                    u32::from(packed_code_sizes[packed_code_size_index]),
1272                    [2, 3, 7][code - 16],
1273                );
1274                packed_code_size_index += 1;
1275            }
1276        }
1277
1278        Ok(())
1279    }
1280}

1284    /// The maximum number of checks in the hash chain, for the initial,
1285    /// and the lazy match respectively.
1286    pub max_probes: [u32; 2],
1287    /// Buffer of input data.
1288    /// Padded with 1 byte to simplify matching code in `compress_fast`.
1289    pub b: HashBuffers,
1290

1298const fn probes_from_flags(flags: u32) -> [u32; 2] {
1299    [
1300        1 + ((flags & 0xFFF) + 2) / 3,
1301        1 + (((flags & 0xFFF) >> 2) + 2) / 3,
1302    ]
1303}
1304
1305impl DictOxide {
1306    fn new(flags: u32) -> Self {
1307        DictOxide {
1308            max_probes: probes_from_flags(flags),
1309            b: HashBuffers::default(),
1310            code_buf_dict_pos: 0,
1311            lookahead_size: 0,
1312            lookahead_pos: 0,
1313            size: 0,
1314            loop_len: 32,
1315        }
1316    }
1317
1318    fn update_flags(&mut self, flags: u32) {
1319        self.max_probes = probes_from_flags(flags);
1320    }
1321
1322    fn reset(&mut self) {
1323        self.b.reset();
1324        self.code_buf_dict_pos = 0;
1325        self.lookahead_size = 0;
1326        self.lookahead_pos = 0;
1327        self.size = 0;
1328    }
1329
1330    /// Do an unaligned read of the data at `pos` in the dictionary and treat it as if it was of
1331    /// type T.
1332    #[inline]
1333    fn read_unaligned_u32(&self, pos: usize) -> u32 {
1334        // Masking the value here helps avoid bounds checks.
1335        let pos = pos & LZ_DICT_SIZE_MASK;
1336        let end = pos + 4;
1337        // Somehow this assertion makes things faster.
1338        // TODO: as of may 2024 this does not seem to make any difference
1339        // so consider removing.
1340        assert!(end < LZ_DICT_FULL_SIZE);
1341
1342        let bytes: [u8; 4] = self.b.dict[pos..end].try_into().unwrap();
1343        u32::from_le_bytes(bytes)
1344    }
1345
1346    /// Do an unaligned read of the data at `pos` in the dictionary and treat it as if it was of
1347    /// type T.
1348    #[inline]
1349    fn read_unaligned_u64(&self, pos: usize) -> u64 {
1350        // Help evade bounds/panic code check by masking the position value
1351        // This provides a small speedup at the cost of an instruction or two instead of
1352        // having to use unsafe.
1353        let pos = pos & LZ_DICT_SIZE_MASK;
1354        let bytes: [u8; 8] = self.b.dict[pos..pos + 8].try_into().unwrap();
1355        u64::from_le_bytes(bytes)
1356    }
1357
1358    /// Try to find a match for the data at lookahead_pos in the dictionary that is
1359    /// longer than `match_len`.
1360    /// Returns a tuple containing (match_distance, match_length). Will be equal to the input
1361    /// values if no better matches were found.
1362    fn find_match(
1363        &self,
1364        lookahead_pos: usize,
1365        max_dist: usize,
1366        max_match_len: u32,
1367        mut match_dist: u32,
1368        mut match_len: u32,
1369    ) -> (u32, u32) {
1370        // Clamp the match len and max_match_len to be valid. (It should be when this is called, but
1371        // do it for now just in case for safety reasons.)
1372        // This should normally end up as at worst conditional moves,
1373        // so it shouldn't slow us down much.
1374        // TODO: Statically verify these so we don't need to do this.
1375        let max_match_len = cmp::min(MAX_MATCH_LEN as u32, max_match_len);
1376        match_len = cmp::max(match_len, 1);
1377
1378        // If we already have a match of the full length don't bother searching for another one.
1379        if max_match_len <= match_len {
1380            return (match_dist, match_len);
1381        }
1382
1383        let pos = lookahead_pos & LZ_DICT_SIZE_MASK;
1384        let mut probe_pos = pos;
1385        // Number of probes into the hash chains.
1386        let mut num_probes_left = if match_len < 32 {
1387            self.max_probes[0]
1388        } else {
1389            self.max_probes[1]
1390        };
1391
1392        // Read the last byte of the current match, and the next one, used to compare matches.
1393        let mut c01: u16 = read_u16_le(&self.b.dict, pos + match_len as usize - 1);
1394        // Read the two bytes at the end position of the current match.
1395        let s01: u16 = read_u16_le(&self.b.dict, pos);
1396
1397        'outer: loop {
1398            let mut dist;
1399            'found: loop {
1400                num_probes_left -= 1;
1401                if num_probes_left == 0 {
1402                    // We have done as many probes in the hash chain as the current compression
1403                    // settings allow, so return the best match we found, if any.
1404                    return (match_dist, match_len);
1405                }
1406
1407                for _ in 0..3 {
1408                    let next_probe_pos = self.b.next[probe_pos] as usize;
1409
1410                    dist = (lookahead_pos - next_probe_pos) & 0xFFFF;
1411                    // Optimization: The last condition should never be hit but helps the compiler by avoiding
1412                    // doing the bounds check in the read_u16_le call and adding the extra instructions
1413                    // for branching to a panic after that and instead just adds the extra instruction here
1414                    // instead saving some instructions and thus improving performance a bit.
1415                    // May want to investigate whether we can avoid it entirely but as of now the compiler
1416                    // isn't able to deduce that match_len - 1 is bounded to [1-257]
1417                    // Disable clippy lint as it needs to be written in this specific way
1418                    // rather than MAX_MATCH_LEN to work
1419                    // because the compiler isn't super smart....
1420                    #[allow(clippy::int_plus_one)]
1421                    if next_probe_pos == 0
1422                        || dist > max_dist
1423                        || match_len as usize - 1 >= MAX_MATCH_LEN
1424                    {
1425                        // We reached the end of the hash chain, or the next value is further away
1426                        // than the maximum allowed distance, so return the best match we found, if
1427                        // any.
1428                        return (match_dist, match_len);
1429                    }
1430
1431                    // Mask the position value to get the position in the hash chain of the next
1432                    // position to match against.
1433                    probe_pos = next_probe_pos & LZ_DICT_SIZE_MASK;
1434
1435                    if read_u16_le(&self.b.dict, probe_pos + match_len as usize - 1) == c01 {
1436                        break 'found;
1437                    }
1438                }
1439            }
1440
1441            if dist == 0 {
1442                // We've looked through the whole match range, so return the best match we
1443                // found.
1444                return (match_dist, match_len);
1445            }
1446
1447            // Check if the two first bytes match.
1448            if read_u16_le(&self.b.dict, probe_pos) != s01 {
1449                continue;
1450            }
1451
1452            let mut p = pos + 2;
1453            let mut q = probe_pos + 2;
1454            // The first two bytes matched, so check the full length of the match.
1455            // TODO: This is a workaround for an upstream issue introduced after a LLVM upgrade in rust 1.82.
1456            // the compiler is too smart and ends up unrolling the loop which causes the performance to get worse
1457            // Using a variable instead of a constant here to prevent it seems to at least get back some of the performance loss.
1458            for _ in 0..self.loop_len as i32 {
1459                let p_data: u64 = self.read_unaligned_u64(p);
1460                let q_data: u64 = self.read_unaligned_u64(q);
1461                // Compare of 8 bytes at a time by using unaligned loads of 64-bit integers.
1462                let xor_data = p_data ^ q_data;
1463                if xor_data == 0 {
1464                    p += 8;
1465                    q += 8;
1466                } else {
1467                    // If not all of the last 8 bytes matched, check how may of them did.
1468                    let trailing = xor_data.trailing_zeros();
1469
1470                    let probe_len = p - pos + (trailing as usize >> 3);
1471                    if probe_len > match_len as usize {
1472                        match_dist = dist as u32;
1473                        match_len = cmp::min(max_match_len, probe_len as u32);
1474                        if match_len >= max_match_len {
1475                            // We found a match that had the maximum allowed length,
1476                            // so there is now point searching further.
1477                            return (match_dist, match_len);
1478                        }
1479                        // We found a better match, so save the last two bytes for further match
1480                        // comparisons.
1481                        // Optimization: use saturating_sub makes the compiler able to evade the bounds check
1482                        // at the cost of some extra instructions since it avoids any possibility of wraparound.
1483                        // need to see if we can find a better way to do this since this is still a bit costly.
1484                        c01 =
1485                            read_u16_le(&self.b.dict, (pos + match_len as usize).saturating_sub(1));
1486                    }
1487                    continue 'outer;
1488                }
1489            }
1490
1491            return (dist as u32, cmp::min(max_match_len, MAX_MATCH_LEN as u32));
1492        }
1493    }
1494}

1637fn compress_lz_codes(
1638    huff: &HuffmanOxide,
1639    output: &mut OutputBufferOxide,
1640    lz_code_buf: &[u8; LZ_CODE_BUF_SIZE],
1641    lz_code_buf_used_len: usize,
1642) -> Result<bool> {
1643    let mut flags = 1;
1644    let mut bb = BitBuffer {
1645        bit_buffer: u64::from(output.bit_buffer),
1646        bits_in: output.bits_in,
1647    };
1648
1649    // Help out the compiler know this variable won't be larger than
1650    // the buffer length since the constants won't propagate through the function call.
1651    let lz_code_buf_used_len = cmp::min(lz_code_buf.len(), lz_code_buf_used_len);
1652
1653    let mut i: usize = 0;
1654    while i < lz_code_buf_used_len {
1655        if flags == 1 {
1656            flags = u32::from(lz_code_buf[i]) | 0x100;
1657            i += 1;
1658        }
1659
1660        // The lz code was a length code
1661        if flags & 1 == 1 {
1662            flags >>= 1;
1663
1664            let sym;
1665            let num_extra_bits;
1666
1667            let match_len = lz_code_buf[i & LZ_CODE_BUF_MASK] as usize;
1668
1669            let match_dist = lz_code_buf[(i + 1) & LZ_CODE_BUF_MASK] as u16
1670                | ((lz_code_buf[(i + 2) & LZ_CODE_BUF_MASK] as u16) << 8);
1671
1672            i += 3;
1673
1674            debug_assert!(huff.code_sizes[0][LEN_SYM[match_len] as usize + LEN_SYM_OFFSET] != 0);
1675            let len_sym = (LEN_SYM[match_len] & 31) as usize + LEN_SYM_OFFSET;
1676
1677            bb.put_fast(
1678                u64::from(huff.codes[0][len_sym]),
1679                u32::from(huff.code_sizes[0][len_sym]),
1680            );
1681            bb.put_fast(
1682                match_len as u64 & u64::from(BITMASKS[(LEN_EXTRA[match_len] & 7) as usize]),
1683                u32::from(LEN_EXTRA[match_len]),
1684            );
1685
1686            if match_dist < 512 {
1687                sym = SMALL_DIST_SYM[match_dist as usize] as usize;
1688                num_extra_bits = SMALL_DIST_EXTRA[(match_dist >> 2) as usize] as usize;
1689            } else {
1690                sym = LARGE_DIST_SYM[(match_dist >> 8) as usize] as usize;
1691                num_extra_bits = LARGE_DIST_EXTRA[(match_dist >> 8) as usize] as usize;
1692            }
1693
1694            debug_assert!(huff.code_sizes[1][sym] != 0);
1695            bb.put_fast(
1696                u64::from(huff.codes[1][sym]),
1697                u32::from(huff.code_sizes[1][sym]),
1698            );
1699            bb.put_fast(
1700                u64::from(match_dist) & u64::from(BITMASKS[num_extra_bits & 15]),
1701                num_extra_bits as u32,
1702            );
1703        } else {
1704            // The lz code was a literal
1705            for _ in 0..3 {
1706                flags >>= 1;
1707                let lit = lz_code_buf[i & LZ_CODE_BUF_MASK];
1708                i += 1;
1709
1710                debug_assert!(huff.code_sizes[0][lit as usize] != 0);
1711                bb.put_fast(
1712                    u64::from(huff.codes[0][lit as usize]),
1713                    u32::from(huff.code_sizes[0][lit as usize]),
1714                );
1715
1716                if flags & 1 == 1 || i >= lz_code_buf_used_len {
1717                    break;
1718                }
1719            }
1720        }
1721
1722        bb.flush(output)?;
1723    }
1724
1725    output.bits_in = 0;
1726    output.bit_buffer = 0;
1727    while bb.bits_in != 0 {
1728        let n = cmp::min(bb.bits_in, 16);
1729        output.put_bits(bb.bit_buffer as u32 & BITMASKS[n as usize], n);
1730        bb.bit_buffer >>= n;
1731        bb.bits_in -= n;
1732    }
1733
1734    // Output the end of block symbol.
1735    output.put_bits(
1736        u32::from(huff.codes[0][256]),
1737        u32::from(huff.code_sizes[0][256]),
1738    );
1739
1740    Ok(true)
1741}

1743fn compress_block(
1744    huff: &mut HuffmanOxide,
1745    output: &mut OutputBufferOxide,
1746    lz: &LZOxide,
1747    static_block: bool,
1748) -> Result<bool> {
1749    if static_block {
1750        huff.start_static_block(output);
1751    } else {
1752        huff.start_dynamic_block(output)?;
1753    }
1754
1755    compress_lz_codes(huff, output, &lz.codes, lz.code_position)
1756}

1758pub(crate) fn flush_block(
1759    d: &mut CompressorOxide,
1760    callback: &mut CallbackOxide,
1761    flush: TDEFLFlush,
1762) -> Result<i32> {
1763    let mut saved_buffer;
1764    {
1765        let mut output = callback
1766            .out
1767            .new_output_buffer(&mut d.params.local_buf.b, d.params.out_buf_ofs);
1768        output.bit_buffer = d.params.saved_bit_buffer;
1769        output.bits_in = d.params.saved_bits_in;
1770
1771        // If we are at the start of the stream, write the zlib header
1772        // if requested.  Note: Even if block-writing is skipped
1773        // below, `block_index` is still incremented, so this is done
1774        // only once
1775        if d.params.flags & TDEFL_WRITE_ZLIB_HEADER != 0 && d.params.block_index == 0 {
1776            let header = zlib::header_from_flags(d.params.flags, d.params.window_bits_max);
1777            output.put_bits_no_flush(header[0].into(), 8);
1778            output.put_bits(header[1].into(), 8);
1779        }
1780
1781        if d.lz.total_bytes > 0 || flush == TDEFLFlush::Finish {
1782            // TODO: Don't think this second condition should be here but need to verify.
1783            let use_raw_block = (d.params.flags & TDEFL_FORCE_ALL_RAW_BLOCKS != 0)
1784                && (d.dict.lookahead_pos - d.dict.code_buf_dict_pos) <= d.dict.size;
1785            debug_assert_eq!(
1786                use_raw_block,
1787                d.params.flags & TDEFL_FORCE_ALL_RAW_BLOCKS != 0
1788            );
1789
1790            debug_assert!(d.params.flush_remaining == 0);
1791            d.params.flush_ofs = 0;
1792            d.params.flush_remaining = 0;
1793
1794            d.lz.init_flag();
1795
1796            // Output the block header.
1797            output.put_bits((flush == TDEFLFlush::Finish) as u32, 1);
1798
1799            saved_buffer = output.save();
1800
1801            let comp_success = if !use_raw_block {
1802                let use_static = (d.params.flags & TDEFL_FORCE_ALL_STATIC_BLOCKS != 0)
1803                    || (d.lz.total_bytes < 48);
1804                compress_block(&mut d.huff, &mut output, &d.lz, use_static)?
1805            } else {
1806                false
1807            };
1808
1809            // If we failed to compress anything and the output would take up more space than the output
1810            // data, output a stored block instead, which has at most 5 bytes of overhead.
1811            // We only use some simple heuristics for now.
1812            // A stored block will have an overhead of at least 4 bytes containing the block length
1813            // but usually more due to the length parameters having to start at a byte boundary and thus
1814            // requiring up to 5 bytes of padding.
1815            // As a static block will have an overhead of at most 1 bit per byte
1816            // (as literals are either 8 or 9 bytes), a raw block will
1817            // never take up less space if the number of input bytes are less than 32.
1818            let expanded = (d.lz.total_bytes > 32)
1819                && (output.inner_pos - saved_buffer.pos + 1 >= (d.lz.total_bytes as usize))
1820                && (d.dict.lookahead_pos - d.dict.code_buf_dict_pos <= d.dict.size);
1821
1822            if use_raw_block || expanded {
1823                output.load(saved_buffer);
1824
1825                // Block header.
1826                output.put_bits(0, 2);
1827
1828                // Block length has to start on a byte boundary, so pad.
1829                output.pad_to_bytes();
1830
1831                // Block length and ones complement of block length.
1832                output.put_bits(d.lz.total_bytes & 0xFFFF, 16);
1833                output.put_bits(!d.lz.total_bytes & 0xFFFF, 16);
1834
1835                // Write the actual bytes.
1836                let start = d.dict.code_buf_dict_pos & LZ_DICT_SIZE_MASK;
1837                let end =
1838                    (d.dict.code_buf_dict_pos + d.lz.total_bytes as usize) & LZ_DICT_SIZE_MASK;
1839                let dict = &mut d.dict.b.dict;
1840                if start < end {
1841                    // The data does not wrap around.
1842                    output.write_bytes(&dict[start..end]);
1843                } else if d.lz.total_bytes > 0 {
1844                    // The data wraps around and the input was not 0 bytes.
1845                    output.write_bytes(&dict[start..LZ_DICT_SIZE]);
1846                    output.write_bytes(&dict[..end]);
1847                }
1848            } else if !comp_success {
1849                output.load(saved_buffer);
1850                compress_block(&mut d.huff, &mut output, &d.lz, true)?;
1851            }
1852        }
1853
1854        match flush {
1855            TDEFLFlush::Finish => {
1856                output.pad_to_bytes();
1857                if d.params.flags & TDEFL_WRITE_ZLIB_HEADER != 0 {
1858                    let mut adler = d.params.adler32;
1859                    for _ in 0..4 {
1860                        output.put_bits((adler >> 24) & 0xFF, 8);
1861                        adler <<= 8;
1862                    }
1863                }
1864            }
1865            TDEFLFlush::Partial => {
1866                output.put_bits(2, 10);
1867            }
1868            TDEFLFlush::PartialOpt => {
1869                if !output.is_byte_aligned() {
1870                    output.put_bits(2, 10);
1871                }
1872            }
1873            TDEFLFlush::Sync | TDEFLFlush::Full => {
1874                // Output an empty raw block.
1875                output.put_bits(0, 3);
1876                output.pad_to_bytes();
1877                output.put_bits(0, 16);
1878                output.put_bits(0xFFFF, 16);
1879            }
1880            TDEFLFlush::SyncOpt => {
1881                if !output.is_byte_aligned() {
1882                    output.put_bits(0, 3);
1883                    output.pad_to_bytes();
1884                    output.put_bits(0, 16);
1885                    output.put_bits(0xFFFF, 16);
1886                }
1887            }
1888            TDEFLFlush::None | TDEFLFlush::NoSync => (),
1889        }
1890
1891        d.huff.count[0][..MAX_HUFF_SYMBOLS_0].fill(0);
1892        d.huff.count[1][..MAX_HUFF_SYMBOLS_1].fill(0);
1893
1894        // Clear LZ buffer for the next block.
1895        d.lz.code_position = 1;
1896        d.lz.flag_position = 0;
1897        d.lz.num_flags_left = 8;
1898        d.dict.code_buf_dict_pos += d.lz.total_bytes as usize;
1899        d.lz.total_bytes = 0;
1900        d.params.block_index += 1;
1901
1902        saved_buffer = output.save();
1903
1904        d.params.saved_bit_buffer = saved_buffer.bit_buffer;
1905        d.params.saved_bits_in = saved_buffer.bits_in;
1906    }
1907
1908    Ok(callback.flush_output(saved_buffer, &mut d.params))
1909}

1911pub(crate) fn record_literal(h: &mut HuffmanOxide, lz: &mut LZOxide, lit: u8) {
1912    lz.total_bytes += 1;
1913    lz.write_code(lit);
1914
1915    *lz.get_flag() >>= 1;
1916    lz.consume_flag();
1917
1918    h.count[0][lit as usize] += 1;
1919}

1921fn record_match(h: &mut HuffmanOxide, lz: &mut LZOxide, match_len: u32, mut match_dist: u32) {
1922    debug_assert!(match_len >= MIN_MATCH_LEN.into());
1923    debug_assert!(match_dist >= 1);
1924    debug_assert!(match_dist as usize <= LZ_DICT_SIZE);
1925
1926    lz.total_bytes += match_len;
1927    match_dist -= 1;
1928    let match_len = (match_len - u32::from(MIN_MATCH_LEN)) as u8;
1929    lz.write_code(match_len);
1930    lz.write_code(match_dist as u8);
1931    lz.write_code((match_dist >> 8) as u8);
1932
1933    *lz.get_flag() >>= 1;
1934    *lz.get_flag() |= 0x80;
1935    lz.consume_flag();
1936
1937    let symbol = if match_dist < 512 {
1938        SMALL_DIST_SYM[match_dist as usize]
1939    } else {
1940        LARGE_DIST_SYM[((match_dist >> 8) & 127) as usize]
1941    } as usize;
1942    h.count[1][symbol] += 1;
1943    // Mask the values from LEN_SYM here as the compiler isn't quite smart enough to infer
1944    // that it only contains values smaller than 32.
1945    h.count[0][(LEN_SYM[match_len as usize] as usize & 31) + LEN_SYM_OFFSET] += 1;
1946}

1948fn compress_normal(d: &mut CompressorOxide, callback: &mut CallbackOxide) -> bool {
1949    let in_buf = match callback.in_buf {
1950        None => return true,
1951        Some(in_buf) => in_buf,
1952    };
1953
1954    let mut src_pos = d.params.src_pos;
1955    let mut lookahead_size = d.dict.lookahead_size;
1956    let mut lookahead_pos = d.dict.lookahead_pos;
1957    let mut saved_lit = d.params.saved_lit;
1958    let mut saved_match_dist = d.params.saved_match_dist;
1959    let mut saved_match_len = d.params.saved_match_len;
1960
1961    while src_pos < in_buf.len() || (d.params.flush != TDEFLFlush::None && lookahead_size != 0) {
1962        let in_buf_left = &in_buf[src_pos..];
1963        let num_bytes_to_process = cmp::min(in_buf_left.len(), MAX_MATCH_LEN - lookahead_size);
1964        let bytes_to_process = &in_buf_left[..num_bytes_to_process];
1965
1966        if lookahead_size + d.dict.size >= usize::from(MIN_MATCH_LEN) - 1
1967            && num_bytes_to_process > 0
1968        {
1969            let dictb = &mut d.dict.b;
1970
1971            let mut dst_pos = (lookahead_pos + lookahead_size) & LZ_DICT_SIZE_MASK;
1972            let mut ins_pos = lookahead_pos + lookahead_size - 2;
1973            // Start the hash value from the first two bytes
1974            let mut hash = update_hash(
1975                u16::from(dictb.dict[ins_pos & LZ_DICT_SIZE_MASK]),
1976                dictb.dict[(ins_pos + 1) & LZ_DICT_SIZE_MASK],
1977            );
1978
1979            lookahead_size += num_bytes_to_process;
1980
1981            for &c in bytes_to_process {
1982                // Add byte to input buffer.
1983                dictb.dict[dst_pos] = c;
1984                if dst_pos < MAX_MATCH_LEN - 1 {
1985                    dictb.dict[LZ_DICT_SIZE + dst_pos] = c;
1986                }
1987
1988                // Generate hash from the current byte,
1989                hash = update_hash(hash, c);
1990                dictb.next[ins_pos & LZ_DICT_SIZE_MASK] = dictb.hash[hash as usize];
1991                // and insert it into the hash chain.
1992                dictb.hash[hash as usize] = ins_pos as u16;
1993                dst_pos = (dst_pos + 1) & LZ_DICT_SIZE_MASK;
1994                ins_pos += 1;
1995            }
1996
1997            src_pos += num_bytes_to_process;
1998        } else {
1999            let dictb = &mut d.dict.b;
2000            for &c in bytes_to_process {
2001                let dst_pos = (lookahead_pos + lookahead_size) & LZ_DICT_SIZE_MASK;
2002                dictb.dict[dst_pos] = c;
2003                if dst_pos < MAX_MATCH_LEN - 1 {
2004                    dictb.dict[LZ_DICT_SIZE + dst_pos] = c;
2005                }
2006
2007                lookahead_size += 1;
2008                if lookahead_size + d.dict.size >= MIN_MATCH_LEN.into() {
2009                    let ins_pos = lookahead_pos + lookahead_size - 3;
2010                    let hash = ((u32::from(dictb.dict[ins_pos & LZ_DICT_SIZE_MASK])
2011                        << (LZ_HASH_SHIFT * 2))
2012                        ^ ((u32::from(dictb.dict[(ins_pos + 1) & LZ_DICT_SIZE_MASK])
2013                            << LZ_HASH_SHIFT)
2014                            ^ u32::from(c)))
2015                        & (LZ_HASH_SIZE as u32 - 1);
2016
2017                    dictb.next[ins_pos & LZ_DICT_SIZE_MASK] = dictb.hash[hash as usize];
2018                    dictb.hash[hash as usize] = ins_pos as u16;
2019                }
2020            }
2021
2022            src_pos += num_bytes_to_process;
2023        }
2024
2025        d.dict.size = cmp::min(LZ_DICT_SIZE - lookahead_size, d.dict.size);
2026        if d.params.flush == TDEFLFlush::None && lookahead_size < MAX_MATCH_LEN {
2027            break;
2028        }
2029
2030        let mut len_to_move = 1;
2031        let mut cur_match_dist = 0;
2032        let mut cur_match_len = if saved_match_len != 0 {
2033            saved_match_len
2034        } else {
2035            u32::from(MIN_MATCH_LEN) - 1
2036        };
2037        let cur_pos = lookahead_pos & LZ_DICT_SIZE_MASK;
2038        if d.params.flags & TDEFL_RLE_MATCHES != 0 {
2039            // If TDEFL_RLE_MATCHES is set, we only look for repeating sequences of the current byte.
2040            if d.dict.size != 0 {
2041                let c = d.dict.b.dict[(cur_pos.wrapping_sub(1)) & LZ_DICT_SIZE_MASK];
2042                cur_match_len = d.dict.b.dict[cur_pos..(cur_pos + lookahead_size)]
2043                    .iter()
2044                    .take_while(|&x| *x == c)
2045                    .count() as u32;
2046                if cur_match_len < MIN_MATCH_LEN.into() {
2047                    cur_match_len = 0
2048                } else {
2049                    cur_match_dist = 1
2050                }
2051            }
2052        } else {
2053            // Try to find a match for the bytes at the current position.
2054            let dist_len = d.dict.find_match(
2055                lookahead_pos,
2056                d.dict.size,
2057                lookahead_size as u32,
2058                cur_match_dist,
2059                cur_match_len,
2060            );
2061            cur_match_dist = dist_len.0;
2062            cur_match_len = dist_len.1;
2063        }
2064
2065        let far_and_small = cur_match_len == MIN_MATCH_LEN.into() && cur_match_dist >= 8 * 1024;
2066        let filter_small = d.params.flags & TDEFL_FILTER_MATCHES != 0 && cur_match_len <= 5;
2067        if far_and_small || filter_small || cur_pos == cur_match_dist as usize {
2068            cur_match_dist = 0;
2069            cur_match_len = 0;
2070        }
2071
2072        if saved_match_len != 0 {
2073            if cur_match_len > saved_match_len {
2074                record_literal(&mut d.huff, &mut d.lz, saved_lit);
2075                if cur_match_len >= 128 {
2076                    record_match(&mut d.huff, &mut d.lz, cur_match_len, cur_match_dist);
2077                    saved_match_len = 0;
2078                    len_to_move = cur_match_len as usize;
2079                } else {
2080                    saved_lit = d.dict.b.dict[cur_pos];
2081                    saved_match_dist = cur_match_dist;
2082                    saved_match_len = cur_match_len;
2083                }
2084            } else {
2085                record_match(&mut d.huff, &mut d.lz, saved_match_len, saved_match_dist);
2086                len_to_move = (saved_match_len - 1) as usize;
2087                saved_match_len = 0;
2088            }
2089        } else if cur_match_dist == 0 {
2090            record_literal(
2091                &mut d.huff,
2092                &mut d.lz,
2093                d.dict.b.dict[cmp::min(cur_pos, d.dict.b.dict.len() - 1)],
2094            );
2095        } else if d.params.greedy_parsing
2096            || (d.params.flags & TDEFL_RLE_MATCHES != 0)
2097            || cur_match_len >= 128
2098        {
2099            // If we are using lazy matching, check for matches at the next byte if the current
2100            // match was shorter than 128 bytes.
2101            record_match(&mut d.huff, &mut d.lz, cur_match_len, cur_match_dist);
2102            len_to_move = cur_match_len as usize;
2103        } else {
2104            saved_lit = d.dict.b.dict[cmp::min(cur_pos, d.dict.b.dict.len() - 1)];
2105            saved_match_dist = cur_match_dist;
2106            saved_match_len = cur_match_len;
2107        }
2108
2109        lookahead_pos += len_to_move;
2110        debug_assert!(lookahead_size >= len_to_move);
2111        lookahead_size -= len_to_move;
2112        d.dict.size = cmp::min(d.dict.size + len_to_move, LZ_DICT_SIZE);
2113
2114        let lz_buf_tight = d.lz.code_position > LZ_CODE_BUF_SIZE - 8;
2115        let fat = ((d.lz.code_position * 115) >> 7) >= d.lz.total_bytes as usize;
2116        let buf_fat = (d.lz.total_bytes > 31 * 1024) && fat;
2117
2118        if lz_buf_tight || buf_fat {
2119            d.params.src_pos = src_pos;
2120            // These values are used in flush_block, so we need to write them back here.
2121            d.dict.lookahead_size = lookahead_size;
2122            d.dict.lookahead_pos = lookahead_pos;
2123
2124            let n = flush_block(d, callback, TDEFLFlush::None)
2125                .unwrap_or(TDEFLStatus::PutBufFailed as i32);
2126            if n != 0 {
2127                d.params.saved_lit = saved_lit;
2128                d.params.saved_match_dist = saved_match_dist;
2129                d.params.saved_match_len = saved_match_len;
2130                return n > 0;
2131            }
2132        }
2133    }
2134
2135    d.params.src_pos = src_pos;
2136    d.dict.lookahead_size = lookahead_size;
2137    d.dict.lookahead_pos = lookahead_pos;
2138    d.params.saved_lit = saved_lit;
2139    d.params.saved_match_dist = saved_match_dist;
2140    d.params.saved_match_len = saved_match_len;
2141    true
2142}

2144const COMP_FAST_LOOKAHEAD_SIZE: usize = 4096;
2145
2146fn compress_fast(d: &mut CompressorOxide, callback: &mut CallbackOxide) -> bool {
2147    let mut src_pos = d.params.src_pos;
2148    let mut lookahead_size = d.dict.lookahead_size;
2149    let mut lookahead_pos = d.dict.lookahead_pos;
2150
2151    let mut cur_pos = lookahead_pos & LZ_DICT_SIZE_MASK;
2152    let in_buf = match callback.in_buf {
2153        None => return true,
2154        Some(in_buf) => in_buf,
2155    };
2156
2157    debug_assert!(d.lz.code_position < LZ_CODE_BUF_SIZE - 2);
2158
2159    while src_pos < in_buf.len() || (d.params.flush != TDEFLFlush::None && lookahead_size > 0) {
2160        let mut dst_pos = (lookahead_pos + lookahead_size) & LZ_DICT_SIZE_MASK;
2161        let mut num_bytes_to_process = cmp::min(
2162            in_buf.len() - src_pos,
2163            COMP_FAST_LOOKAHEAD_SIZE - lookahead_size,
2164        );
2165        lookahead_size += num_bytes_to_process;
2166
2167        while num_bytes_to_process != 0 {
2168            let n = cmp::min(LZ_DICT_SIZE - dst_pos, num_bytes_to_process);

2347fn flush_output_buffer(c: &mut CallbackOxide, p: &mut ParamsOxide) -> (TDEFLStatus, usize, usize) {
2348    let mut res = (TDEFLStatus::Okay, p.src_pos, 0);
2349    if let CallbackOut::Buf(ref mut cb) = c.out {
2350        let n = cmp::min(cb.out_buf.len() - p.out_buf_ofs, p.flush_remaining as usize);
2351        if n != 0 {
2352            cb.out_buf[p.out_buf_ofs..p.out_buf_ofs + n]
2353                .copy_from_slice(&p.local_buf.b[p.flush_ofs as usize..p.flush_ofs as usize + n]);
2354        }
2355        p.flush_ofs += n as u32;
2356        p.flush_remaining -= n as u32;
2357        p.out_buf_ofs += n;
2358        res.2 = p.out_buf_ofs;
2359    }
2360
2361    if p.finished && p.flush_remaining == 0 {
2362        res.0 = TDEFLStatus::Done
2363    }

2367/// Main compression function. Tries to compress as much as possible from `in_buf` and
2368/// puts compressed output into `out_buf`.
2369///
2370/// The value of `flush` determines if the compressor should attempt to flush all output
2371/// and alternatively try to finish the stream.
2372///
2373/// Use [`TDEFLFlush::Finish`] on the final call to signal that the stream is finishing.
2374///
2375/// Note that this function does not keep track of whether a flush marker has been output, so
2376/// if called using [`TDEFLFlush::Sync`], the caller needs to ensure there is enough space in the
2377/// output buffer if they want to avoid repeated flush markers.
2378/// See #105 for details.
2379///
2380/// # Returns
2381/// Returns a tuple containing the current status of the compressor, the current position
2382/// in the input buffer and the current position in the output buffer.
2383/// A result of [`TDEFLStatus::Done`] indicates that compression is finished, and further calls to this function will
2384/// result in [`TDEFLStatus::BadParam`].
2385/// See [`TDEFLStatus`] for other return values.
2386pub fn compress(
2387    d: &mut CompressorOxide,
2388    in_buf: &[u8],
2389    out_buf: &mut [u8],
2390    flush: TDEFLFlush,
2391) -> (TDEFLStatus, usize, usize) {
2392    compress_inner(
2393        d,
2394        &mut CallbackOxide::new_callback_buf(in_buf, out_buf),
2395        flush,
2396    )
2397}

2401/// # Returns

2427fn compress_inner(
2428    d: &mut CompressorOxide,
2429    callback: &mut CallbackOxide,
2430    flush: TDEFLFlush,
2431) -> (TDEFLStatus, usize, usize) {
2432    d.params.out_buf_ofs = 0;
2433    d.params.src_pos = 0;
2434
2435    let prev_ok = d.params.prev_return_status == TDEFLStatus::Okay;
2436    let flush_finish_once = d.params.flush != TDEFLFlush::Finish || flush == TDEFLFlush::Finish;
2437
2438    d.params.flush = flush;
2439    if !prev_ok || !flush_finish_once {
2440        d.params.prev_return_status = TDEFLStatus::BadParam;
2441        return (d.params.prev_return_status, 0, 0);
2442    }
2443
2444    if d.params.flush_remaining != 0 || d.params.finished {
2445        let res = flush_output_buffer(callback, &mut d.params);
2446        d.params.prev_return_status = res.0;
2447        return res;
2448    }
2449
2450    let one_probe = d.params.flags & MAX_PROBES_MASK == 1;
2451    let greedy = d.params.flags & TDEFL_GREEDY_PARSING_FLAG != 0;
2452    let filter_or_rle = d.params.flags & (TDEFL_FILTER_MATCHES | TDEFL_FORCE_ALL_RAW_BLOCKS) != 0;
2453
2454    let raw = d.params.flags & TDEFL_FORCE_ALL_RAW_BLOCKS != 0;
2455
2456    let compress_success = if raw {
2457        compress_stored(d, callback)
2458    } else if one_probe && greedy && !filter_or_rle {
2459        compress_fast(d, callback)
2460    } else {
2461        compress_normal(d, callback)
2462    };
2463
2464    if !compress_success {
2465        return (
2466            d.params.prev_return_status,
2467            d.params.src_pos,
2468            d.params.out_buf_ofs,
2469        );
2470    }
2471
2472    if let Some(in_buf) = callback.in_buf {
2473        if d.params.flags & (TDEFL_WRITE_ZLIB_HEADER | TDEFL_COMPUTE_ADLER32) != 0 {
2474            d.params.adler32 = update_adler32(d.params.adler32, &in_buf[..d.params.src_pos]);
2475        }
2476    }
2477
2478    let flush_none = d.params.flush == TDEFLFlush::None;
2479    let in_left = callback.in_buf.map_or(0, |buf| buf.len()) - d.params.src_pos;
2480    let remaining = in_left != 0 || d.params.flush_remaining != 0;
2481    if !flush_none && d.dict.lookahead_size == 0 && !remaining {
2482        let flush = d.params.flush;
2483        match flush_block(d, callback, flush) {
2484            Err(_) => {
2485                d.params.prev_return_status = TDEFLStatus::PutBufFailed;
2486                return (
2487                    d.params.prev_return_status,
2488                    d.params.src_pos,
2489                    d.params.out_buf_ofs,
2490                );
2491            }
2492            Ok(x) if x < 0 => {
2493                return (
2494                    d.params.prev_return_status,
2495                    d.params.src_pos,
2496                    d.params.out_buf_ofs,
2497                )
2498            }
2499            _ => {
2500                d.params.finished = d.params.flush == TDEFLFlush::Finish;
2501                if d.params.flush == TDEFLFlush::Full {
2502                    d.dict.b.hash.fill(0);
2503                    d.dict.b.next.fill(0);
2504                    d.dict.size = 0;
2505                }
2506            }
2507        }
2508    }
2509
2510    let res = flush_output_buffer(callback, &mut d.params);
2511    d.params.prev_return_status = res.0;
2512
2513    res
2514}

2516/// Create a set of compression flags using parameters used by zlib and other compressors.
2517/// Mainly intended for use with transition from c libraries as it deals with raw integers.
2518///
2519/// # Parameters
2520/// `level` determines compression level. Clamped to maximum of 10. Negative values result in
2521/// `CompressionLevel::DefaultLevel`.
2522/// `window_bits`: Above 0, wraps the stream in a zlib wrapper, 0 or negative for a raw deflate
2523/// stream.
2524/// `strategy`: Sets the strategy if this conforms to any of the values in `CompressionStrategy`.
2525///
2526/// # Notes
2527/// This function may be removed or moved to the `miniz_oxide_c_api` in the future.
2528pub const fn create_comp_flags_from_zip_params(level: i32, window_bits: i32, strategy: i32) -> u32 {
2529    let num_probes = (if level >= 0 {
2530        // Manual min since cmp::min is not const.
2531        if level > 10 {
2532            10
2533        } else {
2534            level
2535        }
2536        //cmp::min(10, level)
2537    } else {
2538        CompressionLevel::DefaultLevel as i32
2539    }) as usize;
2540    let greedy = if level <= 3 {
2541        TDEFL_GREEDY_PARSING_FLAG
2542    } else {
2543        0
2544    };
2545    let mut comp_flags = NUM_PROBES[num_probes] as u32 | greedy;
2546
2547    if window_bits > 0 {
2548        comp_flags |= TDEFL_WRITE_ZLIB_HEADER;
2549    }
2550
2551    if level == 0 {
2552        comp_flags |= TDEFL_FORCE_ALL_RAW_BLOCKS;
2553    } else if strategy == CompressionStrategy::Filtered as i32 {
2554        comp_flags |= TDEFL_FILTER_MATCHES;
2555    } else if strategy == CompressionStrategy::HuffmanOnly as i32 {
2556        comp_flags &= !MAX_PROBES_MASK;
2557    } else if strategy == CompressionStrategy::Fixed as i32 {
2558        comp_flags |= TDEFL_FORCE_ALL_STATIC_BLOCKS;
2559    } else if strategy == CompressionStrategy::RLE as i32 {
2560        comp_flags |= TDEFL_RLE_MATCHES;
2561    }
2562
2563    comp_flags
2564}
