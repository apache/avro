1//! Core decompression functionality.
2//!
3//! # Using decompress with a wrapping buffer
4//!
5//! [`decompress`] and [`decompress_with_limit`] can be used with a wrapping buffer.
6//!
7//! To decompress with a wrapping buffer you must:
8//! - not pass `TINFL_FLAG_USING_NON_WRAPPING_OUTPUT_BUF` flag
9//! - pass an output buffer with a size of a power of 2
10//! - pass an output buffer  with a size greater or equal to the decompression window
11//!   (which cannot be more than 32KiB, so 32KiB is a safe size)
12//! - pass the same buffer on each call without modification
13//!
14//! You must process return values so that:
15//! - next call pass the input buffer without the first input bytes read skipped
16//! - next call pass the same output buffer
17//! - next call pass an out_pos incremented by the number of bytes output (wrapping to 0 if needed)
18//! - do a next call only if return status is `NeedsMoreInput` or `NeedsMoreInput`
19//!
20//! [`decompress`] will write to any byte after `out_pos` in the output buffer, but will not
21//! wrap around. This means that all bytes after `out_pos` must be saved while the ones before
22//! do not have to.
23//!
24//! [`decompress_with_limit`] will write to any byte after `out_pos` but not more than `out_max`
25//! and will not wrap around. This means that you can use the buffer as a ring buffer for your
26//! application usage, as long as you keep track of the number of disposable bytes.

44/// A struct containing huffman code lengths and the huffman code tree used by the decompressor.
45#[cfg_attr(not(feature = "rustc-dep-of-std"), derive(Clone))]
46#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
47struct HuffmanTable {
48    /// Fast lookup table for shorter huffman codes.
49    ///
50    /// See `HuffmanTable::fast_lookup`.
51    #[cfg_attr(feature = "serde", serde(with = "BigArray"))]
52    pub look_up: [i16; FAST_LOOKUP_SIZE as usize],
53    /// Full huffman tree.
54    ///
55    /// Positive values are edge nodes/symbols, negative values are
56    /// parent nodes/references to other nodes.
57    #[cfg_attr(feature = "serde", serde(with = "BigArray"))]
58    pub tree: [i16; MAX_HUFF_TREE_SIZE],
59}
60
61impl HuffmanTable {
62    const fn new() -> HuffmanTable {
63        HuffmanTable {
64            look_up: [0; FAST_LOOKUP_SIZE as usize],
65            tree: [0; MAX_HUFF_TREE_SIZE],
66        }
67    }
68
69    /// Look for a symbol in the fast lookup table.
70    /// The symbol is stored in the lower 9 bits, the length in the next 6.
71    /// If the returned value is negative, the code wasn't found in the
72    /// fast lookup table and the full tree has to be traversed to find the code.
73    #[inline]
74    const fn fast_lookup(&self, bit_buf: BitBuffer) -> i16 {
75        self.look_up[(bit_buf & (FAST_LOOKUP_SIZE - 1) as BitBuffer) as usize]
76    }
77
78    /// Get the symbol and the code length from the huffman tree.
79    #[inline]
80    fn tree_lookup(&self, fast_symbol: i32, bit_buf: BitBuffer, mut code_len: u8) -> (i32, u32) {
81        let mut symbol = fast_symbol;
82        // We step through the tree until we encounter a positive value, which indicates a
83        // symbol.
84        loop {
85            // symbol here indicates the position of the left (0) node, if the next bit is 1
86            // we add 1 to the lookup position to get the right node.
87            let tree_index = (!symbol + ((bit_buf >> code_len) & 1) as i32) as usize;
88
89            // Use get here to avoid generatic panic code.
90            // The init_tree code should prevent this from actually going out of bounds
91            // but if there were somehow a bug with that
92            // we would at worst end up with corrupted output in release mode.
93            debug_assert!(tree_index < self.tree.len());
94            symbol = i32::from(self.tree.get(tree_index).copied().unwrap_or(i16::MAX));
95            code_len += 1;
96            if symbol >= 0 {
97                break;
98            }
99        }
100        // Note: Using a u8 for code_len inside this function seems to improve performance, but changing it
101        // in localvars seems to worsen things so we convert it to a u32 here.
102        (symbol, u32::from(code_len))
103    }
104
105    #[inline]
106    /// Look up a symbol and code length from the bits in the provided bit buffer.
107    ///
108    /// Returns Some(symbol, length) on success,
109    /// None if the length is 0.
110    ///
111    /// It's possible we could avoid checking for 0 if we can guarantee a sane table.
112    /// TODO: Check if a smaller type for code_len helps performance.
113    fn lookup(&self, bit_buf: BitBuffer) -> (i32, u32) {
114        let symbol = self.fast_lookup(bit_buf).into();
115        if symbol >= 0 {
116            let length = (symbol >> 9) as u32;
117            (symbol, length)
118        } else {
119            // We didn't get a symbol from the fast lookup table, so check the tree instead.
120            self.tree_lookup(symbol, bit_buf, FAST_LOOKUP_BITS)
121        }
122    }
123}

213/// Minimal data representing the [`DecompressorOxide`] state when it is between deflate blocks

236    /// Zlib CMF
237    pub z_header0: u32,

547/// Read an le u16 value from the slice iterator.
548///
549/// # Panics
550/// Panics if there are less than two bytes left.
551#[inline]
552fn read_u16_le(iter: &mut InputWrapper) -> u16 {
553    let ret = {
554        let two_bytes = iter.as_slice()[..2].try_into().unwrap_or_default();
555        u16::from_le_bytes(two_bytes)
556    };
557    iter.advance(2);
558    ret
559}

561/// Ensure that there is data in the bit buffer.
562///
563/// On 64-bit platform, we use a 64-bit value so this will
564/// result in there being at least 32 bits in the bit buffer.
565/// This function assumes that there is at least 4 bytes left in the input buffer.
566#[inline(always)]
567#[cfg(target_pointer_width = "64")]
568fn fill_bit_buffer(l: &mut LocalVars, in_iter: &mut InputWrapper) {
569    // Read four bytes into the buffer at once.
570    if l.num_bits < 30 {
571        l.bit_buf |= BitBuffer::from(in_iter.read_u32_le()) << l.num_bits;
572        l.num_bits += 32;
573    }
574}

576/// Same as previous, but for non-64-bit platforms.
577/// Ensures at least 16 bits are present, requires at least 2 bytes in the in buffer.
578#[inline(always)]
579#[cfg(not(target_pointer_width = "64"))]
580fn fill_bit_buffer(l: &mut LocalVars, in_iter: &mut InputWrapper) {
581    // If the buffer is 32-bit wide, read 2 bytes instead.
582    if l.num_bits < 15 {
583        l.bit_buf |= BitBuffer::from(read_u16_le(in_iter)) << l.num_bits;
584        l.num_bits += 16;
585    }

588/// Check that the zlib header is correct and that there is enough space in the buffer
589/// for the window size specified in the header.
590///
591/// See https://tools.ietf.org/html/rfc1950
592#[inline]
593const fn validate_zlib_header(cmf: u32, flg: u32, flags: u32, mask: usize) -> Action {
594    let mut failed =
595    // cmf + flg should be divisible by 31.
596        (((cmf * 256) + flg) % 31 != 0) ||
597    // If this flag is set, a dictionary was used for this zlib compressed data.
598    // This is currently not supported by miniz or miniz-oxide
599        ((flg & 0b0010_0000) != 0) ||
600    // Compression method. Only 8(DEFLATE) is defined by the standard.
601        ((cmf & 15) != 8);
602
603    let window_size = 1 << ((cmf >> 4) + 8);
604    if (flags & TINFL_FLAG_USING_NON_WRAPPING_OUTPUT_BUF) == 0 {
605        // Bail if the buffer is wrapping and the window size is larger than the buffer.
606        failed |= (mask + 1) < window_size;
607    }
608
609    // Zlib doesn't allow window sizes above 32 * 1024.
610    failed |= window_size > 32_768;
611
612    if failed {
613        Action::Jump(BadZlibHeader)
614    } else {
615        Action::Jump(ReadBlockHeader)
616    }
617}

625/// Try to decode the next huffman code, and puts it in the counter field of the decompressor
626/// if successful.
627///
628/// # Returns
629/// The specified action returned from `f` on success,
630/// `Action::End` if there are not enough data left to decode a symbol.
631fn decode_huffman_code<F>(
632    r: &mut DecompressorOxide,
633    l: &mut LocalVars,
634    table: usize,
635    flags: u32,
636    in_iter: &mut InputWrapper,
637    f: F,
638) -> Action
639where
640    F: FnOnce(&mut DecompressorOxide, &mut LocalVars, i32) -> Action,
641{
642    // As the huffman codes can be up to 15 bits long we need at least 15 bits
643    // ready in the bit buffer to start decoding the next huffman code.
644    if l.num_bits < 15 {
645        // First, make sure there is enough data in the bit buffer to decode a huffman code.
646        if in_iter.bytes_left() < 2 {
647            // If there is less than 2 bytes left in the input buffer, we try to look up
648            // the huffman code with what's available, and return if that doesn't succeed.
649            // Original explanation in miniz:
650            // /* TINFL_HUFF_BITBUF_FILL() is only used rarely, when the number of bytes
651            //  * remaining in the input buffer falls below 2. */
652            // /* It reads just enough bytes from the input stream that are needed to decode
653            //  * the next Huffman code (and absolutely no more). It works by trying to fully
654            //  * decode a */
655            // /* Huffman code by using whatever bits are currently present in the bit buffer.
656            //  * If this fails, it reads another byte, and tries again until it succeeds or
657            //  * until the */
658            // /* bit buffer contains >=15 bits (deflate's max. Huffman code size). */
659            loop {
660                let mut temp = i32::from(r.tables[table].fast_lookup(l.bit_buf));
661                if temp >= 0 {
662                    let code_len = (temp >> 9) as u32;
663                    // TODO: Is there any point to check for code_len != 0 here still?
664                    if (code_len != 0) && (l.num_bits >= code_len) {
665                        break;
666                    }
667                } else if l.num_bits > FAST_LOOKUP_BITS.into() {
668                    let mut code_len = u32::from(FAST_LOOKUP_BITS);
669                    loop {
670                        temp = i32::from(
671                            r.tables[table].tree
672                                [(!temp + ((l.bit_buf >> code_len) & 1) as i32) as usize],
673                        );
674                        code_len += 1;
675                        if temp >= 0 || l.num_bits < code_len + 1 {
676                            break;
677                        }
678                    }
679                    if temp >= 0 {
680                        break;
681                    }
682                }
683
684                // TODO: miniz jumps straight to here after getting here again after failing to read
685                // a byte.
686                // Doing that lets miniz avoid re-doing the lookup that that was done in the
687                // previous call.
688                let mut byte = 0;
689                if let a @ Action::End(_) = read_byte(in_iter, flags, |b| {
690                    byte = b;
691                    Action::None
692                }) {
693                    return a;
694                };
695
696                // Do this outside closure for now to avoid borrowing r.
697                l.bit_buf |= BitBuffer::from(byte) << l.num_bits;
698                l.num_bits += 8;
699
700                if l.num_bits >= 15 {
701                    break;
702                }
703            }
704        } else {
705            // There is enough data in the input buffer, so read the next two bytes
706            // and add them to the bit buffer.
707            // Unwrapping here is fine since we just checked that there are at least two
708            // bytes left.
709            l.bit_buf |= BitBuffer::from(read_u16_le(in_iter)) << l.num_bits;
710            l.num_bits += 16;
711        }
712    }
713
714    // We now have at least 15 bits in the input buffer.
715    let mut symbol = i32::from(r.tables[table].fast_lookup(l.bit_buf));
716    let code_len;
717    // If the symbol was found in the fast lookup table.
718    if symbol >= 0 {
719        // Get the length value from the top bits.
720        // As we shift down the sign bit, converting to an unsigned value
721        // shouldn't overflow.
722        code_len = (symbol >> 9) as u32;
723        // Mask out the length value.
724        symbol &= 511;
725    } else {
726        let res = r.tables[table].tree_lookup(symbol, l.bit_buf, FAST_LOOKUP_BITS);
727        symbol = res.0;
728        code_len = res.1;
729    };
730
731    l.bit_buf >>= code_len;
732    l.num_bits -= code_len;
733    f(r, l, symbol)
734}

736/// Try to read one byte from `in_iter` and call `f` with the read byte as an argument,
737/// returning the result.
738/// If reading fails, `Action::End is returned`
739#[inline]
740fn read_byte<F>(in_iter: &mut InputWrapper, flags: u32, f: F) -> Action
741where
742    F: FnOnce(u8) -> Action,
743{
744    match in_iter.read_byte() {
745        None => end_of_input(flags),
746        Some(byte) => f(byte),
747    }
748}

750// TODO: `l: &mut LocalVars` may be slow similar to decompress_fast (even with inline(always))
751/// Try to read `amount` number of bits from `in_iter` and call the function `f` with the bits as an
752/// an argument after reading, returning the result of that function, or `Action::End` if there are
753/// not enough bytes left.
754#[inline]
755#[allow(clippy::while_immutable_condition)]
756fn read_bits<F>(
757    l: &mut LocalVars,
758    amount: u32,
759    in_iter: &mut InputWrapper,
760    flags: u32,
761    f: F,
762) -> Action
763where
764    F: FnOnce(&mut LocalVars, BitBuffer) -> Action,
765{
766    // Clippy gives a false positive warning here due to the closure.
767    // Read enough bytes from the input iterator to cover the number of bits we want.
768    while l.num_bits < amount {
769        let action = read_byte(in_iter, flags, |byte| {
770            l.bit_buf |= BitBuffer::from(byte) << l.num_bits;
771            l.num_bits += 8;
772            Action::None
773        });
774
775        // If there are not enough bytes in the input iterator, return and signal that we need more.
776        if !matches!(action, Action::None) {
777            return action;
778        }
779    }
780
781    let bits = l.bit_buf & ((1 << amount) - 1);
782    l.bit_buf >>= amount;
783    l.num_bits -= amount;
784    f(l, bits)
785}

824    not(feature = "with-alloc"),
825    target_arch = "aarch64",
826    target_arch = "arm64ec",

829#[inline]
830const fn reverse_bits(n: u16) -> u16 {
831    // Lookup is not used when building as part of std to avoid wasting space
832    // for lookup table in every rust binary
833    // as it's only used for backtraces in the cold path
834    // - see #152
835
836    // armv7 and newer, and loongarch have a cpu instruction for bit reversal so
837    // it's preferable to just use that on those architectures.
838
839    // Also disable lookup table when not using the alloc feature as
840    // we probably don't want to waste space for a lookup table in an environment
841    // without an allocator.
842    n.reverse_bits()
843}

845#[cfg(all(
846    not(any(
847        feature = "rustc-dep-of-std",
848        target_arch = "aarch64",
849        target_arch = "arm64ec",
850        target_arch = "loongarch64"
851    )),
852    feature = "with-alloc"
853))]
854fn reverse_bits(n: u16) -> u16 {
855    static REVERSED_BITS_LOOKUP: [u16; 512] = {
856        let mut table = [0; 512];
857
858        let mut i = 0;
859        while i < 512 {
860            table[i] = (i as u16).reverse_bits();
861            i += 1;
862        }
863
864        table
865    };
866
867    REVERSED_BITS_LOOKUP[n as usize]
868}

870fn init_tree(r: &mut DecompressorOxide, l: &mut LocalVars) -> Option<Action> {
871    loop {
872        let bt = r.block_type as usize;
873
874        let code_sizes = match bt {
875            LITLEN_TABLE => &mut r.code_size_literal[..],
876            DIST_TABLE => &mut r.code_size_dist,
877            HUFFLEN_TABLE => &mut r.code_size_huffman,
878            _ => return None,
879        };
880        let table = &mut r.tables[bt];
881
882        let mut total_symbols = [0u16; 16];
883        // Next code - we use the odd length here to simplify a loop later.
884        let mut next_code = [0u32; 17];
885        const INVALID_CODE: i16 = (1 << 9) | 286;
886        // Set the values in the fast table to return a
887        // non-zero length and an invalid symbol instead of zero
888        // so that we do not have to have a check for a zero
889        // code length in the hot code path later
890        // and can instead error out on the invalid symbol check
891        // on bogus input.
892        table.look_up.fill(INVALID_CODE);
893        // If we are initializing the huffman code length we can skip
894        // this since these codes can't be longer than 3 bits
895        // and thus only use the fast table and this table won't be accessed so
896        // there is no point clearing it.
897        // TODO: Avoid creating this table at all.
898        if bt != HUFFLEN_TABLE {
899            table.tree.fill(0);
900        }
901
902        let table_size = r.table_sizes[bt] as usize;
903        if table_size > code_sizes.len() {
904            return None;
905        }
906
907        for &code_size in &code_sizes[..table_size] {
908            let cs = code_size as usize;
909            // Code sizes are limited to max 15 according to the
910            // deflate spec.
911            // If it is larger than this, something has gone wrong...
912            if cs >= total_symbols.len() {
913                return None;
914            }
915            total_symbols[cs] += 1;
916        }
917
918        let mut used_symbols = 0;
919        let mut total = 0u32;
920        // Count up the total number of used lengths and check that the table is not under or over-subscribed.
921        for (&ts, next) in total_symbols.iter().zip(next_code[1..].iter_mut()).skip(1) {
922            used_symbols += ts;
923            total += u32::from(ts);
924            total <<= 1;
925            *next = total;
926        }
927
928        //
929        // While it's not explicitly stated in the spec, a hufflen table
930        // with a single length (or none) would be invalid as there needs to be
931        // at minimum a length for both a non-zero length huffman code for the end of block symbol
932        // and one of the codes to represent 0 to make sense - so just reject that here as well.
933        //
934        // The distance table is allowed to have a single distance code though according to the spect it is
935        // supposed to be accompanied by a second dummy code. It can also be empty indicating no used codes.
936        //
937        // The literal/length table can not be empty as there has to be an end of block symbol,
938        // The standard doesn't specify that there should be a dummy code in case of a single
939        // symbol (i.e an empty block). Normally that's not an issue though the code will have
940        // to take that into account later on in case of malformed input.
941        if total != 65_536 && (used_symbols > 1 || bt == HUFFLEN_TABLE) {
942            return Some(Action::Jump(BadTotalSymbols));
943        }
944
945        let mut tree_next = -1;
946        for symbol_index in 0..table_size {
947            // Code sizes are limited to 15 according to the spec
948            // It's already checked earlier but the compiler might not be smart enough to know that.
949            let code_size = code_sizes[symbol_index] & 15;
950            if code_size == 0 {
951                continue;
952            }
953
954            let cur_code = next_code[code_size as usize];
955            next_code[code_size as usize] += 1;
956
957            let n = (cur_code & (u32::MAX >> (32 - code_size))) as u16;
958
959            let mut rev_code = if n < 512 {
960                // Using a lookup table
961                // for a small speedup here,
962                // Seems to only really make a difference on very short
963                // inputs however.
964                // 512 seems to be around a sweet spot.
965                reverse_bits(n)
966            } else {
967                n.reverse_bits()
968            } >> (16 - code_size);
969
970            if code_size <= FAST_LOOKUP_BITS {
971                let k = (i16::from(code_size) << 9) | symbol_index as i16;
972                while rev_code < FAST_LOOKUP_SIZE {
973                    table.look_up[rev_code as usize] = k;
974                    rev_code += 1 << code_size;
975                }
976                continue;
977            }
978
979            let mut tree_cur = table.look_up[(rev_code & (FAST_LOOKUP_SIZE - 1)) as usize];
980            if tree_cur == INVALID_CODE {
981                table.look_up[(rev_code & (FAST_LOOKUP_SIZE - 1)) as usize] = tree_next;
982                tree_cur = tree_next;
983                tree_next -= 2;
984            }
985
986            rev_code >>= FAST_LOOKUP_BITS - 1;
987            for _ in FAST_LOOKUP_BITS + 1..code_size {
988                rev_code >>= 1;
989                tree_cur -= (rev_code & 1) as i16;
990                let tree_index = (-tree_cur - 1) as usize;
991                if tree_index >= table.tree.len() {
992                    return None;
993                }
994                if table.tree[tree_index] == 0 {
995                    table.tree[tree_index] = tree_next;
996                    tree_cur = tree_next;
997                    tree_next -= 2;
998                } else {
999                    tree_cur = table.tree[tree_index];
1000                }
1001            }
1002
1003            rev_code >>= 1;
1004            tree_cur -= (rev_code & 1) as i16;
1005            let tree_index = (-tree_cur - 1) as usize;
1006            if tree_index >= table.tree.len() {
1007                return None;
1008            }
1009            table.tree[tree_index] = symbol_index as i16;
1010        }
1011
1012        if r.block_type == HUFFLEN_TABLE as u8 {
1013            l.counter = 0;
1014            return Some(Action::Jump(ReadLitlenDistTablesCodeSize));
1015        }
1016
1017        if r.block_type == LITLEN_TABLE as u8 {
1018            break;
1019        }
1020        r.block_type -= 1;
1021    }
1022
1023    l.counter = 0;
1024
1025    Some(Action::Jump(DecodeLitlen))
1026}

1028// A helper macro for generating the state machine.
1029//
1030// As Rust doesn't have fallthrough on matches, we have to return to the match statement
1031// and jump for each state change. (Which would ideally be optimized away, but often isn't.)
1032macro_rules! generate_state {
1033    ($state: ident, $state_machine: tt, $f: expr) => {
1034        loop {
1035            match $f {
1036                Action::None => continue,
1037                Action::Jump(new_state) => {
1038                    $state = new_state;
1039                    continue $state_machine;
1040                },
1041                Action::End(result) => break $state_machine result,
1042            }
1043        }
1044    };
1045}

1056#[inline]
1057fn transfer(
1058    out_slice: &mut [u8],
1059    mut source_pos: usize,
1060    mut out_pos: usize,
1061    match_len: usize,
1062    out_buf_size_mask: usize,
1063) {
1064    // special case that comes up surprisingly often. in the case that `source_pos`
1065    // is 1 less than `out_pos`, we can say that the entire range will be the same
1066    // value and optimize this to be a simple `memset`
1067    let source_diff = if source_pos > out_pos {
1068        source_pos - out_pos
1069    } else {
1070        out_pos - source_pos
1071    };
1072
1073    // The last 3 bytes can wrap as those are dealt with separately at the end.
1074    // Use wrapping_sub rather than saturating for performance reasons here as
1075    // if source_pos + match_len  is < 3 we just want to jump to the end
1076    // condition anyhow.
1077    let not_wrapping = (out_buf_size_mask == usize::MAX)
1078        || ((source_pos + match_len).wrapping_sub(3) < out_slice.len());
1079
1080    let end_pos = ((match_len >> 2) * 4) + out_pos;
1081    if not_wrapping && source_diff == 1 && out_pos > source_pos {
1082        let end = (match_len >> 2) * 4 + out_pos;
1083        let init = out_slice[out_pos - 1];
1084        out_slice[out_pos..end].fill(init);
1085        out_pos = end;
1086        source_pos = end - 1;
1087    // if the difference between `source_pos` and `out_pos` is greater than 3,
1088    // and we are not wrapping, we
1089    // can do slightly better than the naive case by copying everything at once
1090    } else if not_wrapping && out_pos > source_pos && (out_pos - source_pos >= 4) {
1091        let end_pos = cmp::min(end_pos, out_slice.len().saturating_sub(3));
1092        while out_pos < end_pos {
1093            out_slice.copy_within(source_pos..=source_pos + 3, out_pos);
1094            source_pos += 4;
1095            out_pos += 4;
1096        }
1097    } else {
1098        let end_pos = cmp::min(end_pos, out_slice.len().saturating_sub(3));
1099        while out_pos < end_pos {
1100            // Placing these assertions moves some bounds check before the accesses which
1101            // makes the compiler able to optimize better.
1102            // Ideally we would find a safe way to remove them entirely.
1103            assert!(out_pos + 3 < out_slice.len());
1104            assert!((source_pos + 3) & out_buf_size_mask < out_slice.len());
1105
1106            out_slice[out_pos] = out_slice[source_pos & out_buf_size_mask];
1107            out_slice[out_pos + 1] = out_slice[(source_pos + 1) & out_buf_size_mask];
1108            out_slice[out_pos + 2] = out_slice[(source_pos + 2) & out_buf_size_mask];
1109            out_slice[out_pos + 3] = out_slice[(source_pos + 3) & out_buf_size_mask];
1110            source_pos += 4;
1111            out_pos += 4;
1112        }
1113    }
1114
1115    match match_len & 3 {
1116        0 => (),
1117        1 => out_slice[out_pos] = out_slice[source_pos & out_buf_size_mask],
1118        2 => {
1119            assert!(out_pos + 1 < out_slice.len());
1120            assert!((source_pos + 1) & out_buf_size_mask < out_slice.len());
1121            out_slice[out_pos] = out_slice[source_pos & out_buf_size_mask];
1122            out_slice[out_pos + 1] = out_slice[(source_pos + 1) & out_buf_size_mask];
1123        }
1124        3 => {
1125            assert!(out_pos + 2 < out_slice.len());
1126            assert!((source_pos + 2) & out_buf_size_mask < out_slice.len());
1127            out_slice[out_pos] = out_slice[source_pos & out_buf_size_mask];
1128            out_slice[out_pos + 1] = out_slice[(source_pos + 1) & out_buf_size_mask];
1129            out_slice[out_pos + 2] = out_slice[(source_pos + 2) & out_buf_size_mask];
1130        }
1131        _ => unreachable!(),
1132    }
1133}

1135/// Presumes that there is at least match_len bytes in output left.
1136#[inline]
1137fn apply_match(
1138    out_slice: &mut [u8],
1139    out_pos: usize,
1140    dist: usize,
1141    match_len: usize,
1142    out_buf_size_mask: usize,
1143) {
1144    debug_assert!(out_pos.checked_add(match_len).unwrap() <= out_slice.len());
1145
1146    let source_pos = out_pos.wrapping_sub(dist) & out_buf_size_mask;
1147
1148    if match_len == 3 {
1149        let out_slice = Cell::from_mut(out_slice).as_slice_of_cells();
1150        if let Some(dst) = out_slice.get(out_pos..out_pos + 3) {
1151            // Moving bounds checks before any memory mutation allows the optimizer
1152            // combine them together.
1153            let src = out_slice
1154                .get(source_pos)
1155                .zip(out_slice.get((source_pos + 1) & out_buf_size_mask))
1156                .zip(out_slice.get((source_pos + 2) & out_buf_size_mask));
1157            if let Some(((a, b), c)) = src {
1158                // For correctness, the memory reads and writes have to be interleaved.
1159                // Cells make it possible for read and write references to overlap.
1160                dst[0].set(a.get());
1161                dst[1].set(b.get());
1162                dst[2].set(c.get());
1163            }
1164        }
1165        return;
1166    }
1167
1168    if cfg!(not(any(target_arch = "x86", target_arch = "x86_64"))) {
1169        // The copy from slice code seems to not give any added performance at least on
1170        // armv7 so transfer manually
1171        // Need to test on other platforms.
1172        transfer(out_slice, source_pos, out_pos, match_len, out_buf_size_mask);
1173        return;
1174    }
1175
1176    if source_pos >= out_pos && (source_pos - out_pos) < match_len {
1177        transfer(out_slice, source_pos, out_pos, match_len, out_buf_size_mask);
1178    } else if match_len <= dist && source_pos + match_len < out_slice.len() {
1179        // Destination and source segments does not intersect and source does not wrap.
1180        // TODO: An invalid before start of data wrapping match reached here before
1181        // it was fixed (it wrapped around and ended overlapping again)- need
1182        // to check that we are not wrapping here.
1183        if source_pos < out_pos {
1184            let (from_slice, to_slice) = out_slice.split_at_mut(out_pos);
1185            to_slice[..match_len].copy_from_slice(&from_slice[source_pos..source_pos + match_len]);
1186        } else {
1187            let (to_slice, from_slice) = out_slice.split_at_mut(source_pos);
1188            to_slice[out_pos..out_pos + match_len].copy_from_slice(&from_slice[..match_len]);
1189        }
1190    } else {
1191        transfer(out_slice, source_pos, out_pos, match_len, out_buf_size_mask);
1192    }
1193}

1195/// Fast inner decompression loop which is run  while there is at least
1196/// 259 bytes left in the output buffer, and at least 6 bytes left in the input buffer
1197/// (The maximum one match would need + 1).
1198///
1199/// This was inspired by a similar optimization in zlib, which uses this info to do
1200/// faster unchecked copies of multiple bytes at a time.
1201/// Currently we don't do this here, but this function does avoid having to jump through the
1202/// big match loop on each state change(as rust does not have fallthrough or gotos at the moment),
1203/// and already improves decompression speed a fair bit.
1204fn decompress_fast(
1205    r: &mut DecompressorOxide,
1206    in_iter: &mut InputWrapper,
1207    out_buf: &mut OutputBuffer,
1208    flags: u32,
1209    local_vars: &mut LocalVars,
1210    out_buf_size_mask: usize,
1211) -> (TINFLStatus, State) {
1212    // Make a local copy of the most used variables, to avoid having to update and read from values
1213    // in a random memory location and to encourage more register use.
1214    let mut l = *local_vars;
1215    let mut state;
1216
1217    let status: TINFLStatus = 'o: loop {
1218        state = State::DecodeLitlen;
1219        loop {
1220            // This function assumes that there is at least 259 bytes left in the output buffer,
1221            // and that there is at least 14 bytes left in the input buffer. 14 input bytes:
1222            // 15 (prev lit) + 15 (length) + 5 (length extra) + 15 (dist)
1223            // + 29 + 32 (left in bit buf, including last 13 dist extra) = 111 bits < 14 bytes
1224            // We need the one extra byte as we may write one length and one full match
1225            // before checking again.
1226            if out_buf.bytes_left() < 259 || in_iter.bytes_left() < 14 {
1227                state = State::DecodeLitlen;
1228                break 'o TINFLStatus::Done;
1229            }
1230
1231            fill_bit_buffer(&mut l, in_iter);
1232
1233            let (symbol, code_len) = r.tables[LITLEN_TABLE].lookup(l.bit_buf);
1234            l.counter = symbol as u32;
1235            l.bit_buf >>= code_len;
1236            l.num_bits -= code_len;
1237
1238            if (l.counter & 256) != 0 {
1239                // The symbol is not a literal.
1240                break;
1241            } else {
1242                // If we have a 32-bit buffer we need to read another two bytes now
1243                // to have enough bits to keep going.
1244                if cfg!(not(target_pointer_width = "64")) {
1245                    fill_bit_buffer(&mut l, in_iter);
1246                }
1247
1248                let (symbol, code_len) = r.tables[LITLEN_TABLE].lookup(l.bit_buf);
1249                l.bit_buf >>= code_len;
1250                l.num_bits -= code_len;
1251                // The previous symbol was a literal, so write it directly and check
1252                // the next one.
1253                out_buf.write_byte(l.counter as u8);
1254                if (symbol & 256) != 0 {
1255                    l.counter = symbol as u32;
1256                    // The symbol is a length value.
1257                    break;
1258                } else {
1259                    // The symbol is a literal, so write it directly and continue.
1260                    out_buf.write_byte(symbol as u8);
1261                }
1262            }
1263        }
1264
1265        // Mask the top bits since they may contain length info.
1266        l.counter &= 511;
1267        if l.counter == 256 {
1268            // We hit the end of block symbol.
1269            state.begin(BlockDone);
1270            break 'o TINFLStatus::Done;
1271        } else if l.counter > 285 {
1272            // Invalid code.
1273            // We already verified earlier that the code is > 256.
1274            state.begin(InvalidLitlen);
1275            break 'o TINFLStatus::Failed;
1276        } else {
1277            // The symbol was a length code.
1278            // # Optimization
1279            // Mask the value to avoid bounds checks
1280            // While the maximum is checked, the compiler isn't able to know that the
1281            // value won't wrap around here.
1282            l.num_extra = LENGTH_EXTRA[(l.counter - 257) as usize & BASE_EXTRA_MASK];
1283            l.counter = u32::from(LENGTH_BASE[(l.counter - 257) as usize & BASE_EXTRA_MASK]);
1284            // Length and distance codes have a number of extra bits depending on
1285            // the base, which together with the base gives us the exact value.
1286
1287            // We need to make sure we have at least 33 (so min 5 bytes) bits in the buffer at this spot.
1288            fill_bit_buffer(&mut l, in_iter);
1289            if l.num_extra != 0 {
1290                let extra_bits = l.bit_buf & ((1 << l.num_extra) - 1);
1291                l.bit_buf >>= l.num_extra;
1292                l.num_bits -= u32::from(l.num_extra);
1293                l.counter += extra_bits as u32;
1294            }
1295
1296            // We found a length code, so a distance code should follow.
1297
1298            if cfg!(not(target_pointer_width = "64")) {
1299                fill_bit_buffer(&mut l, in_iter);
1300            }
1301
1302            let (mut symbol, code_len) = r.tables[DIST_TABLE].lookup(l.bit_buf);
1303            symbol &= 511;
1304            l.bit_buf >>= code_len;
1305            l.num_bits -= code_len;
1306            if symbol > 29 {
1307                state.begin(InvalidDist);
1308                break 'o TINFLStatus::Failed;
1309            }
1310
1311            l.num_extra = num_extra_bits_for_distance_code(symbol as u8);
1312            l.dist = u32::from(DIST_BASE[symbol as usize]);
1313
1314            if l.num_extra != 0 {
1315                fill_bit_buffer(&mut l, in_iter);
1316                let extra_bits = l.bit_buf & ((1 << l.num_extra) - 1);
1317                l.bit_buf >>= l.num_extra;
1318                l.num_bits -= u32::from(l.num_extra);
1319                l.dist += extra_bits as u32;
1320            }
1321
1322            let position = out_buf.position();
1323            if (l.dist as usize > out_buf.position()
1324                && (flags & TINFL_FLAG_USING_NON_WRAPPING_OUTPUT_BUF != 0))
1325                || (l.dist as usize > out_buf.get_ref().len())
1326            {
1327                // We encountered a distance that refers a position before
1328                // the start of the decoded data, so we can't continue.
1329                state.begin(DistanceOutOfBounds);
1330                break TINFLStatus::Failed;
1331            }
1332
1333            apply_match(
1334                out_buf.get_mut(),
1335                position,
1336                l.dist as usize,
1337                l.counter as usize,
1338                out_buf_size_mask,
1339            );
1340
1341            out_buf.set_position(position + l.counter as usize);
1342        }
1343    };
1344
1345    *local_vars = l;
1346    (status, state)
1347}

1349/// Main decompression function. Keeps decompressing data from `in_buf` until the `in_buf` is
1350/// empty, `out` is full, the end of the deflate stream is hit, or there is an error in the
1351/// deflate stream.
1352///
1353/// # Arguments
1354///
1355/// `r` is a [`DecompressorOxide`] struct with the state of this stream.
1356///
1357/// `in_buf` is a reference to the compressed data that is to be decompressed. The decompressor will
1358/// start at the first byte of this buffer.
1359///
1360/// `out` is a reference to the buffer that will store the decompressed data, and that
1361/// stores previously decompressed data if any.
1362///
1363/// * The offset given by `out_pos` indicates where in the output buffer slice writing should start.
1364/// * If [`TINFL_FLAG_USING_NON_WRAPPING_OUTPUT_BUF`] is not set, the output buffer is used in a
1365///   wrapping manner, and it's size is required to be a power of 2.
1366/// * The decompression function normally needs access to 32KiB of the previously decompressed data
1367///   (or to the beginning of the decompressed data if less than 32KiB has been decompressed.)
1368///     - If this data is not available, decompression may fail.
1369///     - Some deflate compressors allow specifying a window size which limits match distances to
1370///       less than this, or alternatively an RLE mode where matches will only refer to the previous byte
1371///       and thus allows a smaller output buffer. The window size can be specified in the zlib
1372///       header structure, however, the header data should not be relied on to be correct.
1373///
1374/// `flags` indicates settings and status to the decompression function.
1375/// * The [`TINFL_FLAG_HAS_MORE_INPUT`] has to be specified if more compressed data is to be provided
1376///   in a subsequent call to this function.
1377/// * See the the [`inflate_flags`] module for details on other flags.
1378///
1379/// # Returns
1380///
1381/// Returns a tuple containing the status of the compressor, the number of input bytes read, and the
1382/// number of bytes output to `out`.
1383pub fn decompress(
1384    r: &mut DecompressorOxide,
1385    in_buf: &[u8],
1386    out: &mut [u8],
1387    out_pos: usize,
1388    flags: u32,
1389) -> (TINFLStatus, usize, usize) {
1390    decompress_with_limit(r, in_buf, out, out_pos, usize::MAX, flags)
1391}
1392
1393/// Same as [`decompress()`] with a maximum decompressed byte count.
1394///
1395/// By default [`decompress()`] decompress untill end of `out` buffer if possible.
1396/// `decompress_with_limit` will stop when `out_max` bytes have been decompressed,
1397/// or when `out` buffer is full, whichever comes first.
1398///
1399/// This is especially useful when using a wrapping output buffer. This helps keeping
1400/// some data that has not yet been consumed in the buffer while decompressing new bytes.
1401///
1402/// `out_max` is the maximum number of *bytes* that decompress will write
1403pub fn decompress_with_limit(
1404    r: &mut DecompressorOxide,
1405    in_buf: &[u8],
1406    out: &mut [u8],
1407    out_pos: usize,
1408    out_max: usize,
1409    flags: u32,
1410) -> (TINFLStatus, usize, usize) {
1411    let out_buf_size_mask = if flags & TINFL_FLAG_USING_NON_WRAPPING_OUTPUT_BUF != 0 {
1412        usize::MAX
1413    } else {
1414        // In the case of zero len, any attempt to write would produce HasMoreOutput,
1415        // so to gracefully process the case of there really being no output,
1416        // set the mask to all zeros.
1417        out.len().saturating_sub(1)
1418    };
1419
1420    // Ensure the output buffer's size is a power of 2, unless the output buffer
1421    // is large enough to hold the entire output file (in which case it doesn't
1422    // matter).
1423    // Also make sure that the output buffer position is not past the end of the output buffer.
1424    if (out_buf_size_mask.wrapping_add(1) & out_buf_size_mask) != 0 || out_pos > out.len() {
1425        return (TINFLStatus::BadParam, 0, 0);
1426    }
1427
1428    let mut in_iter = InputWrapper::from_slice(in_buf);
1429
1430    let mut state = r.state;
1431
1432    let mut out_buf = OutputBuffer::from_slice_pos_and_max(out, out_pos, out_max);
1433
1434    // Make a local copy of the important variables here so we can work with them on the stack.
1435    let mut l = LocalVars {
1436        bit_buf: r.bit_buf,
1437        num_bits: r.num_bits,
1438        dist: r.dist,
1439        counter: r.counter,
1440        num_extra: r.num_extra,
1441    };
1442
1443    let mut status = 'state_machine: loop {
1444        match state {
1445            Start => generate_state!(state, 'state_machine, {
1446                l.bit_buf = 0;
1447                l.num_bits = 0;
1448                l.dist = 0;
1449                l.counter = 0;
1450                l.num_extra = 0;
1451                r.z_header0 = 0;
1452                r.z_header1 = 0;
1453                r.z_adler32 = 1;
1454                r.check_adler32 = 1;
1455                if flags & TINFL_FLAG_PARSE_ZLIB_HEADER != 0 {
1456                    Action::Jump(State::ReadZlibCmf)
1457                } else {
1458                    Action::Jump(State::ReadBlockHeader)
1459                }
1460            }),
1461
1462            ReadZlibCmf => generate_state!(state, 'state_machine, {
1463                read_byte(&mut in_iter, flags, |cmf| {
1464                    r.z_header0 = u32::from(cmf);
1465                    Action::Jump(State::ReadZlibFlg)
1466                })
1467            }),
1468
1469            ReadZlibFlg => generate_state!(state, 'state_machine, {
1470                read_byte(&mut in_iter, flags, |flg| {
1471                    r.z_header1 = u32::from(flg);
1472                    validate_zlib_header(r.z_header0, r.z_header1, flags, out_buf_size_mask)
1473                })
1474            }),
1475
1476            // Read the block header and jump to the relevant section depending on the block type.
1477            ReadBlockHeader => generate_state!(state, 'state_machine, {
1478                read_bits(&mut l, 3, &mut in_iter, flags, |l, bits| {
1479                    r.finish = (bits & 1) as u8;
1480                    r.block_type = ((bits >> 1) & 3) as u8;
1481                    match r.block_type {
1482                        0 => Action::Jump(BlockTypeNoCompression),
1483                        1 => {
1484                            start_static_table(r);
1485                            init_tree(r, l).unwrap_or(Action::End(TINFLStatus::Failed))
1486                        },
1487                        2 => {
1488                            l.counter = 0;
1489                            Action::Jump(ReadTableSizes)
1490                        },
1491                        3 => Action::Jump(BlockTypeUnexpected),
1492                        _ => unreachable!()
1493                    }
1494                })
1495            }),
1496
1497            // Raw/Stored/uncompressed block.
1498            BlockTypeNoCompression => generate_state!(state, 'state_machine, {
1499                pad_to_bytes(&mut l, &mut in_iter, flags, |l| {
1500                    l.counter = 0;
1501                    Action::Jump(RawHeader)
1502                })
1503            }),
1504
1505            // Check that the raw block header is correct.
1506            RawHeader => generate_state!(state, 'state_machine, {
1507                if l.counter < 4 {
1508                    // Read block length and block length check.
1509                    if l.num_bits != 0 {
1510                        read_bits(&mut l, 8, &mut in_iter, flags, |l, bits| {
1511                            r.raw_header[l.counter as usize] = bits as u8;
1512                            l.counter += 1;
1513                            Action::None
1514                        })
1515                    } else {
1516                        read_byte(&mut in_iter, flags, |byte| {
1517                            r.raw_header[l.counter as usize] = byte;
1518                            l.counter += 1;
1519                            Action::None
1520                        })
1521                    }
1522                } else {
1523                    // Check if the length value of a raw block is correct.
1524                    // The 2 first (2-byte) words in a raw header are the length and the
1525                    // ones complement of the length.
1526                    let length = u16::from(r.raw_header[0]) | (u16::from(r.raw_header[1]) << 8);
1527                    let check = u16::from(r.raw_header[2]) | (u16::from(r.raw_header[3]) << 8);
1528                    let valid = length == !check;
1529                    l.counter = length.into();
1530
1531                    if !valid {
1532                        Action::Jump(BadRawLength)
1533                    } else if l.counter == 0 {
1534                        // Empty raw block. Sometimes used for synchronization.
1535                        Action::Jump(BlockDone)
1536                    } else if l.num_bits != 0 {
1537                        // There is some data in the bit buffer, so we need to write that first.
1538                        Action::Jump(RawReadFirstByte)
1539                    } else {
1540                        // The bit buffer is empty, so memcpy the rest of the uncompressed data from
1541                        // the block.
1542                        Action::Jump(RawMemcpy1)
1543                    }
1544                }
1545            }),
1546
1547            // Read the byte from the bit buffer.
1548            RawReadFirstByte => generate_state!(state, 'state_machine, {
1549                read_bits(&mut l, 8, &mut in_iter, flags, |l, bits| {
1550                    l.dist = bits as u32;
1551                    Action::Jump(RawStoreFirstByte)
1552                })
1553            }),
1554
1555            // Write the byte we just read to the output buffer.
1556            RawStoreFirstByte => generate_state!(state, 'state_machine, {
1557                if out_buf.bytes_left() == 0 {
1558                    Action::End(TINFLStatus::HasMoreOutput)
1559                } else {
1560                    out_buf.write_byte(l.dist as u8);
1561                    l.counter -= 1;
1562                    if l.counter == 0 || l.num_bits == 0 {
1563                        Action::Jump(RawMemcpy1)
1564                    } else {
1565                        // There is still some data left in the bit buffer that needs to be output.
1566                        // TODO: Changed this to jump to `RawReadfirstbyte` rather than
1567                        // `RawStoreFirstByte` as that seemed to be the correct path, but this
1568                        // needs testing.
1569                        Action::Jump(RawReadFirstByte)
1570                    }
1571                }
1572            }),
1573
1574            RawMemcpy1 => generate_state!(state, 'state_machine, {
1575                if l.counter == 0 {
1576                    Action::Jump(BlockDone)
1577                } else if out_buf.bytes_left() == 0 {
1578                    Action::End(TINFLStatus::HasMoreOutput)
1579                } else {
1580                    Action::Jump(RawMemcpy2)
1581                }
1582            }),
1583
1584            RawMemcpy2 => generate_state!(state, 'state_machine, {
1585                if in_iter.bytes_left() > 0 {
1586                    // Copy as many raw bytes as possible from the input to the output using memcpy.
1587                    // Raw block lengths are limited to 64 * 1024, so casting through usize and u32
1588                    // is not an issue.
1589                    let space_left = out_buf.bytes_left();
1590                    let bytes_to_copy = cmp::min(cmp::min(
1591                        space_left,
1592                        in_iter.bytes_left()),
1593                        l.counter as usize
1594                    );
1595
1596                    out_buf.write_slice(&in_iter.as_slice()[..bytes_to_copy]);
1597
1598                    in_iter.advance(bytes_to_copy);
1599                    l.counter -= bytes_to_copy as u32;
1600                    Action::Jump(RawMemcpy1)
1601                } else {
1602                    end_of_input(flags)
1603                }
1604            }),
1605
1606            // Read how many huffman codes/symbols are used for each table.
1607            ReadTableSizes => generate_state!(state, 'state_machine, {
1608                if l.counter < 3 {
1609                    let num_bits = [5, 5, 4][l.counter as usize];
1610                    read_bits(&mut l, num_bits, &mut in_iter, flags, |l, bits| {
1611                        r.table_sizes[l.counter as usize] =
1612                            bits as u16 + MIN_TABLE_SIZES[l.counter as usize];
1613                        l.counter += 1;
1614                        Action::None
1615                    })
1616                } else {
1617                    r.code_size_huffman.fill(0);
1618                    l.counter = 0;
1619                    // Check that the litlen and distance are within spec.
1620                    // litlen table should be <=286 acc to the RFC and
1621                    // additionally zlib rejects dist table sizes larger than 30.
1622                    // NOTE this the final sizes after adding back predefined values, not
1623                    // raw value in the data.
1624                    // See miniz_oxide issue #130 and https://github.com/madler/zlib/issues/82.
1625                    if r.table_sizes[LITLEN_TABLE] <= 286 && r.table_sizes[DIST_TABLE] <= 30 {
1626                        Action::Jump(ReadHufflenTableCodeSize)
1627                    }
1628                    else {
1629                        Action::Jump(BadDistOrLiteralTableLength)
1630                    }
1631                }
1632            }),
1633
1634            // Read the 3-bit lengths of the huffman codes describing the huffman code lengths used
1635            // to decode the lengths of the main tables.
1636            ReadHufflenTableCodeSize => generate_state!(state, 'state_machine, {
1637                if l.counter < r.table_sizes[HUFFLEN_TABLE].into() {
1638                    read_bits(&mut l, 3, &mut in_iter, flags, |l, bits| {
1639                        // These lengths are not stored in a normal ascending order, but rather one
1640                        // specified by the deflate specification intended to put the most used
1641                        // values at the front as trailing zero lengths do not have to be stored.
1642                        r.code_size_huffman[HUFFMAN_LENGTH_ORDER[l.counter as usize] as usize] =
1643                                bits as u8;
1644                        l.counter += 1;
1645                        Action::None
1646                    })
1647                } else {
1648                    r.table_sizes[HUFFLEN_TABLE] = MAX_HUFF_SYMBOLS_2 as u16;
1649                    init_tree(r, &mut l).unwrap_or(Action::End(TINFLStatus::Failed))
1650                }
1651            }),
1652
1653            ReadLitlenDistTablesCodeSize => generate_state!(state, 'state_machine, {
1654                if l.counter < u32::from(r.table_sizes[LITLEN_TABLE]) + u32::from(r.table_sizes[DIST_TABLE]) {
1655                    decode_huffman_code(
1656                        r, &mut l, HUFFLEN_TABLE,
1657                        flags, &mut in_iter, |r, l, symbol| {
1658                            l.dist = symbol as u32;
1659                            if l.dist < 16 {
1660                                r.len_codes[l.counter as usize & LEN_CODES_MASK] = l.dist as u8;
1661                                l.counter += 1;
1662                                Action::None
1663                            } else if l.dist == 16 && l.counter == 0 {
1664                                Action::Jump(BadCodeSizeDistPrevLookup)
1665                            } else {
1666                                // Last value is a dummy to allow mask.
1667                                l.num_extra = [2, 3, 7, 0][(l.dist as usize - 16) & 3];
1668                                Action::Jump(ReadExtraBitsCodeSize)
1669                            }
1670                        }
1671                    )
1672                } else if l.counter != u32::from(r.table_sizes[LITLEN_TABLE]) + u32::from(r.table_sizes[DIST_TABLE]) {
1673                    Action::Jump(BadCodeSizeSum)
1674                } else {
1675
1676                    r.code_size_literal[..r.table_sizes[LITLEN_TABLE] as usize]
1677                        .copy_from_slice(&r.len_codes[..r.table_sizes[LITLEN_TABLE] as usize & LEN_CODES_MASK]);
1678
1679                    let dist_table_start = r.table_sizes[LITLEN_TABLE] as usize;
1680                    debug_assert!(dist_table_start < r.len_codes.len());
1681                    let dist_table_end = (r.table_sizes[LITLEN_TABLE] +
1682                                          r.table_sizes[DIST_TABLE]) as usize;
1683                    let code_size_dist_end = r.table_sizes[DIST_TABLE] as usize;
1684                    debug_assert!(dist_table_end < r.len_codes.len());
1685                    debug_assert!(code_size_dist_end < r.code_size_dist.len());
1686                    let dist_table_start = dist_table_start & LEN_CODES_MASK;
1687                    let dist_table_end = dist_table_end & LEN_CODES_MASK;
1688                    r.code_size_dist[..code_size_dist_end & (MAX_HUFF_SYMBOLS_1 - 1)]
1689                        .copy_from_slice(&r.len_codes[dist_table_start..dist_table_end]);
1690
1691                    r.block_type -= 1;
1692                    init_tree(r, &mut l).unwrap_or(Action::End(TINFLStatus::Failed))
1693                }
1694            }),
1695
1696            ReadExtraBitsCodeSize => generate_state!(state, 'state_machine, {
1697                let num_extra = l.num_extra.into();
1698                read_bits(&mut l, num_extra, &mut in_iter, flags, |l, mut extra_bits| {
1699                    // Mask to avoid a bounds check.
1700                    // We can use 2 since the 2 first values are the same.
1701                    extra_bits += [3, 3, 11][(l.dist as usize - 16) & 2];
1702                    let val = if l.dist == 16 {
1703                        debug_assert!(l.counter as usize - 1 < r.len_codes.len());
1704                        r.len_codes[(l.counter as usize - 1) & LEN_CODES_MASK]
1705                    } else {
1706                        0
1707                    };
1708
1709                    let fill_start = l.counter as usize;
1710                    let fill_end = l.counter as usize + extra_bits as usize;
1711                    debug_assert!(fill_start < r.len_codes.len());
1712                    debug_assert!(fill_end < r.len_codes.len());
1713
1714                    r.len_codes[
1715                            fill_start & LEN_CODES_MASK..fill_end & LEN_CODES_MASK
1716                        ].fill(val);
1717                    l.counter += extra_bits as u32;
1718                    Action::Jump(ReadLitlenDistTablesCodeSize)
1719                })
1720            }),
1721
1722            DecodeLitlen => generate_state!(state, 'state_machine, {
1723                if in_iter.bytes_left() < 4 || out_buf.bytes_left() < 2 {
1724                    // See if we can decode a literal with the data we have left.
1725                    // Jumps to next state (WriteSymbol) if successful.
1726                    decode_huffman_code(
1727                        r,
1728                        &mut l,
1729                        LITLEN_TABLE,
1730                        flags,
1731                        &mut in_iter,
1732                        |_r, l, symbol| {
1733                            l.counter = symbol as u32;
1734                            Action::Jump(WriteSymbol)
1735                        },
1736                    )
1737                } else if
1738                // If there is enough space, use the fast inner decompression
1739                // function.
1740                    out_buf.bytes_left() >= 259 &&
1741                    in_iter.bytes_left() >= 14
1742                {
1743                    let (status, new_state) = decompress_fast(
1744                        r,
1745                        &mut in_iter,
1746                        &mut out_buf,
1747                        flags,
1748                        &mut l,
1749                        out_buf_size_mask,
1750                    );
1751
1752                    state = new_state;
1753                    if status == TINFLStatus::Done {
1754                        Action::Jump(new_state)
1755                    } else {
1756                        Action::End(status)
1757                    }
1758                } else {
1759                    fill_bit_buffer(&mut l, &mut in_iter);
1760
1761                    let (symbol, code_len) = r.tables[LITLEN_TABLE].lookup(l.bit_buf);
1762
1763                    l.counter = symbol as u32;
1764                    l.bit_buf >>= code_len;
1765                    l.num_bits -= code_len;
1766
1767                    if (l.counter & 256) != 0 {
1768                        // The symbol is not a literal.
1769                        Action::Jump(HuffDecodeOuterLoop1)
1770                    } else {
1771                        // If we have a 32-bit buffer we need to read another two bytes now
1772                        // to have enough bits to keep going.
1773                        if cfg!(not(target_pointer_width = "64")) {
1774                            fill_bit_buffer(&mut l, &mut in_iter);
1775                        }
1776
1777                        let (symbol, code_len) = r.tables[LITLEN_TABLE].lookup(l.bit_buf);
1778
1779                            l.bit_buf >>= code_len;
1780                            l.num_bits -= code_len;
1781                            // The previous symbol was a literal, so write it directly and check
1782                            // the next one.
1783                            out_buf.write_byte(l.counter as u8);
1784                            if (symbol & 256) != 0 {
1785                                l.counter = symbol as u32;
1786                                // The symbol is a length value.
1787                                Action::Jump(HuffDecodeOuterLoop1)
1788                            } else {
1789                                // The symbol is a literal, so write it directly and continue.
1790                                out_buf.write_byte(symbol as u8);
1791                                Action::None
1792                            }
1793
1794                    }
1795
1796                }
1797            }),
1798
1799            WriteSymbol => generate_state!(state, 'state_machine, {
1800                if l.counter >= 256 {
1801                    Action::Jump(HuffDecodeOuterLoop1)
1802                } else if out_buf.bytes_left() > 0 {
1803                    out_buf.write_byte(l.counter as u8);
1804                    Action::Jump(DecodeLitlen)
1805                } else {
1806                    Action::End(TINFLStatus::HasMoreOutput)
1807                }
1808            }),
1809
1810            HuffDecodeOuterLoop1 => generate_state!(state, 'state_machine, {
1811                // Mask the top bits since they may contain length info.
1812                l.counter &= 511;
1813
1814                if l.counter
1815                    == 256 {
1816                    // We hit the end of block symbol.
1817                    Action::Jump(BlockDone)
1818                } else if l.counter > 285 {
1819                    // Invalid code.
1820                    // We already verified earlier that the code is > 256.
1821                    Action::Jump(InvalidLitlen)
1822                } else {
1823                    // # Optimization
1824                    // Mask the value to avoid bounds checks
1825                    // We could use get_unchecked later if can statically verify that
1826                    // this will never go out of bounds.
1827                    l.num_extra =
1828                        LENGTH_EXTRA[(l.counter - 257) as usize & BASE_EXTRA_MASK];
1829                    l.counter = u32::from(LENGTH_BASE[(l.counter - 257) as usize & BASE_EXTRA_MASK]);
1830                    // Length and distance codes have a number of extra bits depending on
1831                    // the base, which together with the base gives us the exact value.
1832                    if l.num_extra != 0 {
1833                        Action::Jump(ReadExtraBitsLitlen)
1834                    } else {
1835                        Action::Jump(DecodeDistance)
1836                    }
1837                }
1838            }),
1839
1840            ReadExtraBitsLitlen => generate_state!(state, 'state_machine, {
1841                let num_extra = l.num_extra.into();
1842                read_bits(&mut l, num_extra, &mut in_iter, flags, |l, extra_bits| {
1843                    l.counter += extra_bits as u32;
1844                    Action::Jump(DecodeDistance)
1845                })
1846            }),
1847
1848            DecodeDistance => generate_state!(state, 'state_machine, {
1849                // Try to read a huffman code from the input buffer and look up what
1850                // length code the decoded symbol refers to.
1851                decode_huffman_code(r, &mut l, DIST_TABLE, flags, &mut in_iter, |_r, l, symbol| {
1852                    // # Optimizaton - transform the value into usize here before the check so
1853                    // the compiler can optimize the bounds check later - ideally it should
1854                    // know that the value can't be negative from earlier in the
1855                    // decode_huffman_code function but it seems it may not be able
1856                    // to make the assumption that it can't be negative and thus
1857                    // overflow if it's converted after the check.
1858                    let symbol = symbol as usize;
1859                    if symbol > 29 {
1860                        // Invalid distance code.
1861                        return Action::Jump(InvalidDist)
1862                    }
1863                    l.num_extra = num_extra_bits_for_distance_code(symbol as u8);
1864                    l.dist = u32::from(DIST_BASE[symbol]);
1865                    if l.num_extra != 0 {
1866                        // ReadEXTRA_BITS_DISTACNE
1867                        Action::Jump(ReadExtraBitsDistance)
1868                    } else {
1869                        Action::Jump(HuffDecodeOuterLoop2)
1870                    }
1871                })
1872            }),
1873
1874            ReadExtraBitsDistance => generate_state!(state, 'state_machine, {
1875                let num_extra = l.num_extra.into();
1876                read_bits(&mut l, num_extra, &mut in_iter, flags, |l, extra_bits| {
1877                    l.dist += extra_bits as u32;
1878                    Action::Jump(HuffDecodeOuterLoop2)
1879                })
1880            }),
1881
1882            HuffDecodeOuterLoop2 => generate_state!(state, 'state_machine, {
1883                if (l.dist as usize > out_buf.position() &&
1884                    (flags & TINFL_FLAG_USING_NON_WRAPPING_OUTPUT_BUF != 0)) || (l.dist as usize > out_buf.get_ref().len())
1885                {
1886                    // We encountered a distance that refers a position before
1887                    // the start of the decoded data, so we can't continue.
1888                    Action::Jump(DistanceOutOfBounds)
1889                } else {
1890                    let out_pos = out_buf.position();
1891                    let source_pos = out_buf.position()
1892                        .wrapping_sub(l.dist as usize) & out_buf_size_mask;
1893
1894                    let out_len = out_buf.bytes_left();
1895                    let match_end_pos = out_buf.position() + l.counter as usize;
1896
1897                    if match_end_pos > out_len ||
1898                        // miniz doesn't do this check here. Not sure how it makes sure
1899                        // that this case doesn't happen.
1900                        (source_pos >= out_pos && (source_pos - out_pos) < l.counter as usize)
1901                    {
1902                        // Not enough space for all of the data in the output buffer,
1903                        // so copy what we have space for.
1904                        if l.counter == 0 {
1905                            Action::Jump(DecodeLitlen)
1906                        } else {
1907                            Action::Jump(WriteLenBytesToEnd)
1908                        }
1909                    } else {
1910                        apply_match(
1911                            out_buf.get_mut(),
1912                            out_pos,
1913                            l.dist as usize,
1914                            l.counter as usize,
1915                            out_buf_size_mask
1916                        );
1917                        out_buf.set_position(out_pos + l.counter as usize);
1918                        Action::Jump(DecodeLitlen)
1919                    }
1920                }
1921            }),
1922
1923            WriteLenBytesToEnd => generate_state!(state, 'state_machine, {
1924                if out_buf.bytes_left() > 0 {
1925                    let out_pos = out_buf.position();
1926                    let source_pos = out_buf.position()
1927                        .wrapping_sub(l.dist as usize) & out_buf_size_mask;
1928
1929
1930                    let len = cmp::min(out_buf.bytes_left(), l.counter as usize);
1931
1932                    transfer(out_buf.get_mut(), source_pos, out_pos, len, out_buf_size_mask);
1933
1934                    out_buf.set_position(out_pos + len);
1935                    l.counter -= len as u32;
1936                    if l.counter == 0 {
1937                        Action::Jump(DecodeLitlen)
1938                    } else {
1939                        Action::None
1940                    }
1941                } else {
1942                    Action::End(TINFLStatus::HasMoreOutput)
1943                }
1944            }),
1945
1946            BlockDone => generate_state!(state, 'state_machine, {
1947                // End once we've read the last block.
1948                if r.finish != 0 {
1949                    pad_to_bytes(&mut l, &mut in_iter, flags, |_| Action::None);
1950
1951                    let in_consumed = in_buf.len() - in_iter.bytes_left();
1952                    let undo = undo_bytes(&mut l, in_consumed as u32) as usize;
1953                    in_iter = InputWrapper::from_slice(in_buf[in_consumed - undo..].iter().as_slice());
1954
1955                    l.bit_buf &= ((1 as BitBuffer) << l.num_bits) - 1;
1956                    debug_assert_eq!(l.num_bits, 0);
1957
1958                    if flags & TINFL_FLAG_PARSE_ZLIB_HEADER != 0 {
1959                        l.counter = 0;
1960                        Action::Jump(ReadAdler32)
1961                    } else {
1962                        Action::Jump(DoneForever)
1963                    }
1964                } else {
1965                    #[cfg(feature = "block-boundary")]
1966                    if flags & TINFL_FLAG_STOP_ON_BLOCK_BOUNDARY != 0 {
1967                        Action::End(TINFLStatus::BlockBoundary)
1968                    } else {
1969                        Action::Jump(ReadBlockHeader)
1970                    }
1971                    #[cfg(not(feature = "block-boundary"))]
1972                    {
1973                        Action::Jump(ReadBlockHeader)
1974                    }
1975                }
1976            }),
1977
1978            ReadAdler32 => generate_state!(state, 'state_machine, {
1979                if l.counter < 4 {
1980                    if l.num_bits != 0 {
1981                        read_bits(&mut l, 8, &mut in_iter, flags, |l, bits| {
1982                            r.z_adler32 <<= 8;
1983                            r.z_adler32 |= bits as u32;
1984                            l.counter += 1;
1985                            Action::None
1986                        })
1987                    } else {
1988                        read_byte(&mut in_iter, flags, |byte| {
1989                            r.z_adler32 <<= 8;
1990                            r.z_adler32 |= u32::from(byte);
1991                            l.counter += 1;
1992                            Action::None
1993                        })
1994                    }
1995                } else {
1996                    Action::Jump(DoneForever)
1997                }
1998            }),
1999
2000            // We are done.
2001            DoneForever => break TINFLStatus::Done,
2002
2003            // Anything else indicates failure.
2004            // BadZlibHeader | BadRawLength | BadDistOrLiteralTableLength | BlockTypeUnexpected |
2005            // DistanceOutOfBounds |
2006            // BadTotalSymbols | BadCodeSizeDistPrevLookup | BadCodeSizeSum | InvalidLitlen |
2007            // InvalidDist | InvalidCodeLen
2008            _ => break TINFLStatus::Failed,
2009        };
2010    };
2011
2012    let in_undo = if status != TINFLStatus::NeedsMoreInput
2013        && status != TINFLStatus::FailedCannotMakeProgress
2014    {
2015        undo_bytes(&mut l, (in_buf.len() - in_iter.bytes_left()) as u32) as usize
2016    } else {
2017        0
2018    };
2019
2020    // If we're returning after completing a block, prepare for the next block when called again.
2021    #[cfg(feature = "block-boundary")]
2022    if status == TINFLStatus::BlockBoundary {
2023        state = State::ReadBlockHeader;
2024    }
2025
2026    // Make sure HasMoreOutput overrides NeedsMoreInput if the output buffer is full.
2027    // (Unless the missing input is the adler32 value in which case we don't need to write anything.)
2028    // TODO: May want to see if we can do this in a better way.
2029    if status == TINFLStatus::NeedsMoreInput
2030        && out_buf.bytes_left() == 0
2031        && state != State::ReadAdler32
2032    {
2033        status = TINFLStatus::HasMoreOutput
2034    }
2035
2036    r.state = state;
2037    r.bit_buf = l.bit_buf;
2038    r.num_bits = l.num_bits;
2039    r.dist = l.dist;
2040    r.counter = l.counter;
2041    r.num_extra = l.num_extra;
2042
2043    r.bit_buf &= ((1 as BitBuffer) << r.num_bits) - 1;
2044
2045    // If this is a zlib stream, and update the adler32 checksum with the decompressed bytes if
2046    // requested.
2047    let need_adler = if (flags & TINFL_FLAG_IGNORE_ADLER32) == 0 {
2048        flags & (TINFL_FLAG_PARSE_ZLIB_HEADER | TINFL_FLAG_COMPUTE_ADLER32) != 0
2049    } else {
2050        // If TINFL_FLAG_IGNORE_ADLER32 is enabled, ignore the checksum.
2051        false
2052    };
2053    if need_adler && status as i32 >= 0 {
2054        let out_buf_pos = out_buf.position();
2055        r.check_adler32 = update_adler32(r.check_adler32, &out_buf.get_ref()[out_pos..out_buf_pos]);
2056
2057        // disabled so that random input from fuzzer would not be rejected early,
2058        // before it has a chance to reach interesting parts of code
2059        if !cfg!(fuzzing) {
2060            // Once we are done, check if the checksum matches with the one provided in the zlib header.
2061            if status == TINFLStatus::Done
2062                && flags & TINFL_FLAG_PARSE_ZLIB_HEADER != 0
2063                && r.check_adler32 != r.z_adler32
2064            {
2065                status = TINFLStatus::Adler32Mismatch;
2066            }
2067        }
2068    }
2069
2070    (
2071        status,
2072        in_buf.len() - in_iter.bytes_left() - in_undo,
2073        out_buf.position() - out_pos,
2074    )
2075}
