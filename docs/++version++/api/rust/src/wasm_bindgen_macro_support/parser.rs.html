24/// Return an [`Err`] if the given string contains a comment close syntax (`*/``).
25fn check_js_comment_close(str: &str, span: Span) -> Result<(), Diagnostic> {
26    if str.contains("*/") {
27        Err(Diagnostic::span_error(
28            span,
29            "contains comment close syntax",
30        ))
31    } else {
32        Ok(())
33    }
34}

36/// Return an [`Err`] if the given string is a JS keyword or contains a comment close syntax (`*/``).
37fn check_invalid_type(str: &str, span: Span) -> Result<(), Diagnostic> {
38    if is_js_keyword(str) {
39        return Err(Diagnostic::span_error(span, "collides with JS keyword"));
40    }
41    check_js_comment_close(str, span)?;
42    Ok(())
43}

128macro_rules! methods {
129    ($(($name:ident, $invalid_unused:literal, $variant:ident($($contents:tt)*)),)*) => {
130        $(methods!(@method $name, $variant($($contents)*));)*
131
132        fn enforce_used(self) -> Result<(), Diagnostic> {
133            // Account for the fact this method was called
134            ATTRS.with(|state| state.checks.set(state.checks.get() + 1));
135
136            let mut errors = Vec::new();
137            for (used, attr) in self.attrs.iter() {
138                if used.get() {
139                    continue
140                }
141                let span = match attr {
142                    $(BindgenAttr::$variant(span, ..) => span,)*
143                };
144                errors.push(Diagnostic::span_error(*span, "unused wasm_bindgen attribute"));
145            }
146            Diagnostic::from_vec(errors)
147        }
148
149        fn check_used(self) {
150            // Account for the fact this method was called
151            ATTRS.with(|state| {
152                state.checks.set(state.checks.get() + 1);
153
154                state.unused_attrs.borrow_mut().extend(
155                    self.attrs
156                    .iter()
157                    .filter_map(|(used, attr)| if used.get() { None } else { Some(attr) })
158                    .map(|attr| {
159                        match attr {
160                            $(BindgenAttr::$variant(span, ..) => {
161                                UnusedState {
162                                    error: $invalid_unused,
163                                    ident: syn::parse_quote_spanned!(*span => $name)
164                                }
165                            })*
166                        }
167                    })
168                );
169            });
170        }
171    };
172
173    (@method $name:ident, $variant:ident(Span, String, Span)) => {
174        pub(crate) fn $name(&self) -> Option<(&str, Span)> {
175            self.attrs
176                .iter()
177                .find_map(|a| match &a.1 {
178                    BindgenAttr::$variant(_, s, span) => {
179                        a.0.set(true);
180                        Some((&s[..], *span))
181                    }
182                    _ => None,
183                })
184        }
185    };
186
187    (@method $name:ident, $variant:ident(Span, JsNamespace, Vec<Span>)) => {
188        pub(crate) fn $name(&self) -> Option<(JsNamespace, &[Span])> {
189            self.attrs

232impl BindgenAttrs {
233    /// Find and parse the wasm_bindgen attributes.
234    fn find(attrs: &mut Vec<syn::Attribute>) -> Result<BindgenAttrs, Diagnostic> {
235        let mut ret = BindgenAttrs::default();
236        loop {
237            let pos = attrs
238                .iter()
239                .enumerate()
240                .find(|&(_, m)| m.path().segments[0].ident == "wasm_bindgen")
241                .map(|a| a.0);
242            let pos = match pos {
243                Some(i) => i,
244                None => return Ok(ret),
245            };
246            let attr = attrs.remove(pos);
247            let tokens = match attr.meta {
248                syn::Meta::Path(_) => continue,
249                syn::Meta::List(syn::MetaList {
250                    delimiter: MacroDelimiter::Paren(_),
251                    tokens,
252                    ..
253                }) => tokens,
254                syn::Meta::List(_) | syn::Meta::NameValue(_) => {
255                    bail_span!(attr, "malformed #[wasm_bindgen] attribute")
256                }
257            };
258            let mut attrs: BindgenAttrs = syn::parse2(tokens)?;
259            ret.attrs.append(&mut attrs.attrs);
260            attrs.check_used();
261        }
262    }
263
264    fn get_thread_local(&self) -> Result<Option<ThreadLocal>, Diagnostic> {
265        let mut thread_local = self.thread_local_v2().map(|_| ThreadLocal::V2);
266
267        if let Some(span) = self.thread_local() {
268            if thread_local.is_some() {
269                return Err(Diagnostic::span_error(
270                    *span,
271                    "`thread_local` can't be used with `thread_local_v2`",
272                ));
273            } else {
274                thread_local = Some(ThreadLocal::V1)
275            }
276        }
277
278        Ok(thread_local)
279    }

281    attrgen!(methods);
282}

318impl Parse for BindgenAttr {
319    fn parse(input: ParseStream) -> SynResult<Self> {
320        let original = input.fork();
321        let attr: AnyIdent = input.parse()?;
322        let attr = attr.0;
323        let attr_span = attr.span();
324        let attr_string = attr.to_string();
325        let raw_attr_string = format!("r#{attr_string}");
326
327        macro_rules! parsers {
328            ($(($name:ident, $_:literal, $($contents:tt)*),)*) => {
329                $(
330                    if attr_string == stringify!($name) || raw_attr_string == stringify!($name) {
331                        parsers!(
332                            @parser
333                            $($contents)*
334                        );
335                    }
336                )*
337            };
338
339            (@parser $variant:ident(Span)) => ({
340                return Ok(BindgenAttr::$variant(attr_span));
341            });
342
343            (@parser $variant:ident(Span, Ident)) => ({
344                input.parse::<Token![=]>()?;
345                let ident = input.parse::<AnyIdent>()?.0;
346                return Ok(BindgenAttr::$variant(attr_span, ident))
347            });
348
349            (@parser $variant:ident(Span, Option<String>)) => ({
350                if input.parse::<Token![=]>().is_ok() {
351                    if input.peek(syn::LitStr) {
352                        let litstr = input.parse::<syn::LitStr>()?;
353                        return Ok(BindgenAttr::$variant(attr_span, Some(litstr.value())))
354                    }
355
356                    let ident = input.parse::<AnyIdent>()?.0;
357                    return Ok(BindgenAttr::$variant(attr_span, Some(ident.to_string())))
358                } else {
359                    return Ok(BindgenAttr::$variant(attr_span, None));
360                }
361            });
362
363            (@parser $variant:ident(Span, syn::Path)) => ({
364                input.parse::<Token![=]>()?;
365                return Ok(BindgenAttr::$variant(attr_span, input.parse()?));
366            });
367
368            (@parser $variant:ident(Span, syn::Expr)) => ({
369                input.parse::<Token![=]>()?;
370                return Ok(BindgenAttr::$variant(attr_span, input.parse()?));
371            });
372
373            (@parser $variant:ident(Span, String, Span)) => ({
374                input.parse::<Token![=]>()?;
375                let (val, span) = match input.parse::<syn::LitStr>() {
376                    Ok(str) => (str.value(), str.span()),
377                    Err(_) => {
378                        let ident = input.parse::<AnyIdent>()?.0;
379                        (ident.to_string(), ident.span())
380                    }
381                };
382                return Ok(BindgenAttr::$variant(attr_span, val, span))
383            });
384
385            (@parser $variant:ident(Span, JsNamespace, Vec<Span>)) => ({
386                input.parse::<Token![=]>()?;
387                let (vals, spans) = match input.parse::<syn::ExprArray>() {
388                    Ok(exprs) => {
389                        let mut vals = vec![];
390                        let mut spans = vec![];
391
392                        for expr in exprs.elems.iter() {
393                            if let syn::Expr::Lit(syn::ExprLit {
394                                lit: syn::Lit::Str(ref str),
395                                ..
396                            }) = expr {
397                                vals.push(str.value());
398                                spans.push(str.span());
399                            } else {
400                                return Err(syn::Error::new(expr.span(), "expected string literals"));
401                            }
402                        }
403
404                        if vals.is_empty() {
405                            return Err(syn::Error::new(exprs.span(), "Empty namespace lists are not allowed."));
406                        }
407
408                        (vals, spans)
409                    },
410                    // Try parsing as a string literal, then fall back to identifier
411                    Err(_) => match input.parse::<syn::LitStr>() {
412                        Ok(str) => (vec![str.value()], vec![str.span()]),
413                        Err(_) => {
414                            let ident = input.parse::<AnyIdent>()?.0;
415                            (vec![ident.to_string()], vec![ident.span()])
416                        }
417                    }
418                };
419
420                let first = &vals[0];
421                if is_non_value_js_keyword(first) && first != "default" {
422                    let msg = format!("Namespace cannot start with the JS keyword `{}`", first);
423                    return Err(syn::Error::new(spans[0], msg));
424                }
425
426                return Ok(BindgenAttr::$variant(attr_span, JsNamespace(vals), spans))
427            });
428        }
429
430        attrgen!(parsers);
431
432        Err(original.error(if attr_string.starts_with('_') {
433            "unknown attribute: it's safe to remove unused attributes entirely."
434        } else {
435            "unknown attribute"
436        }))
437    }
438}

464impl ConvertToAst<&ast::Program> for &mut syn::ItemStruct {
465    type Target = ast::Struct;
466
467    fn convert(self, program: &ast::Program) -> Result<Self::Target, Diagnostic> {
468        if !self.generics.params.is_empty() {
469            bail_span!(
470                self.generics,
471                "structs with #[wasm_bindgen] cannot have lifetime or \
472                 type parameters currently"
473            );
474        }
475        let attrs = BindgenAttrs::find(&mut self.attrs)?;
476
477        // the `wasm_bindgen` option has been used before
478        let _ = attrs.wasm_bindgen();
479
480        let mut fields = Vec::new();
481        let js_name = attrs
482            .js_name()
483            .map(|s| s.0.to_string())
484            .unwrap_or(self.ident.unraw().to_string());
485        if is_js_keyword(&js_name) && js_name != "default" {
486            bail_span!(
487                self.ident,
488                "struct cannot use the JS keyword `{}` as its name",
489                js_name
490            );
491        }
492
493        let is_inspectable = attrs.inspectable().is_some();
494        let getter_with_clone = attrs.getter_with_clone();
495        for (i, field) in self.fields.iter_mut().enumerate() {
496            match field.vis {
497                syn::Visibility::Public(..) => {}
498                _ => continue,
499            }
500            let (js_field_name, member) = match &field.ident {
501                Some(ident) => (ident.unraw().to_string(), syn::Member::Named(ident.clone())),
502                None => (i.to_string(), syn::Member::Unnamed(i.into())),
503            };
504
505            let attrs = BindgenAttrs::find(&mut field.attrs)?;
506            if attrs.skip().is_some() {
507                attrs.check_used();
508                continue;
509            }
510
511            let js_field_name = match attrs.js_name() {
512                Some((name, _)) => name.to_string(),
513                None => js_field_name,
514            };
515
516            let comments = extract_doc_comments(&field.attrs);
517            let getter = wasm_bindgen_shared::struct_field_get(&js_name, &js_field_name);
518            let setter = wasm_bindgen_shared::struct_field_set(&js_name, &js_field_name);
519
520            fields.push(ast::StructField {
521                rust_name: member,
522                js_name: js_field_name,
523                struct_name: self.ident.clone(),
524                readonly: attrs.readonly().is_some(),
525                ty: field.ty.clone(),
526                getter: Ident::new(&getter, Span::call_site()),
527                setter: Ident::new(&setter, Span::call_site()),
528                comments,
529                generate_typescript: attrs.skip_typescript().is_none(),
530                generate_jsdoc: attrs.skip_jsdoc().is_none(),
531                getter_with_clone: attrs.getter_with_clone().or(getter_with_clone).copied(),
532                wasm_bindgen: program.wasm_bindgen.clone(),
533            });
534            attrs.check_used();
535        }
536        let generate_typescript = attrs.skip_typescript().is_none();
537        let private = attrs.private().is_some();
538        let comments: Vec<String> = extract_doc_comments(&self.attrs);
539        let js_namespace = attrs.js_namespace().map(|(ns, _)| ns.0);
540        attrs.check_used();
541        Ok(ast::Struct {
542            rust_name: self.ident.clone(),
543            js_name,
544            fields,
545            comments,
546            is_inspectable,
547            generate_typescript,
548            private,
549            js_namespace,
550            wasm_bindgen: program.wasm_bindgen.clone(),
551        })
552    }
553}

571impl<'a> ConvertToAst<(&ast::Program, BindgenAttrs, &'a Option<ast::ImportModule>)>
572    for syn::ForeignItemFn
573{
574    type Target = ast::ImportKind;
575
576    fn convert(
577        self,
578        (program, opts, module): (&ast::Program, BindgenAttrs, &'a Option<ast::ImportModule>),
579    ) -> Result<Self::Target, Diagnostic> {
580        let (mut wasm, _) = function_from_decl(
581            &self.sig.ident,
582            &opts,
583            self.sig.clone(),
584            self.attrs.clone(),
585            self.vis.clone(),
586            FunctionPosition::Extern,
587            None,
588        )?;
589        let catch = opts.catch().is_some();
590        let variadic = opts.variadic().is_some();
591        let js_ret = if catch {
592            // TODO: this assumes a whole bunch:
593            //
594            // * The outer type is actually a `Result`
595            // * The error type is a `JsValue`
596            // * The actual type is the first type parameter
597            //
598            // should probably fix this one day...
599            extract_first_ty_param(wasm.ret.as_ref().map(|ret| &ret.r#type))?
600        } else {
601            wasm.ret.as_ref().map(|ret| ret.r#type.clone())
602        };
603
604        let operation_kind = operation_kind(&opts);
605
606        let kind = if opts.method().is_some() {
607            let class = wasm.arguments.first().ok_or_else(|| {
608                err_span!(self, "imported methods must have at least one argument")
609            })?;
610            let class = match get_ty(&class.pat_type.ty) {
611                syn::Type::Reference(syn::TypeReference {
612                    mutability: None,
613                    elem,
614                    ..
615                }) => &**elem,
616                _ => bail_span!(
617                    class.pat_type.ty,
618                    "first argument of method must be a shared reference"
619                ),
620            };
621            let class_name = match get_ty(class) {
622                syn::Type::Path(syn::TypePath {
623                    qself: None,
624                    ref path,
625                }) => path,
626                _ => bail_span!(class, "first argument of method must be a path"),
627            };
628            let class_name = extract_path_ident(class_name)?;
629            let class_name = opts
630                .js_class()
631                .map(|p| p.0.into())
632                .unwrap_or_else(|| class_name.to_string());
633
634            let kind = ast::MethodKind::Operation(ast::Operation {
635                is_static: false,
636                kind: operation_kind,
637            });
638
639            ast::ImportFunctionKind::Method {
640                class: class_name,
641                ty: class.clone(),
642                kind,
643            }
644        } else if let Some(cls) = opts.static_method_of() {
645            let class = opts
646                .js_class()
647                .map(|p| p.0.into())
648                .unwrap_or_else(|| cls.to_string());
649
650            let ty = syn::Type::Path(syn::TypePath {
651                qself: None,
652                path: syn::Path {
653                    leading_colon: None,
654                    segments: std::iter::once(syn::PathSegment {
655                        ident: cls.clone(),
656                        arguments: syn::PathArguments::None,
657                    })
658                    .collect(),
659                },
660            });
661
662            let kind = ast::MethodKind::Operation(ast::Operation {
663                is_static: true,
664                kind: operation_kind,
665            });
666
667            ast::ImportFunctionKind::Method { class, ty, kind }
668        } else if opts.constructor().is_some() {
669            let class = match js_ret {
670                Some(ref ty) => ty,
671                _ => bail_span!(self, "constructor returns must be bare types"),
672            };
673            let class_name = match get_ty(class) {
674                syn::Type::Path(syn::TypePath {
675                    qself: None,
676                    ref path,
677                }) => path,
678                _ => bail_span!(self, "return value of constructor must be a bare path"),
679            };
680            let class_name = extract_path_ident(class_name)?;
681            let class_name = opts
682                .js_class()
683                .map(|p| p.0.into())
684                .unwrap_or_else(|| class_name.to_string());
685
686            ast::ImportFunctionKind::Method {
687                class: class_name,
688                ty: class.clone(),
689                kind: ast::MethodKind::Constructor,
690            }
691        } else {
692            ast::ImportFunctionKind::Normal
693        };
694
695        // Validate that reexport is not used on methods/constructors/static methods
696        if opts.reexport().is_some() && matches!(kind, ast::ImportFunctionKind::Method { .. }) {
697            return Err(Diagnostic::span_error(
698                self.sig.ident.span(),
699                "`reexport` cannot be used on methods, constructors, or static methods. \
700                Use `reexport` on the type import instead.",
701            ));
702        }
703
704        let shim = {
705            let ns = match kind {
706                ast::ImportFunctionKind::Normal => (0, "n"),
707                ast::ImportFunctionKind::Method { ref class, .. } => (1, &class[..]),
708            };
709            let data = (ns, self.sig.to_token_stream().to_string(), module);
710            format!(
711                "__wbg_{}_{}",
712                wasm.name
713                    .chars()
714                    .filter(|&c| c.is_ascii_alphanumeric() || c == '_')
715                    .collect::<String>(),
716                ShortHash(data)
717            )
718        };
719        if let Some(span) = opts.r#final() {
720            if opts.structural().is_some() {
721                let msg = "cannot specify both `structural` and `final`";
722                return Err(Diagnostic::span_error(*span, msg));
723            }
724        }
725        let assert_no_shim = opts.assert_no_shim().is_some();
726
727        let mut doc_comment = String::new();
728        // Extract the doc comments from our list of attributes.
729        wasm.rust_attrs.retain(|attr| {
730            /// Returns the contents of the passed `#[doc = "..."]` attribute,
731            /// or `None` if it isn't one.
732            fn get_docs(attr: &syn::Attribute) -> Option<String> {
733                if attr.path().is_ident("doc") {
734                    if let syn::Meta::NameValue(syn::MetaNameValue {
735                        value:
736                            syn::Expr::Lit(syn::ExprLit {
737                                lit: Lit::Str(str), ..
738                            }),
739                        ..
740                    }) = &attr.meta
741                    {
742                        Some(str.value())
743                    } else {
744                        None
745                    }
746                } else {
747                    None
748                }
749            }
750
751            if let Some(docs) = get_docs(attr) {
752                if !doc_comment.is_empty() {
753                    // Add newlines between the doc comments
754                    doc_comment.push('\n');
755                }
756                // Add this doc comment to the complete docs
757                doc_comment.push_str(&docs);
758
759                // Remove it from the list of regular attributes
760                false
761            } else {
762                true
763            }
764        });
765
766        let ret = ast::ImportKind::Function(ast::ImportFunction {
767            function: wasm,
768            assert_no_shim,
769            kind,
770            js_ret,
771            catch,
772            variadic,
773            structural: opts.structural().is_some() || opts.r#final().is_none(),
774            rust_name: self.sig.ident,
775            shim: Ident::new(&shim, Span::call_site()),
776            doc_comment,
777            wasm_bindgen: program.wasm_bindgen.clone(),
778            wasm_bindgen_futures: program.wasm_bindgen_futures.clone(),
779        });
780        opts.check_used();
781
782        Ok(ret)
783    }
784}

986/// Returns whether `self` is passed by reference or by value.
987fn get_self_method(r: syn::Receiver) -> ast::MethodSelf {
988    // The tricky part here is that `r` can have many forms. E.g. `self`,
989    // `&self`, `&mut self`, `self: Self`, `self: &Self`, `self: &mut Self`,
990    // `self: Box<Self>`, `self: Rc<Self>`, etc.
991    // Luckily, syn always populates the `ty` field with the type of `self`, so
992    // e.g. `&self` gets the type `&Self`. So we only have check whether the
993    // type is a reference or not.
994    match &*r.ty {
995        syn::Type::Reference(ty) => {
996            if ty.mutability.is_some() {
997                ast::MethodSelf::RefMutable
998            } else {
999                ast::MethodSelf::RefShared
1000            }
1001        }
1002        _ => ast::MethodSelf::ByValue,
1003    }
1004}

1008    Free,
1009    Impl { self_ty: &'a Ident },

1012/// Construct a function (and gets the self type if appropriate) for our AST from a syn function.
1013#[allow(clippy::too_many_arguments)]
1014fn function_from_decl(
1015    decl_name: &syn::Ident,
1016    opts: &BindgenAttrs,
1017    sig: syn::Signature,
1018    attrs: Vec<syn::Attribute>,
1019    vis: syn::Visibility,
1020    position: FunctionPosition,
1021    args_attrs: Option<Vec<FnArgAttrs>>,
1022) -> Result<(ast::Function, Option<ast::MethodSelf>), Diagnostic> {
1023    if sig.variadic.is_some() {
1024        bail_span!(sig.variadic, "can't #[wasm_bindgen] variadic functions");
1025    }
1026    if !sig.generics.params.is_empty() {
1027        bail_span!(
1028            sig.generics,
1029            "can't #[wasm_bindgen] functions with lifetime or type parameters",
1030        );
1031    }
1032
1033    assert_no_lifetimes(&sig)?;
1034
1035    let syn::Signature { inputs, output, .. } = sig;
1036
1037    // A helper function to replace `Self` in the function signature of methods.
1038    // E.g. `fn get(&self) -> Option<Self>` to `fn get(&self) -> Option<MyType>`
1039    // The following comment explains why this is necessary:
1040    // https://github.com/wasm-bindgen/wasm-bindgen/issues/3105#issuecomment-1275160744
1041    let replace_self = |mut t: syn::Type| {
1042        if let FunctionPosition::Impl { self_ty } = position {
1043            // This uses a visitor to replace all occurrences of `Self` with
1044            // the actual type identifier. The visitor guarantees that we find
1045            // all occurrences of `Self`, even if deeply nested and even if
1046            // future Rust versions add more places where `Self` can appear.
1047            struct SelfReplace(Ident);
1048            impl VisitMut for SelfReplace {
1049                fn visit_ident_mut(&mut self, i: &mut proc_macro2::Ident) {
1050                    if i == "Self" {
1051                        *i = self.0.clone();
1052                    }
1053                }
1054            }
1055
1056            let mut replace = SelfReplace(self_ty.clone());
1057            replace.visit_type_mut(&mut t);
1058        }
1059        t
1060    };
1061
1062    // A helper function to replace argument names that are JS keywords.
1063    // E.g. this will replace `fn foo(class: u32)` to `fn foo(_class: u32)`
1064    let replace_colliding_arg = |i: &mut syn::PatType| {
1065        if let syn::Pat::Ident(ref mut i) = *i.pat {
1066            let ident = i.ident.unraw().to_string();
1067            // JS keywords are NEVER allowed as argument names. Since argument
1068            // names are considered an implementation detail in JS, we can
1069            // safely rename them to avoid collisions.
1070            if is_js_keyword(&ident) {
1071                i.ident = Ident::new(format!("_{ident}").as_str(), i.ident.span());
1072            }
1073        }
1074    };
1075
1076    let mut method_self = None;
1077    let mut arguments = Vec::new();
1078    for arg in inputs.into_iter() {
1079        match arg {
1080            syn::FnArg::Typed(mut c) => {
1081                // typical arguments like foo: u32
1082                replace_colliding_arg(&mut c);
1083                *c.ty = replace_self(*c.ty);
1084                arguments.push(c);
1085            }
1086            syn::FnArg::Receiver(r) => {
1087                // the self argument, so self, &self, &mut self, self: Box<Self>, etc.
1088
1089                // `self` is only allowed for `fn`s inside an `impl` block.
1090                match position {
1091                    FunctionPosition::Free => {
1092                        bail_span!(
1093                            r.self_token,
1094                            "the `self` argument is only allowed for functions in `impl` blocks.\n\n\
1095                            If the function is already in an `impl` block, did you perhaps forget to add `#[wasm_bindgen]` to the `impl` block?"
1096                        );
1097                    }
1098                    FunctionPosition::Extern => {
1099                        bail_span!(
1100                            r.self_token,
1101                            "the `self` argument is not allowed for `extern` functions.\n\n\
1102                            Did you perhaps mean `this`? For more information on importing JavaScript functions, see:\n\
1103                            https://wasm-bindgen.github.io/wasm-bindgen/examples/import-js.html"
1104                        );
1105                    }
1106                    FunctionPosition::Impl { .. } => {}
1107                }
1108
1109                // We need to know *how* `self` is passed to the method (by
1110                // value or by reference) to generate the correct JS shim.
1111                assert!(method_self.is_none());
1112                method_self = Some(get_self_method(r));
1113            }
1114        }
1115    }
1116
1117    // process function return data
1118    let ret_ty_override = opts.unchecked_return_type();
1119    let ret_desc = opts.return_description();
1120    let ret = match output {
1121        syn::ReturnType::Default => None,
1122        syn::ReturnType::Type(_, ty) => Some(ast::FunctionReturnData {
1123            r#type: replace_self(*ty),
1124            js_type: ret_ty_override
1125                .as_ref()
1126                .map_or::<Result<_, Diagnostic>, _>(Ok(None), |(ty, span)| {
1127                    check_invalid_type(ty, *span)?;
1128                    Ok(Some(ty.to_string()))
1129                })?,
1130            desc: ret_desc.as_ref().map_or::<Result<_, Diagnostic>, _>(
1131                Ok(None),
1132                |(desc, span)| {
1133                    check_js_comment_close(desc, *span)?;
1134                    Ok(Some(desc.to_string()))
1135                },
1136            )?,
1137        }),
1138    };
1139    // error if there were description or type override specified for
1140    // function return while it doesn't actually return anything
1141    if ret.is_none() && (ret_ty_override.is_some() || ret_desc.is_some()) {
1142        if let Some((_, span)) = ret_ty_override {
1143            return Err(Diagnostic::span_error(
1144                span,
1145                "cannot specify return type for a function that doesn't return",
1146            ));
1147        }
1148        if let Some((_, span)) = ret_desc {
1149            return Err(Diagnostic::span_error(
1150                span,
1151                "cannot specify return description for a function that doesn't return",
1152            ));
1153        }
1154    }
1155
1156    let (name, name_span) = if let Some((js_name, js_name_span)) = opts.js_name() {
1157        let kind = operation_kind(opts);
1158        let prefix = match kind {
1159            OperationKind::Setter(_) => "set_",
1160            _ => "",
1161        };
1162        (format!("{prefix}{js_name}"), js_name_span)
1163    } else {
1164        (decl_name.unraw().to_string(), decl_name.span())
1165    };
1166
1167    Ok((
1168        ast::Function {
1169            name_span,
1170            name,
1171            rust_attrs: attrs,
1172            rust_vis: vis,
1173            r#unsafe: sig.unsafety.is_some(),
1174            r#async: sig.asyncness.is_some(),
1175            generate_typescript: opts.skip_typescript().is_none(),
1176            generate_jsdoc: opts.skip_jsdoc().is_none(),
1177            variadic: opts.variadic().is_some(),
1178            ret,
1179            arguments: arguments
1180                .into_iter()
1181                .zip(
1182                    args_attrs
1183                        .into_iter()
1184                        .flatten()
1185                        .chain(iter::repeat(FnArgAttrs::default())),
1186                )
1187                .map(|(pat_type, attrs)| ast::FunctionArgumentData {
1188                    pat_type,
1189                    js_name: attrs.js_name,
1190                    js_type: attrs.js_type,
1191                    desc: attrs.desc,
1192                })
1193                .collect(),
1194        },
1195        method_self,
1196    ))
1197}

1199/// Helper struct to store extracted function argument attrs
1200#[derive(Default, Clone)]
1201struct FnArgAttrs {
1202    js_name: Option<String>,
1203    js_type: Option<String>,
1204    desc: Option<String>,
1205}
1206
1207/// Extracts function arguments attributes
1208fn extract_args_attrs(sig: &mut syn::Signature) -> Result<Vec<FnArgAttrs>, Diagnostic> {
1209    let mut args_attrs = vec![];
1210    for input in sig.inputs.iter_mut() {
1211        if let syn::FnArg::Typed(pat_type) = input {
1212            let attrs = BindgenAttrs::find(&mut pat_type.attrs)?;
1213            let arg_attrs = FnArgAttrs {
1214                js_name: attrs
1215                    .js_name()
1216                    .map_or(Ok(None), |(js_name_override, span)| {
1217                        if is_js_keyword(js_name_override) || !is_valid_ident(js_name_override) {
1218                            return Err(Diagnostic::span_error(span, "invalid JS identifier"));
1219                        }
1220                        Ok(Some(js_name_override.to_string()))
1221                    })?,
1222                js_type: attrs
1223                    .unchecked_param_type()
1224                    .map_or::<Result<_, Diagnostic>, _>(Ok(None), |(ty, span)| {
1225                        check_invalid_type(ty, span)?;
1226                        Ok(Some(ty.to_string()))
1227                    })?,
1228                desc: attrs
1229                    .param_description()
1230                    .map_or::<Result<_, Diagnostic>, _>(Ok(None), |(description, span)| {
1231                        check_js_comment_close(description, span)?;
1232                        Ok(Some(description.to_string()))
1233                    })?,
1234            };
1235            // throw error for any unused attrs
1236            attrs.enforce_used()?;
1237            args_attrs.push(arg_attrs);
1238        }
1239    }
1240    Ok(args_attrs)
1241}

1243pub(crate) trait MacroParse<Ctx> {
1244    /// Parse the contents of an object into our AST, with a context if necessary.
1245    ///
1246    /// The context is used to have access to the attributes on `#[wasm_bindgen]`, and to allow
1247    /// writing to the output `TokenStream`.
1248    fn macro_parse(self, program: &mut ast::Program, context: Ctx) -> Result<(), Diagnostic>;
1249}
1250
1251impl<'a> MacroParse<(Option<BindgenAttrs>, &'a mut TokenStream)> for syn::Item {
1252    fn macro_parse(
1253        self,
1254        program: &mut ast::Program,
1255        (opts, tokens): (Option<BindgenAttrs>, &'a mut TokenStream),
1256    ) -> Result<(), Diagnostic> {
1257        match self {
1258            syn::Item::Fn(mut f) => {
1259                let opts = opts.unwrap_or_default();
1260                if let Some(path) = opts.wasm_bindgen() {
1261                    program.wasm_bindgen = path.clone();
1262                }
1263                if let Some(path) = opts.js_sys() {
1264                    program.js_sys = path.clone();
1265                }
1266                if let Some(path) = opts.wasm_bindgen_futures() {
1267                    program.wasm_bindgen_futures = path.clone();
1268                }
1269
1270                if opts.main().is_some() {
1271                    opts.check_used();
1272                    return main(program, f, tokens);
1273                }
1274
1275                let no_mangle = f
1276                    .attrs
1277                    .iter()
1278                    .enumerate()
1279                    .find(|(_, m)| m.path().is_ident("no_mangle"));
1280                if let Some((i, _)) = no_mangle {
1281                    f.attrs.remove(i);
1282                }
1283                // extract fn args attributes before parsing to tokens stream
1284                let args_attrs = extract_args_attrs(&mut f.sig)?;
1285                let comments = extract_doc_comments(&f.attrs);
1286                // If the function isn't used for anything other than being exported to JS,
1287                // it'll be unused when not building for the Wasm target and produce a
1288                // `dead_code` warning. So, add `#[allow(dead_code)]` before it to avoid that.
1289                tokens.extend(quote::quote! { #[allow(dead_code)] });
1290                f.to_tokens(tokens);
1291                if opts.start().is_some() {
1292                    if !f.sig.generics.params.is_empty() {
1293                        bail_span!(&f.sig.generics, "the start function cannot have generics",);
1294                    }
1295                    if !f.sig.inputs.is_empty() {
1296                        bail_span!(&f.sig.inputs, "the start function cannot have arguments",);
1297                    }
1298                }
1299                let method_kind = ast::MethodKind::Operation(ast::Operation {
1300                    is_static: true,
1301                    kind: operation_kind(&opts),
1302                });
1303                let rust_name = f.sig.ident.clone();
1304                let start = opts.start().is_some();
1305
1306                if opts.this().is_some() && f.sig.inputs.is_empty() {
1307                    bail_span!(
1308                        &f.sig.inputs,
1309                        "functions taking a 'this' argument must have at least one parameter"
1310                    );
1311                }
1312
1313                let js_namespace = opts.js_namespace().map(|(ns, _)| ns.0);
1314                program.exports.push(ast::Export {
1315                    comments,
1316                    function: f.convert((opts, args_attrs))?,
1317                    js_class: None,
1318                    js_namespace,
1319                    method_kind,
1320                    method_self: None,
1321                    rust_class: None,
1322                    rust_name,
1323                    start,
1324                    wasm_bindgen: program.wasm_bindgen.clone(),
1325                    wasm_bindgen_futures: program.wasm_bindgen_futures.clone(),
1326                });
1327            }
1328            syn::Item::Impl(mut i) => {
1329                let opts = opts.unwrap_or_default();
1330                (&mut i).macro_parse(program, opts)?;
1331                i.to_tokens(tokens);
1332            }
1333            syn::Item::ForeignMod(mut f) => {
1334                let opts = match opts {
1335                    Some(opts) => opts,
1336                    None => BindgenAttrs::find(&mut f.attrs)?,
1337                };
1338                f.macro_parse(program, opts)?;
1339            }
1340            syn::Item::Enum(mut e) => {
1341                let opts = match opts {
1342                    Some(opts) => opts,
1343                    None => BindgenAttrs::find(&mut e.attrs)?,
1344                };
1345                e.macro_parse(program, (tokens, opts))?;
1346            }
1347            syn::Item::Const(mut c) => {
1348                let opts = match opts {
1349                    Some(opts) => opts,
1350                    None => BindgenAttrs::find(&mut c.attrs)?,
1351                };
1352                c.macro_parse(program, opts)?;
1353            }
1354            _ => {
1355                bail_span!(
1356                    self,
1357                    "#[wasm_bindgen] can only be applied to a function, \
1358                     struct, enum, impl, or extern block",
1359                );
1360            }
1361        }
1362
1363        Ok(())
1364    }
1365}

1412// Prepare for recursion into an `impl` block. Here we want to attach an
1413// internal attribute, `__wasm_bindgen_class_marker`, with any metadata we need
1414// to pass from the impl to the impl item. Recursive macro expansion will then
1415// expand the `__wasm_bindgen_class_marker` attribute.
1416//
1417// Note that we currently do this because inner items may have things like cfgs
1418// on them, so we want to expand the impl first, let the insides get cfg'd, and
1419// then go for the rest.
1420fn prepare_for_impl_recursion(
1421    item: &mut syn::ImplItem,
1422    class: &syn::Path,
1423    program: &ast::Program,
1424    impl_opts: &BindgenAttrs,
1425) -> Result<(), Diagnostic> {
1426    let method = match item {
1427        syn::ImplItem::Fn(m) => m,
1428        syn::ImplItem::Const(_) => {
1429            bail_span!(
1430                &*item,
1431                "const definitions aren't supported with #[wasm_bindgen]"
1432            );
1433        }
1434        syn::ImplItem::Type(_) => bail_span!(
1435            &*item,
1436            "type definitions in impls aren't supported with #[wasm_bindgen]"
1437        ),
1438        syn::ImplItem::Macro(_) => {
1439            // In theory we want to allow this, but we have no way of expanding
1440            // the macro and then placing our magical attributes on the expanded
1441            // functions. As a result, just disallow it for now to hopefully
1442            // ward off buggy results from this macro.
1443            bail_span!(&*item, "macros in impls aren't supported");
1444        }
1445        syn::ImplItem::Verbatim(_) => panic!("unparsed impl item?"),
1446        other => bail_span!(other, "failed to parse this item as a known item"),
1447    };
1448
1449    let ident = extract_path_ident(class)?;
1450
1451    let js_class = impl_opts
1452        .js_class()
1453        .map(|s| s.0.to_string())
1454        .unwrap_or(ident.to_string());
1455
1456    let wasm_bindgen = &program.wasm_bindgen;
1457    let wasm_bindgen_futures = &program.wasm_bindgen_futures;
1458    method.attrs.insert(
1459        0,
1460        syn::Attribute {
1461            pound_token: Default::default(),
1462            style: syn::AttrStyle::Outer,
1463            bracket_token: Default::default(),
1464            meta: syn::parse_quote! { #wasm_bindgen::prelude::__wasm_bindgen_class_marker(#class = #js_class, wasm_bindgen = #wasm_bindgen, wasm_bindgen_futures = #wasm_bindgen_futures) },
1465        },
1466    );
1467
1468    Ok(())
1469}

1555fn string_enum(
1556    enum_: syn::ItemEnum,
1557    program: &mut ast::Program,
1558    js_name: String,
1559    generate_typescript: bool,
1560    comments: Vec<String>,
1561    js_namespace: Option<Vec<String>>,
1562) -> Result<(), Diagnostic> {
1563    let mut variants = vec![];
1564    let mut variant_values = vec![];
1565
1566    for v in enum_.variants.iter() {
1567        let (_, expr) = match &v.discriminant {
1568            Some(pair) => pair,
1569            None => {
1570                bail_span!(v, "all variants of a string enum must have a string value");
1571            }
1572        };
1573        match get_expr(expr) {
1574            syn::Expr::Lit(syn::ExprLit {
1575                attrs: _,
1576                lit: syn::Lit::Str(str_lit),
1577            }) => {
1578                variants.push(v.ident.clone());
1579                variant_values.push(str_lit.value());
1580            }
1581            expr => bail_span!(
1582                expr,
1583                "enums with #[wasm_bindgen] cannot mix string and non-string values",
1584            ),
1585        }
1586    }
1587
1588    program.imports.push(ast::Import {
1589        module: None,
1590        js_namespace: None,
1591        reexport: None,
1592        kind: ast::ImportKind::Enum(ast::StringEnum {
1593            vis: enum_.vis,
1594            name: enum_.ident,
1595            export_name: js_name,
1596            variants,
1597            variant_values,
1598            comments,
1599            rust_attrs: enum_.attrs,
1600            generate_typescript,
1601            js_namespace,
1602            wasm_bindgen: program.wasm_bindgen.clone(),
1603        }),
1604    });
1605
1606    Ok(())
1607}

1609/// Represents a possibly negative numeric value as base 10 digits.
1610struct NumericValue<'a> {
1611    negative: bool,
1612    base10_digits: &'a str,
1613}
1614impl<'a> NumericValue<'a> {
1615    fn from_expr(expr: &'a syn::Expr) -> Option<Self> {
1616        match get_expr(expr) {
1617            syn::Expr::Lit(syn::ExprLit {
1618                lit: syn::Lit::Int(int_lit),
1619                ..
1620            }) => Some(Self {
1621                negative: false,
1622                base10_digits: int_lit.base10_digits(),
1623            }),
1624            syn::Expr::Unary(syn::ExprUnary {
1625                op: syn::UnOp::Neg(_),
1626                expr,
1627                ..
1628            }) => Self::from_expr(expr).map(|n| n.neg()),
1629            _ => None,
1630        }
1631    }
1632
1633    fn parse(&self) -> Option<i64> {
1634        let mut value = self.base10_digits.parse::<i64>().ok()?;
1635        if self.negative {
1636            value = -value;
1637        }
1638        Some(value)
1639    }
1640
1641    fn neg(self) -> Self {
1642        Self {
1643            negative: !self.negative,
1644            base10_digits: self.base10_digits,
1645        }
1646    }
1647}

1649impl<'a> MacroParse<(&'a mut TokenStream, BindgenAttrs)> for syn::ItemEnum {
1650    fn macro_parse(
1651        self,
1652        program: &mut ast::Program,
1653        (tokens, opts): (&'a mut TokenStream, BindgenAttrs),
1654    ) -> Result<(), Diagnostic> {
1655        if self.variants.is_empty() {
1656            bail_span!(self, "cannot export empty enums to JS");
1657        }
1658        for variant in self.variants.iter() {
1659            match variant.fields {
1660                syn::Fields::Unit => (),
1661                _ => bail_span!(
1662                    variant.fields,
1663                    "enum variants with associated data are not supported with #[wasm_bindgen]"
1664                ),
1665            }
1666        }
1667
1668        let generate_typescript = opts.skip_typescript().is_none();
1669        let private = opts.private().is_some();
1670        let comments = extract_doc_comments(&self.attrs);
1671        let js_name = opts
1672            .js_name()
1673            .map(|s| s.0)
1674            .map_or_else(|| self.ident.to_string(), |s| s.to_string());
1675        if is_js_keyword(&js_name) && js_name != "default" {
1676            bail_span!(
1677                self.ident,
1678                "enum cannot use the JS keyword `{}` as its name",
1679                js_name
1680            );
1681        }
1682
1683        let js_namespace = opts.js_namespace().map(|(ns, _)| ns.0);
1684        opts.check_used();
1685
1686        // Check if the enum is a string enum, by checking whether any variant has a string discriminant.
1687        let is_string_enum = self.variants.iter().any(|v| {
1688            if let Some((_, expr)) = &v.discriminant {
1689                if let syn::Expr::Lit(syn::ExprLit {
1690                    lit: syn::Lit::Str(_),
1691                    ..
1692                }) = get_expr(expr)
1693                {
1694                    return true;
1695                }
1696            }
1697            false
1698        });
1699        if is_string_enum {
1700            return string_enum(
1701                self,
1702                program,
1703                js_name,
1704                generate_typescript,
1705                comments,
1706                js_namespace,
1707            );
1708        }
1709
1710        match self.vis {
1711            syn::Visibility::Public(_) => {}
1712            _ => bail_span!(self, "only public enums are allowed with #[wasm_bindgen]"),
1713        }
1714
1715        // Go through all variants once first to determine whether the enum is
1716        // signed or unsigned. We don't need to actually parse the discriminant
1717        // values yet, we just need to know their sign. The actual parsing is
1718        // done in a second pass.
1719        let signed = self.variants.iter().any(|v| match &v.discriminant {
1720            Some((_, expr)) => NumericValue::from_expr(expr).is_some_and(|n| n.negative),
1721            None => false,
1722        });
1723        let underlying_min = if signed { i32::MIN as i64 } else { 0 };
1724        let underlying_max = if signed {
1725            i32::MAX as i64
1726        } else {
1727            u32::MAX as i64
1728        };
1729
1730        let mut last_discriminant: Option<i64> = None;
1731        let mut discriminant_map: HashMap<i64, &syn::Variant> = HashMap::new();
1732
1733        let variants = self
1734            .variants
1735            .iter()
1736            .map(|v| {
1737                let value: i64 = match &v.discriminant {
1738                    Some((_, expr)) => match NumericValue::from_expr(expr).and_then(|n| n.parse()) {
1739                        Some(value) => value,
1740                        _ => bail_span!(
1741                            expr,
1742                            "C-style enums with #[wasm_bindgen] may only have \
1743                             numeric literal values that fit in a 32-bit integer as discriminants. \
1744                             Expressions or variables are not supported.",
1745                        ),
1746                    },
1747                    None => {
1748                        // Use the same algorithm as rustc to determine the next discriminant
1749                        // https://doc.rust-lang.org/reference/items/enumerations.html#implicit-discriminants
1750                        last_discriminant.map_or(0, |last| last + 1)
1751                    }
1752                };
1753
1754                last_discriminant = Some(value);
1755
1756                // check that the value fits within the underlying type
1757                let underlying = if signed { "i32" } else { "u32" };
1758                let numbers = if signed { "signed numbers" } else { "unsigned numbers" };
1759                if value < underlying_min {
1760                    bail_span!(
1761                        v,
1762                        "C-style enums with #[wasm_bindgen] can only support {0} that can be represented by `{2}`, \
1763                        but `{1}` is too small for `{2}`",
1764                        numbers,
1765                        value,
1766                        underlying
1767                    );
1768                }
1769                if value > underlying_max {
1770                    bail_span!(
1771                        v,
1772                        "C-style enums with #[wasm_bindgen] can only support {0} that can be represented by `{2}`, \
1773                        but `{1}` is too large for `{2}`",
1774                        numbers,
1775                        value,
1776                        underlying
1777                    );
1778                }
1779
1780                // detect duplicate discriminants
1781                if let Some(old) = discriminant_map.insert(value, v) {
1782                    bail_span!(
1783                        v,
1784                        "discriminant value `{}` is already used by {} in this enum",
1785                        value,
1786                        old.ident
1787                    );
1788                }
1789
1790                let comments = extract_doc_comments(&v.attrs);
1791                Ok(ast::Variant {
1792                    name: v.ident.clone(),
1793                    // due to the above checks, we know that the value fits
1794                    // within 32 bits, so this cast doesn't lose any information
1795                    value: value as u32,
1796                    comments,
1797                })
1798            })
1799            .collect::<Result<Vec<_>, Diagnostic>>()?;
1800
1801        // To make all the code handling holes simpler, we only consider
1802        // non-negative holes. This allows us to use `u32` to represent holes.
1803        let hole = (0..=underlying_max)
1804            .find(|v| !discriminant_map.contains_key(v))
1805            .unwrap() as u32;
1806
1807        self.to_tokens(tokens);
1808
1809        program.enums.push(ast::Enum {
1810            rust_name: self.ident,
1811            js_name,
1812            signed,
1813            variants,
1814            comments,
1815            hole,
1816            generate_typescript,
1817            private,
1818            js_namespace,
1819            wasm_bindgen: program.wasm_bindgen.clone(),
1820        });
1821        Ok(())
1822    }
1823}

1883impl MacroParse<ForeignItemCtx> for syn::ForeignItem {
1884    fn macro_parse(
1885        mut self,
1886        program: &mut ast::Program,
1887        ctx: ForeignItemCtx,
1888    ) -> Result<(), Diagnostic> {
1889        let item_opts = {
1890            let attrs = match self {
1891                syn::ForeignItem::Fn(ref mut f) => &mut f.attrs,
1892                syn::ForeignItem::Type(ref mut t) => &mut t.attrs,
1893                syn::ForeignItem::Static(ref mut s) => &mut s.attrs,
1894                syn::ForeignItem::Verbatim(v) => {
1895                    let mut item: syn::ItemStatic =
1896                        syn::parse(v.into()).expect("only foreign functions/types allowed for now");
1897                    let item_opts = BindgenAttrs::find(&mut item.attrs)?;
1898                    let reexport = item_opts.reexport().cloned();
1899                    let kind = item.convert((program, item_opts, &ctx.module))?;
1900
1901                    program.imports.push(ast::Import {
1902                        module: None,
1903                        js_namespace: None,
1904                        reexport,
1905                        kind,
1906                    });
1907
1908                    return Ok(());
1909                }
1910                _ => panic!("only foreign functions/types allowed for now"),
1911            };
1912            BindgenAttrs::find(attrs)?
1913        };
1914
1915        let js_namespace = item_opts
1916            .js_namespace()
1917            .map(|(s, _)| s)
1918            .or(ctx.js_namespace)
1919            .map(|s| s.0);
1920        let module = ctx.module;
1921        let reexport = item_opts.reexport().cloned();
1922
1923        let kind = match self {
1924            syn::ForeignItem::Fn(f) => f.convert((program, item_opts, &module))?,
1925            syn::ForeignItem::Type(t) => t.convert((program, item_opts))?,
1926            syn::ForeignItem::Static(s) => s.convert((program, item_opts, &module))?,
1927            _ => panic!("only foreign functions/types allowed for now"),
1928        };
1929
1930        // check for JS keywords
1931
1932        // We only need to check if there isn't a JS namespace or module. If
1933        // there is namespace, then we already checked the namespace while
1934        // parsing. If there is a module, we can rename the import symbol to
1935        // avoid using keywords.
1936        let needs_check = js_namespace.is_none() && module.is_none();
1937        if needs_check {
1938            match &kind {
1939                ast::ImportKind::Function(import_function) => {
1940                    if matches!(import_function.kind, ast::ImportFunctionKind::Normal)
1941                        && is_non_value_js_keyword(&import_function.function.name)
1942                    {
1943                        bail_span!(
1944                            import_function.rust_name,
1945                            "Imported function cannot use the JS keyword `{}` as its name.",
1946                            import_function.function.name
1947                        );
1948                    }
1949                }
1950                ast::ImportKind::Static(import_static) => {
1951                    if is_non_value_js_keyword(&import_static.js_name) {
1952                        bail_span!(
1953                            import_static.rust_name,
1954                            "Imported static cannot use the JS keyword `{}` as its name.",
1955                            import_static.js_name
1956                        );
1957                    }
1958                }
1959                ast::ImportKind::String(_) => {
1960                    // static strings don't have JS names, so we don't need to check for JS keywords
1961                }
1962                ast::ImportKind::Type(import_type) => {
1963                    if is_non_value_js_keyword(&import_type.js_name) {
1964                        bail_span!(
1965                            import_type.rust_name,
1966                            "Imported type cannot use the JS keyword `{}` as its name.",
1967                            import_type.js_name
1968                        );
1969                    }
1970                }
1971                ast::ImportKind::Enum(_) => {
1972                    // string enums aren't possible here
1973                }
1974            }
1975        }
1976
1977        program.imports.push(ast::Import {
1978            module,
1979            js_namespace,
1980            reexport,
1981            kind,
1982        });
1983
1984        Ok(())
1985    }

2032/// Get the first type parameter of a generic type, errors on incorrect input.
2033fn extract_first_ty_param(ty: Option<&syn::Type>) -> Result<Option<syn::Type>, Diagnostic> {
2034    let t = match ty {
2035        Some(t) => t,
2036        None => return Ok(None),
2037    };
2038    let path = match *get_ty(t) {
2039        syn::Type::Path(syn::TypePath {
2040            qself: None,
2041            ref path,

2044    };
2045    let seg = path
2046        .segments
2047        .last()
2048        .ok_or_else(|| err_span!(t, "must have at least one segment"))?;
2049    let generics = match seg.arguments {
2050        syn::PathArguments::AngleBracketed(ref t) => t,
2051        _ => bail_span!(t, "must be Result<...>"),

2054        .args

2056        .ok_or_else(|| err_span!(t, "must have at least one generic parameter"))?;

2058        syn::GenericArgument::Type(t) => t,
2059        other => bail_span!(other, "must be a type parameter"),
2060    };
2061    match get_ty(ty) {

2063        _ => {}
2064    }

2071        .iter()
2072        .filter_map(|a| {

2074            // this is a doc comment

2103// Unescapes a quoted string. char::escape_debug() was used to escape the text.
2104fn try_unescape(mut s: &str) -> Option<String> {
2105    s = s.strip_prefix('"').unwrap_or(s);
2106    s = s.strip_suffix('"').unwrap_or(s);
2107    let mut result = String::with_capacity(s.len());
2108    let mut chars = s.chars();
2109    while let Some(c) = chars.next() {
2110        if c == '\\' {
2111            let c = chars.next()?;
2112            match c {
2113                't' => result.push('\t'),
2114                'r' => result.push('\r'),
2115                'n' => result.push('\n'),
2116                '\\' | '\'' | '"' => result.push(c),
2117                'u' => {
2118                    if chars.next() != Some('{') {
2119                        return None;
2120                    }
2121                    let (c, next) = unescape_unicode(&mut chars)?;
2122                    result.push(c);
2123                    if next != '}' {
2124                        return None;
2125                    }
2126                }
2127                _ => return None,
2128            }
2129        } else {
2130            result.push(c);
2131        }
2132    }
2133    Some(result)
2134}

2136fn unescape_unicode(chars: &mut Chars) -> Option<(char, char)> {
2137    let mut value = 0;
2138    for (i, c) in chars.enumerate() {
2139        match (i, c.to_digit(16)) {
2140            (0..=5, Some(num)) => value = (value << 4) | num,
2141            (1.., None) => return Some((char::from_u32(value)?, c)),
2142            _ => break,
2143        }
2144    }
2145    None
2146}
