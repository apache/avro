25pub fn encode(program: &ast::Program) -> Result<EncodeResult, Diagnostic> {
26    let mut e = Encoder::new();
27    let i = Interner::new();
28    shared_program(program, &i)?.encode(&mut e);
29    let custom_section = e.finish();
30    let included_files = i
31        .files
32        .borrow()
33        .values()
34        .map(|p| &p.path)
35        .cloned()
36        .collect();
37    Ok(EncodeResult {
38        custom_section,
39        included_files,
40    })
41}

43struct Interner {
44    bump: bumpalo::Bump,
45    files: RefCell<HashMap<String, LocalFile>>,
46    root: PathBuf,
47    crate_name: String,
48    has_package_json: Cell<bool>,
49}

51struct LocalFile {
52    path: PathBuf,
53    definition: Span,
54    new_identifier: String,
55    linked_module: bool,
56}
57
58impl Interner {
59    fn new() -> Interner {
60        let root = env::var_os("CARGO_MANIFEST_DIR")
61            .expect("should have CARGO_MANIFEST_DIR env var")
62            .into();
63        let crate_name = env::var("CARGO_PKG_NAME").expect("should have CARGO_PKG_NAME env var");
64        Interner {
65            bump: bumpalo::Bump::new(),
66            files: RefCell::new(HashMap::new()),
67            root,
68            crate_name,
69            has_package_json: Cell::new(false),
70        }
71    }
72
73    fn intern(&self, s: &Ident) -> &str {
74        self.intern_str(&s.to_string())
75    }
76
77    fn intern_str(&self, s: &str) -> &str {
78        // NB: eventually this could be used to intern `s` to only allocate one
79        // copy, but for now let's just "transmute" `s` to have the same
80        // lifetime as this struct itself (which is our main goal here)
81        self.bump.alloc_str(s)
82    }
83
84    /// Given an import to a local module `id` this generates a unique module id
85    /// to assign to the contents of `id`.
86    ///
87    /// Note that repeated invocations of this function will be memoized, so the
88    /// same `id` will always return the same resulting unique `id`.
89    fn resolve_import_module(
90        &self,
91        id: &str,
92        span: Span,
93        linked_module: bool,
94    ) -> Result<ImportModule<'_>, Diagnostic> {
95        let mut files = self.files.borrow_mut();
96        if let Some(file) = files.get(id) {
97            return Ok(ImportModule::Named(self.intern_str(&file.new_identifier)));
98        }
99        self.check_for_package_json();
100        let path = if let Some(id) = id.strip_prefix('/') {
101            self.root.join(id)
102        } else if id.starts_with("./") || id.starts_with("../") {
103            let msg = "relative module paths aren't supported yet";
104            return Err(Diagnostic::span_error(span, msg));
105        } else {
106            return Ok(ImportModule::RawNamed(self.intern_str(id)));
107        };
108
109        // Generate a unique ID which is somewhat readable as well, so mix in
110        // the crate name, hash to make it unique, and then the original path.
111        let new_identifier = format!("{}{id}", self.unique_crate_identifier());
112        let file = LocalFile {
113            path,
114            definition: span,
115            new_identifier,
116            linked_module,
117        };
118        files.insert(id.to_string(), file);
119        drop(files);
120        self.resolve_import_module(id, span, linked_module)
121    }
122
123    fn unique_crate_identifier(&self) -> String {
124        format!("{}-{}", self.crate_name, ShortHash(0))
125    }
126
127    fn check_for_package_json(&self) {
128        if self.has_package_json.get() {
129            return;
130        }
131        let path = self.root.join("package.json");
132        if path.exists() {
133            self.has_package_json.set(true);
134        }
135    }
136}

138fn shared_program<'a>(
139    prog: &'a ast::Program,
140    intern: &'a Interner,
141) -> Result<Program<'a>, Diagnostic> {
142    Ok(Program {
143        exports: prog
144            .exports
145            .iter()
146            .map(|a| shared_export(a, intern))
147            .collect::<Result<Vec<_>, _>>()?,
148        structs: prog
149            .structs
150            .iter()
151            .map(|a| shared_struct(a, intern))
152            .collect(),
153        enums: prog.enums.iter().map(|a| shared_enum(a, intern)).collect(),
154        imports: prog
155            .imports
156            .iter()
157            .map(|a| shared_import(a, intern))
158            .collect::<Result<Vec<_>, _>>()?,
159        typescript_custom_sections: prog
160            .typescript_custom_sections
161            .iter()
162            .map(|x| shared_lit_or_expr(x, intern))
163            .collect(),
164        linked_modules: prog
165            .linked_modules
166            .iter()
167            .enumerate()
168            .map(|(i, a)| shared_linked_module(&prog.link_function_name(i), a, intern))
169            .collect::<Result<Vec<_>, _>>()?,
170        local_modules: intern
171            .files
172            .borrow()
173            .values()
174            .map(|file| {
175                fs::read_to_string(&file.path)
176                    .map(|s| LocalModule {
177                        identifier: intern.intern_str(&file.new_identifier),
178                        contents: intern.intern_str(&s),
179                        linked_module: file.linked_module,
180                    })
181                    .map_err(|e| {
182                        let msg = format!("failed to read file `{}`: {e}", file.path.display());
183                        Diagnostic::span_error(file.definition, msg)
184                    })
185            })
186            .collect::<Result<Vec<_>, _>>()?,
187        inline_js: prog
188            .inline_js
189            .iter()
190            .map(|js| intern.intern_str(js))
191            .collect(),
192        unique_crate_identifier: intern.intern_str(&intern.unique_crate_identifier()),
193        package_json: if intern.has_package_json.get() {
194            Some(intern.intern_str(intern.root.join("package.json").to_str().unwrap()))
195        } else {
196            None
197        },
198    })
199}

221fn shared_function<'a>(func: &'a ast::Function, _intern: &'a Interner) -> Function<'a> {
222    let args =
223        func.arguments
224            .iter()
225            .enumerate()
226            .map(|(idx, arg)| FunctionArgumentData {
227                // use argument's "js_name" if it was provided via attributes
228                // if not use the original Rust argument ident
229                name: arg.js_name.clone().unwrap_or(
230                    if let syn::Pat::Ident(x) = &*arg.pat_type.pat {
231                        x.ident.unraw().to_string()
232                    } else {
233                        format!("arg{idx}")
234                    },
235                ),
236                ty_override: arg.js_type.as_deref(),
237                desc: arg.desc.as_deref(),
238            })
239            .collect::<Vec<_>>();
240
241    Function {
242        args,
243        asyncness: func.r#async,
244        name: &func.name,
245        generate_typescript: func.generate_typescript,
246        generate_jsdoc: func.generate_jsdoc,
247        variadic: func.variadic,
248        ret_ty_override: func.ret.as_ref().and_then(|v| v.js_type.as_deref()),
249        ret_desc: func.ret.as_ref().and_then(|v| v.desc.as_deref()),
250    }
251}

280fn shared_import<'a>(i: &'a ast::Import, intern: &'a Interner) -> Result<Import<'a>, Diagnostic> {
281    // Resolve reexport name: use explicit rename if provided, otherwise use the import's name
282    let reexport = i.reexport.as_ref().map(|rename_opt| {
283        rename_opt.clone().unwrap_or_else(|| {
284            // Get the default name from the import kind
285            match &i.kind {
286                ast::ImportKind::Type(t) => t.js_name.clone(),
287                ast::ImportKind::Function(f) => f.function.name.clone(),
288                ast::ImportKind::Static(s) => s.js_name.clone(),
289                _ => unreachable!("reexport only supported on types, functions, and statics"),
290            }
291        })
292    });
293
294    Ok(Import {
295        module: i
296            .module
297            .as_ref()
298            .map(|m| shared_module(m, intern, false))
299            .transpose()?,
300        js_namespace: i.js_namespace.clone(),
301        reexport,
302        kind: shared_import_kind(&i.kind, intern)?,
303    })
304}

438trait Encode {
439    fn encode(&self, dst: &mut Encoder);
440}

442struct Encoder {
443    dst: Vec<EncodeChunk>,
444}
445
446enum LitOrExpr<'a> {
447    Expr(&'a syn::Expr),
448    Lit(&'a str),
449}
450
451impl Encode for LitOrExpr<'_> {
452    fn encode(&self, dst: &mut Encoder) {
453        match self {
454            LitOrExpr::Expr(expr) => {
455                dst.dst.push(EncodeChunk::StrExpr((*expr).clone()));
456            }
457            LitOrExpr::Lit(s) => s.encode(dst),
458        }
459    }
460}
461
462impl Encoder {
463    fn new() -> Encoder {
464        Encoder { dst: vec![] }
465    }
466
467    fn finish(self) -> Vec<EncodeChunk> {
468        self.dst
469    }
470
471    fn byte(&mut self, byte: u8) {
472        if let Some(EncodeChunk::EncodedBuf(buf)) = self.dst.last_mut() {
473            buf.push(byte);
474        } else {
475            self.dst.push(EncodeChunk::EncodedBuf(vec![byte]));
476        }
477    }
478
479    fn extend_from_slice(&mut self, slice: &[u8]) {
480        if let Some(EncodeChunk::EncodedBuf(buf)) = self.dst.last_mut() {
481            buf.extend_from_slice(slice);
482        } else {
483            self.dst.push(EncodeChunk::EncodedBuf(slice.to_owned()));
484        }
485    }
486}
487
488impl Encode for bool {
489    fn encode(&self, dst: &mut Encoder) {
490        dst.byte(*self as u8);
491    }
492}
493
494impl Encode for u32 {
495    fn encode(&self, dst: &mut Encoder) {
496        let mut val = *self;
497        while (val >> 7) != 0 {
498            dst.byte((val as u8) | 0x80);
499            val >>= 7;
500        }
501        assert_eq!(val >> 7, 0);
502        dst.byte(val as u8);
503    }
504}
505
506impl Encode for usize {
507    fn encode(&self, dst: &mut Encoder) {
508        assert!(*self <= u32::MAX as usize);
509        (*self as u32).encode(dst);
510    }
511}

513impl Encode for &[u8] {
514    fn encode(&self, dst: &mut Encoder) {
515        self.len().encode(dst);
516        dst.extend_from_slice(self);
517    }
518}

520impl Encode for &str {
521    fn encode(&self, dst: &mut Encoder) {

532impl<T: Encode> Encode for Vec<T> {
533    fn encode(&self, dst: &mut Encoder) {
534        self.len().encode(dst);
535        for item in self {

541impl<T: Encode> Encode for Option<T> {
542    fn encode(&self, dst: &mut Encoder) {
543        match self {
544            None => dst.byte(0),
545            Some(val) => {
546                dst.byte(1);
547                val.encode(dst)

553macro_rules! encode_struct {
554    ($name:ident ($($lt:tt)*) $($field:ident: $ty:ty,)*) => {
555        struct $name $($lt)* {
556            $($field: $ty,)*
557        }
558
559        impl $($lt)* Encode for $name $($lt)* {
560            fn encode(&self, _dst: &mut Encoder) {
561                $(self.$field.encode(_dst);)*
562            }
563        }
564    }
565}

567macro_rules! encode_enum {
568    ($name:ident ($($lt:tt)*) $($fields:tt)*) => (
569        enum $name $($lt)* { $($fields)* }
570
571        impl$($lt)* Encode for $name $($lt)* {
572            fn encode(&self, dst: &mut Encoder) {
573                use self::$name::*;
574                encode_enum!(@arms self dst (0) () $($fields)*)
575            }
576        }
577    );
578
579    (@arms $me:ident $dst:ident ($cnt:expr) ($($arms:tt)*)) => (
580        encode_enum!(@expr match $me { $($arms)* })
581    );
582
583    (@arms $me:ident $dst:ident ($cnt:expr) ($($arms:tt)*) $name:ident, $($rest:tt)*) => (
584        encode_enum!(
585            @arms
586            $me
587            $dst
588            ($cnt+1)
589            ($($arms)* $name => $dst.byte($cnt),)
590            $($rest)*
591        )
592    );
593
594    (@arms $me:ident $dst:ident ($cnt:expr) ($($arms:tt)*) $name:ident($t:ty), $($rest:tt)*) => (
595        encode_enum!(
596            @arms
597            $me
598            $dst
599            ($cnt+1)
600            ($($arms)* $name(val) => { $dst.byte($cnt); val.encode($dst) })
601            $($rest)*
602        )
603    );
604
605    (@expr $e:expr) => ($e);
606}

608macro_rules! encode_api {
609    () => ();
610    (struct $name:ident<'a> { $($fields:tt)* } $($rest:tt)*) => (
611        encode_struct!($name (<'a>) $($fields)*);
612        encode_api!($($rest)*);
613    );
614    (struct $name:ident { $($fields:tt)* } $($rest:tt)*) => (
615        encode_struct!($name () $($fields)*);
616        encode_api!($($rest)*);
617    );
618    (enum $name:ident<'a> { $($variants:tt)* } $($rest:tt)*) => (
619        encode_enum!($name (<'a>) $($variants)*);
620        encode_api!($($rest)*);
621    );
622    (enum $name:ident { $($variants:tt)* } $($rest:tt)*) => (
623        encode_enum!($name () $($variants)*);
624        encode_api!($($rest)*);
625    );
626}

629fn from_ast_method_kind<'a>(
630    function: &'a ast::Function,
631    intern: &'a Interner,
632    method_kind: &'a ast::MethodKind,
633) -> Result<MethodKind<'a>, Diagnostic> {
634    Ok(match method_kind {
635        ast::MethodKind::Constructor => MethodKind::Constructor,
636        ast::MethodKind::Operation(ast::Operation { is_static, kind }) => {
637            let is_static = *is_static;
638            let kind = match kind {
639                ast::OperationKind::Getter(g) => {
640                    let g = g.as_ref().map(|g| intern.intern_str(g));
641                    OperationKind::Getter(g.unwrap_or_else(|| function.infer_getter_property()))
642                }
643                ast::OperationKind::Regular => OperationKind::Regular,
644                ast::OperationKind::RegularThis => OperationKind::RegularThis,
645                ast::OperationKind::Setter(s) => {
646                    let s = s.as_ref().map(|s| intern.intern_str(s));
647                    OperationKind::Setter(match s {
648                        Some(s) => s,
649                        None => intern.intern_str(&function.infer_setter_property()?),
650                    })
651                }
652                ast::OperationKind::IndexingGetter => OperationKind::IndexingGetter,
653                ast::OperationKind::IndexingSetter => OperationKind::IndexingSetter,
654                ast::OperationKind::IndexingDeleter => OperationKind::IndexingDeleter,
655            };
656            MethodKind::Operation(Operation { is_static, kind })
657        }
658    })
