Generated is_*() methods for each variant. E.g. Color.is_red().


use strum_macros::EnumIs;

#[derive(EnumIs, Debug)]
enum Color {
    Red,
    Green { range: usize },
}

assert!(Color::Red.is_red());
assert!(Color::Green{range: 0}.is_green());
